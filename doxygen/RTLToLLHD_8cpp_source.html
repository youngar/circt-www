67 /// This is the main entrypoint for the RTL to LLHD conversion pass.
68 void RTLToLLHDPass::runOnOperation() {
69  MLIRContext &context = getContext();
70  ModuleOp module = getOperation();
71 
72  ConversionTarget target(context);
73  target.addLegalDialect<LLHDDialect>();
74  target.addIllegalOp<RTLModuleOp>();
75 
76  RTLToLLHDTypeConverter typeConverter;
77  OwningRewritePatternList patterns;
78  populateFunctionLikeTypeConversionPattern<RTLModuleOp>(patterns, &context,
79  typeConverter);
80  patterns.insert<ConvertRTLModule>(typeConverter, &context);
81  patterns.insert<ConvertOutput>(typeConverter, &context);
82 
83  if (failed(applyPartialConversion(module, target, std::move(patterns))))
84  signalPassFailure();
85 }

99 /// This works on each RTL module, creates corresponding entities, moves the
100 /// bodies of the modules into the entities, and converts the bodies.
101 struct ConvertRTLModule : public OpConversionPattern<RTLModuleOp> {
102  using OpConversionPattern::OpConversionPattern;
103 
104  LogicalResult
105  matchAndRewrite(RTLModuleOp module, ArrayRef<Value> operands,
106  ConversionPatternRewriter &rewriter) const override {
107  // Collect the RTL module's port types.
108  FunctionType moduleType = module.getType();
109  unsigned numInputs = moduleType.getNumInputs();
110  TypeRange moduleInputs = moduleType.getInputs();
111  TypeRange moduleOutputs = moduleType.getResults();
112 
113  // LLHD entities port types are all expressed as block arguments to the op,
114  // so collect all of the types in the expected order (inputs then outputs).
115  SmallVector<Type, 4> entityTypes(moduleInputs);
116  entityTypes.append(moduleOutputs.begin(), moduleOutputs.end());
117 
118  // Ensure the input and output types have all been converted already. This
119  // is handled separately by the upstream FunctionLikeTypeConversionPattern.
120  if (!llvm::all_of(entityTypes,
121  [](Type type) { return type.isa<SigType>(); }))
122  return rewriter.notifyMatchFailure(module, "Not all ports had SigType");
123 
124  // Create the entity.
125  auto entity = rewriter.create<EntityOp>(module.getLoc(), numInputs);
126 
127  // Inline the RTL module body into the entity body.
128  Region &entityBodyRegion = entity.getBodyRegion();
129  rewriter.inlineRegionBefore(module.getBodyRegion(), entityBodyRegion,
130  entityBodyRegion.end());
131 
132  // Add the LLHD terminator op after the RTL module's output ops.
133  rewriter.setInsertionPointToEnd(entity.getBodyBlock());
134  rewriter.create<llhd::TerminatorOp>(entity.getLoc());
135 
136  // Set the entity type and name attributes. Add block arguments for each
137  // output, since LLHD entity outputs are still block arguments to the op.
138  auto entityType = rewriter.getFunctionType(entityTypes, {});
139  rewriter.updateRootInPlace(entity, [&] {
140  entity->setAttr(entity.getTypeAttrName(), TypeAttr::get(entityType));
141  entity.setName(module.getName());
142  entityBodyRegion.addArguments(moduleOutputs);
143  });
144 
145  // Erase the RTL module.
146  rewriter.eraseOp(module);
147 
148  return success();
149  }
150 };

152 /// This works on each output op, creating ops to drive the appropriate results.
153 struct ConvertOutput : public OpConversionPattern<OutputOp> {
154  using OpConversionPattern::OpConversionPattern;
155 
156  LogicalResult
157  matchAndRewrite(OutputOp output, ArrayRef<Value> operands,
158  ConversionPatternRewriter &rewriter) const override {
159  // Construct the `1d` time value for the drive.
160  auto timeType = TimeType::get(rewriter.getContext());
161  auto deltaAttr = TimeAttr::get(timeType, {0, 1, 0}, "ns");
162  auto delta = rewriter.create<ConstOp>(output.getLoc(), timeType, deltaAttr);
163 
164  // Get the number of inputs in the entity to offset into the block args.
165  auto entity = output->getParentOfType<EntityOp>();
166  size_t numInputs = entity.ins();
167 
168  // Drive the results from the mapped operands.
169  for (size_t i = 0, e = operands.size(); i != e; ++i) {
170  // Get the source and destination signals.
171  auto src = operands[i];
172  auto dest = entity.getArgument(numInputs + i);
173  assert(src && dest && "output operand must map to result block arg");
174 
175  // If the source has a signal type, probe it.
176  if (auto sigTy = src.getType().dyn_cast<SigType>())
177  src = rewriter.create<PrbOp>(output.getLoc(), sigTy.getUnderlyingType(),
178  src);
179 
180  // Drive the destination block argument value.
181  rewriter.create<DrvOp>(output.getLoc(), dest, src, delta, Value());
182  }
183 
184  // Erase the original output terminator.
185  rewriter.eraseOp(output);
186 
187  return success();
188  }
189 };
