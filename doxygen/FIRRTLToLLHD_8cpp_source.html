72 /// Run the FIRRTL to LLHD conversion pass.
73 void FIRRTLToLLHDPass::runOnOperation() {
74  for (auto &op :
75  llvm::make_early_inc_range(getOperation().getBody()->getOperations())) {
76  if (auto circuit = dyn_cast<firrtl::CircuitOp>(op)) {
77  convertCircuit(circuit);
78  circuit.getOperation()->dropAllReferences();
79  circuit.getOperation()->dropAllDefinedValueUses();
80  circuit.getOperation()->erase();
81  }
82  }
83 }

85 /// Convert an entire FIRRTL circuit.
86 void FIRRTLToLLHDPass::convertCircuit(firrtl::CircuitOp &circuit) {
87  LLVM_DEBUG(llvm::dbgs() << "Converting FIRRTL circuit `" << circuit.name()
88  << "` to LLHD\n");
89 
90  // Setup a builder which we use to emit LLHD ops.
91  OpBuilder theBuilder(&getContext());
92 
93  // Convert each module separately.
94  builder = &theBuilder;
95  for (auto &op : circuit.getBody()->getOperations()) {
96  if (auto module = dyn_cast<firrtl::FModuleOp>(op)) {
97  builder->setInsertionPointAfter(circuit);
98  convertModule(module);
99  } else if (!isa<firrtl::DoneOp>(op)) {
100  op.emitError("expected `firrtl.module` or `firrtl.done`");
101  signalPassFailure();
102  }
103  }
104  builder = nullptr;
105 }

107 /// Convert a single FIRRTL module.
108 void FIRRTLToLLHDPass::convertModule(firrtl::FModuleOp &module) {
109  LLVM_DEBUG(llvm::dbgs() << "Converting FIRRTL module `" << module.getName()
110  << "` to LLHD\n");
111 
112  // Map the potentially complex FIRRTL module ports to LLHD entity inputs and
113  // outputs. This will become fairly involved, since the nested nature of flips
114  // and bundle types requires refactoring of the ports.
115  SmallVector<firrtl::ModulePortInfo, 4> modulePorts;
116  module.getPortInfo(modulePorts);
117 
118  SmallVector<Type, 4> ins;
119  SmallVector<Type, 4> outs;
120  SmallVector<StringAttr, 4> inNames;
121  SmallVector<StringAttr, 4> outNames;
122  SmallVector<unsigned, 4> inIndices;
123  SmallVector<unsigned, 4> outIndices;
124  for (unsigned i = 0; i < modulePorts.size(); i++) {
125  auto &port = modulePorts[i];
126  LLVM_DEBUG(llvm::dbgs()
127  << "Port " << port.name << " of type " << port.type << "\n");
128 
129  // For now, let's do a simple approach where we only support flip at the top
130  // of a port's aggregate type.
131  bool isFlip = false;
132  firrtl::FIRRTLType type = port.type;
133  if (auto flipType = type.dyn_cast<firrtl::FlipType>()) {
134  isFlip = true;
135  type = flipType.getElementType();
136  }
137 
138  // Convert the type. We keep things simple for the time being.
139  auto width = type.getBitWidthOrSentinel();
140  if (width < 0) {
141  module.emitError() << "port " << port.name << " has unsupported type "
142  << port.type;
143  signalPassFailure();
144  continue;
145  }
146  auto convType = SigType::get(builder->getIntegerType(width));
147 
148  // Add to the list of inputs or outputs, depending on flip state.
149  if (isFlip) {
150  outs.push_back(convType);
151  outNames.push_back(port.name);
152  outIndices.push_back(i);
153  } else {
154  ins.push_back(convType);
155  inNames.push_back(port.name);
156  inIndices.push_back(i);
157  }
158  }
159 
160  // Concatenate inputs and outputs and mark the split point for the entity.
161  // Then assemble the entity signature type.
162  auto numIns = ins.size();
163  ins.append(outs.begin(), outs.end());
164  inNames.append(outNames.begin(), outNames.end());
165  inIndices.append(outIndices.begin(), outIndices.end());
166  auto entityType = builder->getFunctionType(ins, llvm::None);
167 
168  // Create an LLHD entity for this module.
169  auto entity = builder->create<EntityOp>(module.getLoc(), numIns);
170  entity.setName(module.getName());
171  entity->setAttr("type", TypeAttr::get(entityType));
172  EntityOp::ensureTerminator(entity.body(), *builder, entity.getLoc());
173 
174  // Populate the arguments for the entity. This includes initializing the value
175  // mapping table with correspondences of the FIRRTL module arguments to the
176  // LLHD entity inputs/outputs. These are likely to be in a different order,
177  // such that we use the indirection indices gathered above.
178  auto args = entity.body().addArguments(ins);
179  for (auto arg : llvm::zip(args, inIndices)) {
180  auto firrtlArg = module.getBodyBlock()->getArgument(std::get<1>(arg));
181  auto llhdArg = std::get<0>(arg);
182  LLVM_DEBUG(llvm::dbgs()
183  << "Map FIRRTL port " << firrtlArg.getArgNumber()
184  << " to LLHD port " << llhdArg.getArgNumber() << "\n");
185  valueMapping[firrtlArg] = llhdArg;
186  }
187 
188  // Populate the entity.
189  builder->setInsertionPoint(entity.body().front().getTerminator());
190  for (auto &op : module.getBodyBlock()->getOperations()) {
191  if (failed(dispatchVisitor(&op)))
192  signalPassFailure();
193  }
194 }

200 /// Get the LLHD value that has been previously emitted for a FIRRTL operand.
201 Value FIRRTLToLLHDPass::getConvertedValue(Value operand) {
202  auto it = valueMapping.find(operand);
203  assert(it != valueMapping.end() && "operand has not been converted to LLHD");
204  return it->second;
205 }

207 /// Get the LLHD value that has been previously emitted for a FIRRTL operand,
208 /// and implicitly extend its type as appropriate.
209 Value FIRRTLToLLHDPass::getConvertedAndExtendedValue(Value operand,
210  Type destType) {
211  if (operand.getType().cast<firrtl::FIRRTLType>().getPassiveType() !=
212  destType.cast<firrtl::FIRRTLType>().getPassiveType()) {
213  emitError(operand.getLoc(), "implicit extension not implemented");
214  return {};
215  }
216  return getConvertedValue(operand);
217 }

233 LogicalResult FIRRTLToLLHDPass::visitStmt(firrtl::ConnectOp op) {
234  LLVM_DEBUG(llvm::dbgs() << "Converting " << op << "\n");
235 
236  auto dst = getConvertedValue(op.dest());
237  auto src = getConvertedAndExtendedValue(op.src(), op.dest().getType());
238  if (!dst || !src)
239  return failure();
240 
241  LLVM_DEBUG(llvm::dbgs() << "Assigning " << src.getType() << " to "
242  << dst.getType() << "\n");
243 
244  // We must be connecting to a signal.
245  assert(dst.getType().isa<SigType>() && "must connect to signal");
246 
247  // If we are connecting two signals, e.g. an input to an output port, use an
248  // `llhd.con` operation. Otherwise use `llhd.drv` to drive the RHS onto the
249  // signal of the LHS.
250  if (auto sigTy = src.getType().dyn_cast<SigType>()) {
251  src = builder->create<PrbOp>(op.getLoc(), sigTy.getUnderlyingType(), src);
252  }
253 
254  // Construct the `1d` time value for the drive.
255  auto timeType = TimeType::get(&getContext());
256  auto deltaAttr =
257  TimeAttr::get(timeType, std::array<unsigned, 3>{0, 1, 0}, "s");
258  auto delta = builder->create<ConstOp>(op.getLoc(), timeType, deltaAttr);
259 
260  // Construct a constant one for the drive condition.
261  auto boolType = builder->getIntegerType(1);
262  auto constOneAttr = IntegerAttr::get(boolType, 1);
263  auto constOne = builder->create<ConstOp>(op.getLoc(), boolType, constOneAttr);
264 
265  // Emit the drive operation.
266  builder->create<DrvOp>(op.getLoc(), dst, src, delta, constOne);
267  return success();
268 }
