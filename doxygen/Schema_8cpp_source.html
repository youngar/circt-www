120 /// Return the number of bits used by a Capnp primitive type.
121 static size_t bits(::capnp::schema::Type::Reader type) {
122  using ty = ::capnp::schema::Type;
123  switch (type.which()) {
124  case ty::VOID:
125  return 0;
126  case ty::BOOL:
127  return 1;
128  case ty::UINT8:
129  case ty::INT8:
130  return 8;
131  case ty::UINT16:
132  case ty::INT16:
133  return 16;
134  case ty::UINT32:
135  case ty::INT32:
136  return 32;
137  case ty::UINT64:
138  case ty::INT64:
139  return 64;
140  default:
141  assert(false && "Type not yet supported");
142  }
143 }

153 /// Write a valid capnp schema to memory, then parse it out of memory using the
154 /// capnp library. Writing and parsing text within a single process is ugly, but
155 /// this is by far the easiest way to do this. This isn't the use case for which
156 /// Cap'nProto was designed.
157 ::capnp::ParsedSchema TypeSchemaImpl::getSchema() const {
158  if (rootSchema != ::capnp::ParsedSchema())
159  return rootSchema;
160 
161  // Write the schema to `schemaText`.
162  std::string schemaText;
163  llvm::raw_string_ostream os(schemaText);
164  emitId(os, 0xFFFFFFFFFFFFFFFF) << ";\n";
165  auto rc = write(os);
166  assert(succeeded(rc) && "Failed schema text output.");
167  os.str();
168 
169  // Write `schemaText` to an in-memory filesystem then parse it. Yes, this is
170  // the only way to do this.
171  kj::Own<kj::Filesystem> fs = kj::newDiskFilesystem();
172  kj::Own<kj::Directory> dir = kj::newInMemoryDirectory(kj::nullClock());
173  kj::Path fakePath = kj::Path::parse("schema.capnp");
174  { // Ensure that 'fakeFile' has flushed.
175  auto fakeFile = dir->openFile(fakePath, kj::WriteMode::CREATE);
176  fakeFile->writeAll(schemaText);
177  }
178  rootSchema = parser.parseFromDirectory(*dir, std::move(fakePath), nullptr);
179  return rootSchema;
180 }

196 // We compute a deterministic hash based on the type. Since llvm::hash_value
197 // changes from execution to execution, we don't use it.
198 uint64_t TypeSchemaImpl::capnpTypeID() const {
199  if (cachedID)
200  return *cachedID;
201 
202  // Get the MLIR asm type, padded to a multiple of 64 bytes.
203  std::string typeName;
204  llvm::raw_string_ostream osName(typeName);
205  osName << type;
206  size_t overhang = osName.tell() % 64;
207  if (overhang != 0)
208  osName.indent(64 - overhang);
209  osName.flush();
210  const char *typeNameC = typeName.c_str();
211 
212  uint64_t hash = esiCosimSchemaVersion;
213  for (size_t i = 0, e = typeName.length() / 64; i < e; ++i)
214  hash =
215  llvm::hashing::detail::hash_33to64_bytes(&typeNameC[i * 64], 64, hash);
216 
217  // Capnp IDs always have a '1' high bit.
218  cachedID = hash | 0x8000000000000000;
219  return *cachedID;
220 }

234 /// Returns the expected size of an array (capnp list) in 64-bit words.
235 static int64_t size(rtl::ArrayType mType, capnp::schema::Field::Reader cField) {
236  assert(cField.isSlot());
237  auto cType = cField.getSlot().getType();
238  assert(cType.isList());
239  size_t elementBits = bits(cType.getList().getElementType());
240  int64_t listBits = mType.getSize() * elementBits;
241  return llvm::divideCeil(listBits, 64);
242 }

244 /// Compute the size of a capnp struct, in 64-bit words.
245 static int64_t size(capnp::schema::Node::Struct::Reader cStruct,
246  ArrayRef<Type> mFields) {
247  using namespace capnp::schema;
248  int64_t size = (1 + // Header
249  cStruct.getDataWordCount() + cStruct.getPointerCount());
250  auto cFields = cStruct.getFields();
251  for (Field::Reader cField : cFields) {
252  assert(!cField.isGroup() && "Capnp groups are not supported");
253  // Capnp code order is the index in the the MLIR fields array.
254  assert(cField.getCodeOrder() < mFields.size());
255 
256  // The size of the thing to which the pointer is pointing, not the size of
257  // the pointer itself.
258  int64_t pointedToSize =
259  TypeSwitch<mlir::Type, int64_t>(mFields[cField.getCodeOrder()])
260  .Case([](IntegerType) { return 0; })
261  .Case([cField](rtl::ArrayType mType) {
262  return ::size(mType, cField);
263  });
264  size += pointedToSize;
265  }
266  return size; // Convert from 64-bit words to bits.
267 }

276 /// Write a valid Capnp name for 'type'.
277 static void emitName(Type type, llvm::raw_ostream &os) {
278  llvm::TypeSwitch<Type>(type)
279  .Case([&os](IntegerType intTy) {
280  std::string intName;
281  llvm::raw_string_ostream(intName) << intTy;
282  // Capnp struct names must start with an uppercase character.
283  intName[0] = toupper(intName[0]);
284  os << intName;
285  })
286  .Case([&os](rtl::ArrayType arrTy) {
287  os << "ArrayOf" << arrTy.getSize() << 'x';
288  emitName(arrTy.getElementType(), os);
289  })
290  .Default([](Type) {
291  assert(false && "Type not supported. Please check support first with "
292  "isSupported()");
293  });
294 }

307 /// Write a valid Capnp type.
308 static void emitCapnpType(Type type, IndentingOStream &os) {
309  llvm::TypeSwitch<Type>(type)
310  .Case([&os](IntegerType intTy) {
311  auto w = intTy.getWidth();
312  if (w == 0) {
313  os.indent() << "Void";
314  } else if (w == 1) {
315  os.indent() << "Bool";
316  } else {
317  if (intTy.isSigned())
318  os << "Int";
319  else
320  os << "UInt";
321 
322  // Round up.
323  if (w <= 8)
324  os << "8";
325  else if (w <= 16)
326  os << "16";
327  else if (w <= 32)
328  os << "32";
329  else if (w <= 64)
330  os << "64";
331  else
332  assert(false && "Type not supported. Integer too wide. Please "
333  "check support first with isSupported()");
334  }
335  })
336  .Case([&os](rtl::ArrayType arrTy) {
337  os << "List(";
338  emitCapnpType(arrTy.getElementType(), os);
339  os << ')';
340  })
341  .Default([](Type) {
342  assert(false && "Type not supported. Please check support first with "
343  "isSupported()");
344  });
345 }

347 /// This function is essentially a placeholder which only supports ints. It'll
348 /// need to be re-worked when we start supporting structs, arrays, unions,
349 /// enums, etc.
350 LogicalResult TypeSchemaImpl::write(llvm::raw_ostream &rawOS) const {
351  IndentingOStream os(rawOS);
352 
353  // Since capnp requires messages to be structs, emit a wrapper struct.
354  os.indent() << "struct ";
355  writeMetadata(rawOS);
356  os << " {\n";
357  os.addIndent();
358 
359  // Specify the actual type, followed by the capnp field.
360  os.indent() << "# Actual type is " << type << ".\n";
361  os.indent() << "i @0 :";
362  emitCapnpType(type, os);
363  os << ";\n";
364 
365  os.reduceIndent();
366  os.indent() << "}\n\n";
367  return success();
368 }

385 /// Contains helper methods to assist with naming and casting.
386 struct GasketComponent {
387 public:
388  GasketComponent(OpBuilder &b, Value init) : builder(&b), s(init) {}
389 
390  /// Set the "name" attribute of a value's op.
391  template <typename T = GasketComponent>
392  T &name(const Twine &name) {
393  std::string nameStr = name.str();
394  if (nameStr.empty())
395  return *(T *)this;
396  auto nameAttr = StringAttr::get(nameStr, ctxt());
397  s.getDefiningOp()->setAttr("name", nameAttr);
398  return *(T *)this;
399  }
400  template <typename T = GasketComponent>
401  T &name(capnp::Text::Reader fieldName, const Twine &nameSuffix) {
402  return name<T>(StringRef(fieldName.cStr()) + nameSuffix);
403  }
404 
405  /// Construct a bitcast.
406  GasketComponent cast(Type t) {
407  auto dst = builder->create<rtl::BitcastOp>(loc(), t, s);
408  return GasketComponent(*builder, dst);
409  }
410 
411  /// Downcast an int, accounting for signedness.
412  GasketComponent downcast(IntegerType t) {
413  // Since the RTL dialect operators only operate on signless integers, we
414  // have to cast to signless first, then cast the sign back.
415  assert(s.getType().isa<IntegerType>());
416  Value signlessVal = s;
417  if (!signlessVal.getType().isSignlessInteger())
418  signlessVal = builder->create<rtl::BitcastOp>(
419  loc(), builder->getIntegerType(s.getType().getIntOrFloatBitWidth()),
420  s);
421 
422  if (!t.isSigned()) {
423  auto extracted =
424  builder->create<rtl::ExtractOp>(loc(), t, signlessVal, 0);
425  return GasketComponent(*builder, extracted).cast(t);
426  }
427  auto magnitude = builder->create<rtl::ExtractOp>(
428  loc(), builder->getIntegerType(t.getWidth() - 1), signlessVal, 0);
429  auto sign = builder->create<rtl::ExtractOp>(
430  loc(), builder->getIntegerType(1), signlessVal, t.getWidth() - 1);
431  auto result = builder->create<rtl::ConcatOp>(loc(), sign, magnitude);
432 
433  // We still have to cast to handle signedness.
434  return GasketComponent(*builder, result).cast(t);
435  }
436 
437  Operation *operator->() const { return s.getDefiningOp(); }
438  Value getValue() const { return s; }
439  Location loc() const { return s.getLoc(); }
440  OpBuilder &b() const { return *builder; }
441  MLIRContext *ctxt() const { return builder->getContext(); }
442  operator Value() { return s; }
443 
444 protected:
445  OpBuilder *builder;
446  Value s;
447 };
448 } // anonymous namespace

451 /// Holds a 'slice' of an array and is able to construct more slice ops, then
452 /// cast to a type. A sub-slice holds a pointer to the slice which created it,
453 /// so it forms a hierarchy. This is so we can easily track offsets from the
454 /// root message for pointer resolution.
455 ///
456 /// Requirement: any slice which has sub-slices must not be free'd before its
457 /// children slices.
458 struct Slice : public GasketComponent {
459 private:
460  Slice(Slice *parent, llvm::Optional<int64_t> offset, Value val)
461  : GasketComponent(*parent->builder, val), parent(parent),
462  offsetIntoParent(offset) {
463  type = val.getType().dyn_cast<rtl::ArrayType>();
464  assert(type && "Value must be array type");
465  }
466 
467 public:
468  Slice(OpBuilder &b, Value val)
469  : GasketComponent(b, val), parent(nullptr), offsetIntoParent(0) {
470  type = val.getType().dyn_cast<rtl::ArrayType>();
471  assert(type && "Value must be array type");
472  }
473 
474  /// Create an op to slice the array from lsb to lsb + size. Return a new slice
475  /// with that op.
476  Slice slice(int64_t lsb, int64_t size) {
477  Value newSlice = builder->create<rtl::ArraySliceOp>(loc(), s, lsb, size);
478  return Slice(this, lsb, newSlice);
479  }
480 
481  /// Create an op to slice the array from lsb to lsb + size. Return a new slice
482  /// with that op. If lsb is greater width thn necessary, lop off the high
483  /// bits.
484  Slice slice(Value lsb, int64_t size) {
485  assert(lsb.getType().isa<IntegerType>());
486 
487  unsigned expIdxWidth = llvm::Log2_64_Ceil(type.getSize());
488  int64_t lsbWidth = lsb.getType().getIntOrFloatBitWidth();
489  if (lsbWidth > expIdxWidth)
490  lsb = builder->create<rtl::ExtractOp>(
491  loc(), builder->getIntegerType(expIdxWidth), lsb, 0);
492  else if (lsbWidth < expIdxWidth)
493  assert(false && "LSB Value must not be smaller than expected.");
494  auto dstTy = rtl::ArrayType::get(type.getElementType(), size);
495  Value newSlice = builder->create<rtl::ArraySliceOp>(loc(), dstTy, s, lsb);
496  return Slice(this, llvm::Optional<int64_t>(), newSlice);
497  }
498  Slice &name(const Twine &name) { return GasketComponent::name<Slice>(name); }
499  Slice &name(capnp::Text::Reader fieldName, const Twine &nameSuffix) {
500  return GasketComponent::name<Slice>(fieldName.cStr(), nameSuffix);
501  }
502  Slice castToSlice(Type elemTy, size_t size, StringRef name = StringRef(),
503  Twine nameSuffix = Twine()) {
504  auto arrTy = rtl::ArrayType::get(elemTy, size);
505  GasketComponent rawCast =
506  GasketComponent::cast(arrTy).name(name + nameSuffix);
507  return Slice(*builder, rawCast);
508  }
509 
510  GasketComponent operator[](Value idx) {
511  return GasketComponent(*builder,
512  builder->create<rtl::ArrayGetOp>(loc(), s, idx));
513  }
514 
515  GasketComponent operator[](size_t idx) {
516  IntegerType idxTy =
517  builder->getIntegerType(llvm::Log2_32_Ceil(type.getSize()));
518  auto idxVal = builder->create<rtl::ConstantOp>(loc(), idxTy, idx);
519  return GasketComponent(*builder,
520  builder->create<rtl::ArrayGetOp>(loc(), s, idxVal));
521  }
522 
523  /// Return the root of this slice hierarchy.
524  const Slice &getRootSlice() {
525  if (parent == nullptr)
526  return *this;
527  return parent->getRootSlice();
528  }
529 
530  llvm::Optional<int64_t> getOffsetFromRoot() {
531  if (parent == nullptr)
532  return 0;
533  auto parentOffset = parent->getOffsetFromRoot();
534  if (!offsetIntoParent || !parentOffset)
535  return llvm::Optional<int64_t>();
536  return *offsetIntoParent + *parentOffset;
537  }
538 
539 private:
540  rtl::ArrayType type;
541  Slice *parent;
542  llvm::Optional<int64_t> offsetIntoParent;
543 };

547 /// Utility class for building sv::AssertOps. Since SV assertions need to be in

554  void assertPred(GasketComponent veg, ICmpPredicate pred, int64_t expected) {
555  if (veg.getValue().getType().isa<IntegerType>()) {
556  assertPred(veg.getValue(), pred, expected);
557  return;
558  }
559 
560  auto valTy = veg.getValue().getType().dyn_cast<rtl::ArrayType>();

565  }

588 /// Build an RTL/SV dialect capnp encoder for this type. Inputs need to be
589 /// packed on unpadded.
590 Value TypeSchemaImpl::buildEncoder(OpBuilder &b, Value clk, Value valid,
591  Value operand) {
592  MLIRContext *ctxt = b.getContext();
593  auto loc = operand.getDefiningOp()->getLoc();
594  ::capnp::schema::Node::Reader rootProto = getTypeSchema().getProto();
595 
596  auto i16 = b.getIntegerType(16);
597  auto i32 = b.getIntegerType(32);
598 
599  auto typeAndOffset = b.create<rtl::ConstantOp>(loc, i32, 0);
600  auto ptrSize = b.create<rtl::ConstantOp>(loc, i16, 0);
601  auto dataSize = b.create<rtl::ConstantOp>(
602  loc, i16, rootProto.getStruct().getDataWordCount());
603  auto structPtr = b.create<rtl::ConcatOp>(
604  loc, ValueRange{ptrSize, dataSize, typeAndOffset});
605 
606  auto operandIntTy = operand.getType().cast<IntegerType>();
607  uint16_t paddingBits =
608  rootProto.getStruct().getDataWordCount() * 64 - operandIntTy.getWidth();
609  auto operandCasted = b.create<rtl::BitcastOp>(
610  loc,
611  IntegerType::get(ctxt, operandIntTy.getWidth(), IntegerType::Signless),
612  operand);
613 
614  IntegerType iPaddingTy = IntegerType::get(ctxt, paddingBits);
615  auto padding = b.create<rtl::ConstantOp>(loc, iPaddingTy, 0);
616  auto dataSection =
617  b.create<rtl::ConcatOp>(loc, ValueRange{padding, operandCasted});
618 
619  return b.create<rtl::ConcatOp>(loc, ValueRange{dataSection, structPtr});
620 }

622 /// Construct the proper operations to decode a capnp list. This only works for
623 /// arrays of ints or bools. Will need to be updated for structs and lists of

625 static GasketComponent decodeList(rtl::ArrayType type,

629  assert(capnpType.isList());
630  assert(capnpType.getList().hasElementType());
631 
632  auto loc = ptrSection.loc();
633  OpBuilder &b = ptrSection.b();
634 
635  // Get the list pointer and break out its parts.
636  auto ptr = ptrSection.slice(field.getSlot().getOffset() * 64, 64)
637  .name(field.getName(), "_ptr");
638  auto ptrType = ptr.slice(0, 2);
639  auto offset = ptr.slice(2, 30)
640  .cast(b.getIntegerType(30))
641  .name(field.getName(), "_offset");
642  auto elemSize = ptr.slice(31, 3);
643  auto length = ptr.slice(34, 29);
644 
645  // Assert that ptr type == list type;
646  asserts.assertEqual(ptrType, 0);
647 
648  // Assert that the element size in the message matches our expectation.
649  auto expectedElemSizeBits = bits(capnpType.getList().getElementType());
650  unsigned expectedElemSizeField;
651  switch (expectedElemSizeBits) {
652  case 0:
653  expectedElemSizeField = 0;
654  break;
655  case 1:
656  expectedElemSizeField = 1;
657  break;
658  case 8:
659  expectedElemSizeField = 2;
660  break;
661  case 16:
662  expectedElemSizeField = 3;
663  break;
664  case 32:
665  expectedElemSizeField = 4;
666  break;
667  case 64:
668  expectedElemSizeField = 5;
669  break;
670  default:
671  assert(false && "bits() returned unexpected value");
672  }
673  asserts.assertEqual(elemSize, expectedElemSizeField);
674 
675  // Assert that the length of the list (array) is at most the length of the
676  // array.
677  auto maxWords = (type.getSize() * expectedElemSizeBits) / (8 * 64);
678  asserts.assertPred(length, ICmpPredicate::sle, maxWords);
679 
680  // Get the entire message slice, compute the offset into the list, then get
681  // the list data in an ArrayType.
682  auto msg = ptr.getRootSlice();
683  auto ptrOffset = ptr.getOffsetFromRoot();
684  assert(ptrOffset);
685  auto listOffset = b.create<rtl::AddOp>(
686  loc, offset,
687  b.create<rtl::ConstantOp>(loc, b.getIntegerType(30), *ptrOffset + 64));
688  auto listSlice =
689  msg.slice(listOffset.getResult(), type.getSize() * expectedElemSizeBits);
690 
691  // Cast to an array of capnp int elements.
692  assert(type.getElementType().isa<IntegerType>() &&
693  "DecodeList() only works on arrays of ints currently");
694  Type capnpElemTy =
695  b.getIntegerType(expectedElemSizeBits, IntegerType::Signless);
696  auto arrayOfElements = listSlice.castToSlice(capnpElemTy, type.getSize());
697  if (arrayOfElements.getValue().getType() == type)
698  return arrayOfElements;
699 
700  // Collect the reduced elements.
701  SmallVector<Value, 64> arrayValues;
702  for (size_t i = 0, e = type.getSize(); i < e; ++i) {
703  auto capnpElem = arrayOfElements[i].name(field.getName(), "_capnp_elem");
704  auto esiElem = capnpElem.downcast(type.getElementType().cast<IntegerType>())
705  .name(field.getName(), "_elem");
706  arrayValues.push_back(esiElem);
707  }
708  auto array = b.create<rtl::ArrayCreateOp>(loc, arrayValues);
709  return GasketComponent(b, array);
710 }

712 /// Construct the proper operations to convert a capnp field to 'type'.
713 static GasketComponent decodeField(Type type,
714  capnp::schema::Field::Reader field,
715  Slice dataSection, Slice ptrSection,
716  AssertBuilder &asserts) {
717  GasketComponent esiValue =
718  TypeSwitch<Type, GasketComponent>(type)
719  .Case([&](IntegerType it) {
720  auto slice = dataSection.slice(field.getSlot().getOffset() *
721  bits(field.getSlot().getType()),
722  it.getWidth());
723  return slice.name(field.getName(), "_bits").cast(type);
724  })
725  .Case([&](rtl::ArrayType at) {
726  return decodeList(at, field, ptrSection, asserts);
727  });
728  esiValue.name(field.getName().cStr(), "Value");
729  return esiValue;
730 }

732 /// Build an RTL/SV dialect capnp decoder for this type. Outputs packed and
733 /// unpadded data.
734 Value TypeSchemaImpl::buildDecoder(OpBuilder &b, Value clk, Value valid,
735  Value operandVal) {
736  // Various useful integer types.
737  auto i16 = b.getIntegerType(16);
738 
739  size_t size = this->size();
740  rtl::ArrayType operandType = operandVal.getType().dyn_cast<rtl::ArrayType>();
741  assert(operandType && operandType.getSize() == size &&
742  "Operand type and length must match the type's capnp size.");
743 
744  Slice operand(b, operandVal);
745  auto loc = operand.loc();
746 
747  auto alwaysAt = b.create<sv::AlwaysOp>(loc, EventControl::AtPosEdge, clk);
748  auto ifValid =
749  OpBuilder(alwaysAt.getBodyRegion()).create<sv::IfOp>(loc, valid);
750  AssertBuilder asserts(loc, ifValid.getBodyRegion());
751 
752  // The next 64-bits of a capnp message is the root struct pointer.
753  ::capnp::schema::Node::Reader rootProto = getTypeSchema().getProto();
754  auto ptr = operand.slice(0, 64).name("rootPointer");
755 
756  // Since this is the root, we _expect_ the offset to be zero but that's only
757  // guaranteed to be the case with canonically-encoded messages.
758  // TODO: support cases where the pointer offset is non-zero.
759  Slice assertPtr(ptr);
760  auto typeAndOffset = assertPtr.slice(0, 32).name("typeAndOffset");
761  asserts.assertEqual(typeAndOffset, 0);
762 
763  // We expect the data section to be equal to the computed data section size.
764  auto dataSectionSize =
765  assertPtr.slice(32, 16).cast(i16).name("dataSectionSize");
766  asserts.assertEqual(dataSectionSize,
767  rootProto.getStruct().getDataWordCount());
768 
769  // We expect the pointer section to be equal to the computed pointer section
770  // size.
771  auto ptrSectionSize =
772  assertPtr.slice(48, 16).cast(i16).name("ptrSectionSize");
773  asserts.assertEqual(ptrSectionSize, rootProto.getStruct().getPointerCount());
774 
775  // Get pointers to the data and pointer sections.
776  auto st = rootProto.getStruct();
777  auto dataSection =
778  operand.slice(64, st.getDataWordCount() * 64).name("dataSection");
779  auto ptrSection = operand
780  .slice(64 + (st.getDataWordCount() * 64),
781  rootProto.getStruct().getPointerCount() * 64)
782  .name("ptrSection");
783 
784  // Loop through fields.
785  SmallVector<GasketComponent, 64> fieldValues;
786  for (auto field : st.getFields()) {
787  uint16_t idx = field.getCodeOrder();
788  assert(idx < fieldTypes.size() && "Capnp struct longer than fieldTypes.");
789  fieldValues.push_back(
790  decodeField(fieldTypes[idx], field, dataSection, ptrSection, asserts));
791  }
792 
793  // What to return depends on the type. (e.g. structs have to be constructed
794  // from the field values.)
795  GasketComponent ret =
796  TypeSwitch<Type, GasketComponent>(type)
797  .Case([&fieldValues](IntegerType) { return fieldValues[0]; })
798  .Case([&fieldValues](rtl::ArrayType) { return fieldValues[0]; });
799  return ret;
800 }
