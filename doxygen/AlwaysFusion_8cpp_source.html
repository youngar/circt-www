27 /// Check the equivalence of operations by doing a deep comparison of operands
28 /// and attributes, but does not compare the content of any regions attached to
29 /// each op.
30 struct SimpleOperationInfo : public llvm::DenseMapInfo<Operation *> {
31  static unsigned getHashValue(const Operation *opC) {
32  return OperationEquivalence::computeHash(const_cast<Operation *>(opC));
33  }
34  static bool isEqual(const Operation *lhsC, const Operation *rhsC) {
35  auto *lhs = const_cast<Operation *>(lhsC);
36  auto *rhs = const_cast<Operation *>(rhsC);
37  if (lhs == rhs)
38  return true;
39  if (lhs == getTombstoneKey() || lhs == getEmptyKey() ||
40  rhs == getTombstoneKey() || rhs == getEmptyKey())
41  return false;
42  return OperationEquivalence::isEquivalentTo(const_cast<Operation *>(lhsC),
43  const_cast<Operation *>(rhsC));
44  }
45 };

47 // Merge two regions together. These regions must only have a one block.
48 static void mergeRegions(Region *region1, Region *region2) {
49  assert(region1->getBlocks().size() <= 1 && region2->getBlocks().size() <= 1 &&
50  "Can only merge regions with a single block");
51  if (region1->empty()) {
52  // If both regions are empty, move on to the next pair of regions
53  if (region2->empty())
54  return;
55  // If the first region has no block, move the second region's block over.
56  region1->getBlocks().splice(region1->end(), region2->getBlocks());
57  return;
58  }
59  // If the second region is not empty, splice its block into the end of the
60  // first region.
61  if (!region2->empty()) {
62  auto &block1 = region1->front();
63  auto &block2 = region2->front();
64  // Remove the terminator from the first block before merging.
65  assert(isa<YieldOp>(block1.back()) &&
66  "Block should be terminated by an sv.yield operation");
67  block1.back().erase();
68  block1.getOperations().splice(block1.end(), block2.getOperations());
69  }
70 }
71 
72 /// Inline all regions from the second operation into the first.
73 static void mergeOperations(Operation *op1, Operation *op2) {
74  assert(op1 != op2 && "Cannot merge an op into itself");
75  for (unsigned i = 0; i < op1->getNumRegions(); ++i)
76  mergeRegions(&op1->getRegion(i), &op2->getRegion(i));
77 }

79 struct AlwaysFusionPass : public AlwaysFusionBase<AlwaysFusionPass> {
80  void runOnOperation() override {
81  auto *op = getOperation();
82 
83  // Keeps track if anything changed during this pass, used to determine if
84  // the analyses were preserved.
85  bool graphChanged = false;
86 
87  auto kindInterface = dyn_cast<mlir::RegionKindInterface>(op);
88  for (unsigned i = 0; i < op->getNumRegions(); ++i) {
89  // If the operation does not implement the region kind interface, all of
90  // its regions are implicitly regular SSACFG region. Since we are blindly
91  // combining alwaysff blocks, make sure they are in a graph region.
92  if (!kindInterface ||
93  kindInterface.getRegionKind(i) == RegionKind::SSACFG)
94  continue;
95 
96  if (op->getRegion(i).empty())
97  continue;
98 
99  // Graph regions only have 1 block.
100  auto &block = op->getRegion(i).front();
101 
102  // A set of operations in the current block which are mergable. Any
103  // operation in this set is a candidate for another similar operation to
104  // merge in to.
105  DenseSet<Operation *, SimpleOperationInfo> foundOps;
106  for (sv::AlwaysFFOp alwaysOp :
107  llvm::make_early_inc_range(block.getOps<sv::AlwaysFFOp>())) {
108  // check if we have encountered an equivalent operation already. If we
109  // have, merge them together and delete the old operation.
110  auto it = foundOps.find(alwaysOp);
111  if (it == foundOps.end()) {
112  // Record new mergable ops as a candidate for merging
113  foundOps.insert(alwaysOp);
114  } else {
115  // Merge with a similar alwaysff
116  mergeOperations(*it, alwaysOp);
117  alwaysOp.erase();
118  graphChanged = true;
119  }
120  }
121  }
122 
123  // If we did not change anything in the graph mark all analysis as
124  // preserved.
125  if (!graphChanged)
126  markAllAnalysesPreserved();
127  }
128 };
