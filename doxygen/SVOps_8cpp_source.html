33 static ParseResult parseImplicitSSAName(OpAsmParser &parser,
34  NamedAttrList &resultAttrs) {
35 
36  if (parser.parseOptionalAttrDict(resultAttrs))
37  return failure();
38 
39  // If the attribute dictionary contains no 'name' attribute, infer it from
40  // the SSA name (if specified).
41  bool hadName = llvm::any_of(
42  resultAttrs, [](NamedAttribute attr) { return attr.first == "name"; });
43 
44  // If there was no name specified, check to see if there was a useful name
45  // specified in the asm file.
46  if (hadName)
47  return success();
48 
49  auto resultName = parser.getResultName(0);
50  if (!resultName.first.empty() && !isdigit(resultName.first[0])) {
51  StringRef name = resultName.first;
52  auto nameAttr = parser.getBuilder().getStringAttr(name);
53  auto *context = parser.getBuilder().getContext();
54  resultAttrs.push_back({Identifier::get("name", context), nameAttr});
55  }
56 
57  return success();
58 }

60 static void printImplicitSSAName(OpAsmPrinter &p, Operation *op,
61  DictionaryAttr attr) {
62  // Note that we only need to print the "name" attribute if the asmprinter
63  // result name disagrees with it. This can happen in strange cases, e.g.
64  // when there are conflicts.
65  bool namesDisagree = false;
66 
67  SmallString<32> resultNameStr;
68  llvm::raw_svector_ostream tmpStream(resultNameStr);
69  p.printOperand(op->getResult(0), tmpStream);
70  auto expectedName = op->getAttrOfType<StringAttr>("name");
71  if (!expectedName ||
72  tmpStream.str().drop_front() != expectedName.getValue()) {
73  namesDisagree = true;
74  }
75 
76  if (namesDisagree)
77  p.printOptionalAttrDict(op->getAttrs());
78  else
79  p.printOptionalAttrDict(op->getAttrs(), {"name"});
80 }

127  llvm::SMLoc typeLoc;

131  parser.getCurrentLocation(&typeLoc) || parser.parseType(resultType))
132  return failure();

136  // If the attribute dictionary contains no 'name' attribute, infer it from

145  return success();

147  auto resultName = parser.getResultName(0);
148  if (!resultName.first.empty() && !isdigit(resultName.first[0])) {

259 static ParseResult
260 parseEventList(OpAsmParser &p, Attribute &eventsAttr,
261  SmallVectorImpl<OpAsmParser::OperandType> &clocksOperands) {
262 
263  // Parse zero or more conditions intoevents and clocksOperands.
264  SmallVector<Attribute> events;
265 
266  auto loc = p.getCurrentLocation();
267  StringRef keyword;
268  if (!p.parseOptionalKeyword(&keyword)) {
269  while (1) {
270  auto kind = symbolizeEventControl(keyword);
271  if (!kind.hasValue())
272  return p.emitError(loc, "expected 'posedge', 'negedge', or 'edge'");
273  auto eventEnum = static_cast<int32_t>(kind.getValue());
274  events.push_back(p.getBuilder().getI32IntegerAttr(eventEnum));
275 
276  clocksOperands.push_back({});
277  if (p.parseOperand(clocksOperands.back()))
278  return failure();
279 
280  if (failed(p.parseOptionalComma()))
281  break;
282  if (p.parseKeyword(&keyword))
283  return failure();
284  }
285  }
286  eventsAttr = p.getBuilder().getArrayAttr(events);
287  return success();
288 }

404 static ParseResult parseModportStructs(OpAsmParser &parser,
405  ArrayAttr &portsAttr) {
406  if (parser.parseLParen())
407  return failure();
408 
409  auto context = parser.getBuilder().getContext();
410 
411  SmallVector<Attribute, 8> ports;
412  do {
413  StringAttr direction;
414  FlatSymbolRefAttr signal;
415  if (parser.parseAttribute(direction) || parser.parseAttribute(signal))
416  return failure();
417 
418  ports.push_back(ModportStructAttr::get(direction, signal, context));
419  } while (succeeded(parser.parseOptionalComma()));
420 
421  if (parser.parseRParen())
422  return failure();
423 
424  portsAttr = ArrayAttr::get(ports, context);
425  return success();
426 }

578 void WireOp::getCanonicalizationPatterns(OwningRewritePatternList &results,
579  MLIRContext *context) {
580  // If this wire is only written to, delete the wire and all writers.
581  struct DropDeadConnect final : public OpRewritePattern<WireOp> {
582  using OpRewritePattern::OpRewritePattern;
583  LogicalResult matchAndRewrite(WireOp op,
584  PatternRewriter &rewriter) const override {
585 
586  // Check that all operations on the wire are sv.connects. All other wire
587  // operations will have been handled by other canonicalization.
588  for (auto &use : op.getResult().getUses())
589  if (!isa<ConnectOp>(use.getOwner()))
590  return failure();
591 
592  // Remove all uses of the wire.
593  for (auto &use : op.getResult().getUses())
594  rewriter.eraseOp(use.getOwner());
595 
596  // Remove the wire.
597  rewriter.eraseOp(op);
598  return success();
599  }
600  };
601  results.insert<DropDeadConnect>(context);
602 }
