32 /// Return true if the specified type is a value RTL Integer type. This checks

34 bool circt::rtl::isRTLIntegerType(mlir::Type type) {
35  auto intType = type.dyn_cast<IntegerType>();

42 /// Return true if the specified type can be used as an RTL value type, that is
43 /// the set of types that can be composed together to represent synthesized,
44 /// hardware but not marker types like InOutType.
45 bool circt::rtl::isRTLValueType(Type type) {
46  if (auto intType = type.dyn_cast<IntegerType>())
47  return intType.isSignless();
48 
49  if (auto array = type.dyn_cast<ArrayType>())
50  return isRTLValueType(array.getElementType());
51 
52  if (auto array = type.dyn_cast<UnpackedArrayType>())
53  return isRTLValueType(array.getElementType());
54 
55  if (auto t = type.dyn_cast<StructType>()) {
56  return std::all_of(t.getElements().begin(), t.getElements().end(),
57  [](const auto &f) { return isRTLValueType(f.type); });
58  }
59 
60  return false;
61 }

63 /// Return the hardware bit width of a type. Does not reflect any encoding,
64 /// padding, or storage scheme, just the bit (and wire width) of a
65 /// statically-size type. Reflects the number of wires needed to transmit a
66 /// value of this type. Returns -1 if the type is not known or cannot be
67 /// statically computed.
68 int64_t circt::rtl::getBitWidth(mlir::Type type) {
69  return llvm::TypeSwitch<::mlir::Type, size_t>(type)
70  .Case<IntegerType>(
71  [](IntegerType t) { return t.getIntOrFloatBitWidth(); })
72  .Case<ArrayType>([](ArrayType a) {
73  int64_t elementBitWidth = getBitWidth(a.getElementType());
74  if (elementBitWidth < 0)
75  return elementBitWidth;
76  return (int64_t)a.getSize() * elementBitWidth;
77  })
78  .Case<StructType>([](StructType s) {
79  int64_t total = 0;
80  for (auto field : s.getElements()) {
81  int64_t fieldSize = getBitWidth(field.type);
82  if (fieldSize < 0)
83  return fieldSize;
84  total += fieldSize;
85  }
86  return total;
87  })
88  .Default([](Type) { return -1; });
89 }

91 /// Return true if the specified type contains known marker types like
92 /// InOutType. Unlike isRTLValueType, this is not conservative, it only returns
93 /// false on known InOut types, rather than any unknown types.
94 bool circt::rtl::hasRTLInOutType(Type type) {
95  if (auto array = type.dyn_cast<ArrayType>())
96  return hasRTLInOutType(array.getElementType());
97 
98  if (auto array = type.dyn_cast<UnpackedArrayType>())
99  return hasRTLInOutType(array.getElementType());
100 
101  if (auto t = type.dyn_cast<StructType>()) {
102  return std::any_of(t.getElements().begin(), t.getElements().end(),
103  [](const auto &f) { return hasRTLInOutType(f.type); });
104  }
105  return type.isa<InOutType>();
106 }

108 /// Parse and print nested RTL types nicely. These helper methods allow eliding
109 /// the "rtl." prefix on array, inout, and other types when in a context that
110 /// expects RTL subelement types.
111 static ParseResult parseRTLElementType(Type &result, DialectAsmParser &p) {
112  // If this is an RTL dialect type, then we don't need/want the !rtl. prefix
113  // redundantly specified.
114  auto fullString = p.getFullSymbolSpec();
115  auto *curPtr = p.getCurrentLocation().getPointer();
116  auto typeString =
117  StringRef(curPtr, fullString.size() - (curPtr - fullString.data()));
118 
119  if (typeString.startswith("array<") || typeString.startswith("inout<") ||
120  typeString.startswith("uarray<")) {
121  llvm::StringRef mnemonic;
122  if (p.parseKeyword(&mnemonic))
123  llvm_unreachable("should have an array or inout keyword here");
124  result = generatedTypeParser(p.getBuilder().getContext(), p, mnemonic);
125  return result ? success() : failure();
126  }
127 
128  return p.parseType(result);
129 }

152 Type StructType::parse(MLIRContext *ctxt, DialectAsmParser &p) {
153  llvm::SmallVector<FieldInfo, 4> parameters;
154  StringRef name;
155  if (p.parseLess())
156  return Type();
157  while (mlir::succeeded(p.parseOptionalKeyword(&name))) {
158  Type type;
159  if (p.parseColon() || p.parseType(type))
160  return Type();
161  parameters.push_back(FieldInfo{name, type});
162  if (p.parseOptionalComma())
163  break;
164  }
165  if (p.parseGreater())
166  return Type();
167  return get(ctxt, parameters);
168 }

194 Type ArrayType::parse(MLIRContext *ctxt, DialectAsmParser &p) {
195  SmallVector<int64_t, 2> dims;
196  Type inner;
197  if (p.parseLess() || p.parseDimensionList(dims, /* allowDynamic */ false) ||
198  parseRTLElementType(inner, p) || p.parseGreater())
199  return Type();
200  if (dims.size() != 1) {
201  p.emitError(p.getNameLoc(), "rtl.array only supports one dimension");
202  return Type();
203  }
204 
205  auto loc = p.getEncodedSourceLoc(p.getCurrentLocation());
206  if (failed(verifyConstructionInvariants(loc, inner, dims[0])))
207  return Type();
208 
209  return get(ctxt, inner, dims[0]);
210 }
