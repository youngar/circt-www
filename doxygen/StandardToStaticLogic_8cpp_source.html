24 valueVector getPipelineArgs(Block &block) {
25  valueVector arguments;
26  for (auto &op : block) {
27  if (op.isKnownNonTerminator()) {
28  for (auto operand : op.getOperands()) {
29  if (operand.getKind() == Value::Kind::BlockArgument) {
30  // Add only unique uses
31  if (std::find(arguments.begin(), arguments.end(), operand) ==
32  arguments.end())
33  arguments.push_back(operand);
34  } else if (operand.getDefiningOp()->getBlock() != &block) {
35  // Add only unique uses
36  if (std::find(arguments.begin(), arguments.end(), operand) ==
37  arguments.end())
38  arguments.push_back(operand);
39  }
40  }
41  }
42  }
43  return arguments;
44 }

46 valueVector getPipelineResults(Block &block) {
47  SmallVector<Value, 4> results;
48  for (auto &op : block) {
49  for (auto result : op.getResults()) {
50  bool isResult = false;
51  for (auto user : result.getUsers()) {
52  if (user->getBlock() != &block || user->isKnownTerminator()) {
53  isResult = true;
54  break;
55  }
56  }
57  if (isResult)
58  results.push_back(result);
59  }
60  }
61  return results;
62 }

64 static void createPipeline(mlir::FuncOp f, OpBuilder &builder) {
65  for (Block &block : f) {
66  if (block.front().isKnownNonTerminator()) {
67 
68  auto arguments = getPipelineArgs(block);
69  auto results = getPipelineResults(block);
70  builder.setInsertionPoint(&block.back());
71  builder.create<staticlogic::ReturnOp>(f.getLoc(), ValueRange(results));
72 
73  // Create pipeline operation, and move all operations except terminator
74  // into the pipeline.
75  builder.setInsertionPoint(&block.front());
76  auto pipeline = builder.create<staticlogic::PipelineOp>(
77  f.getLoc(), ValueRange(arguments), ValueRange(results));
78 
79  auto &body = pipeline.getRegion().front();
80  body.getOperations().splice(body.getOperations().begin(),
81  block.getOperations(), ++block.begin(),
82  --block.end());
83 
84  // Reconnect arguments of the pipeline operation.
85  unsigned argIdx = 0;
86  for (auto value : arguments) {
87  value.replaceUsesWithIf(
88  body.getArgument(argIdx),
89  function_ref<bool(OpOperand &)>([&body](OpOperand &use) -> bool {
90  return use.getOwner()->getBlock() == &body;
91  }));
92  argIdx += 1;
93  }
94 
95  // Reconnect results of the pipeline operation.
96  unsigned resultIdx = 0;
97  for (auto value : results) {
98  value.replaceUsesWithIf(
99  pipeline.getResult(resultIdx),
100  function_ref<bool(OpOperand &)>([&body](OpOperand &use) -> bool {
101  return use.getOwner()->getBlock() != &body;
102  }));
103  resultIdx += 1;
104  }
105  }
106  }
107 }
