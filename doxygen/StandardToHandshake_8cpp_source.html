42 /// Remove basic blocks inside the given FuncOp. This allows the result to be
43 /// a valid graph region, since multi-basic block regions are not allowed to
44 /// be graph regions currently.
45 void removeBasicBlocks(handshake::FuncOp funcOp) {
46  auto &entryBlock = funcOp.getBody().front().getOperations();
47 
48  // Erase all TerminatorOp, and move ReturnOp to the end of entry block.
49  for (auto &block : funcOp) {
50  Operation &termOp = block.back();
51  if (isa<handshake::TerminatorOp>(termOp))
52  termOp.erase();
53  else if (isa<handshake::ReturnOp>(termOp))
54  entryBlock.splice(entryBlock.end(), block.getOperations(), termOp);
55  }
56 
57  // Move all operations to entry block and erase other blocks.
58  for (auto &block : llvm::make_early_inc_range(llvm::drop_begin(funcOp, 1))) {
59  entryBlock.splice(--entryBlock.end(), block.getOperations());
60  }
61  for (auto &block : llvm::make_early_inc_range(llvm::drop_begin(funcOp, 1))) {
62  block.clear();
63  block.dropAllDefinedValueUses();
64  for (size_t i = 0; i < block.getNumArguments(); i++) {
65  block.eraseArgument(i);
66  }
67  block.erase();
68  }
69 }

175  // Returns map of values used in block but defined outside of block
176  // For liveness analysis
177  BlockValues uses;
178 
179  for (Block &block : f) {
180 
181  // Operands of operations in b which do not originate from operations or
182  // arguments of b
183  for (Operation &op : block) {
184  for (int i = 0, e = op.getNumOperands(); i < e; ++i) {
185 
186  Block *operandBlock;
187 
188  if (op.getOperand(i).getKind() == Value::Kind::BlockArgument) {
189  // Operand is block argument, get its owner block
190  operandBlock = op.getOperand(i).cast<BlockArgument>().getOwner();
191  } else {
192  // Operand comes from operation, get the block of its defining op
193  auto *operand = op.getOperand(i).getDefiningOp();
194  assert(operand != NULL);
195  operandBlock = operand->getBlock();
196  }
197 
198  // If operand defined in some other block, add to uses
199  if (operandBlock != &block)
200  // Add only unique uses
201  if (std::find(uses[&block].begin(), uses[&block].end(),
202  op.getOperand(i)) == uses[&block].end())
203  uses[&block].push_back(op.getOperand(i));
204  }
205  }
206  }
207  return uses;

257 BlockValues livenessAnalysis(handshake::FuncOp f) {
258  // Liveness analysis algorithm adapted from:
259  // https://suif.stanford.edu/~courses/cs243/lectures/l2.pdf
260  // See slide 19 (Liveness: Iterative Algorithm)
261 
262  // blockUses: values used in block but not defined in block
263  BlockValues blockUses = getBlockUses(f);
264 
265  // blockDefs: values defined in block
266  BlockValues blockDefs = getBlockDefs(f);
267 
268  BlockValues blockLiveIns;
269 
270  bool change = true;
271  // Iterate while there are any changes to any of the livein sets
272  while (change) {
273  change = false;
274 
275  // liveOuts(b): all liveins of all successors of b
276  // liveOuts(b) = U (blockLiveIns(s)) forall successors s of b
277  for (Block &block : f) {
278  vector<Value> liveOuts;
279  for (int i = 0, e = block.getNumSuccessors(); i < e; ++i) {
280  Block *succ = block.getSuccessor(i);
281  liveOuts = vectorUnion(liveOuts, blockLiveIns[succ]);
282  }
283 
284  // diff(b): liveouts of b which are not defined in b
285  // diff(b) = liveOuts(b) - blockDefs(b)
286  vector<Value> diff = vectorDiff(liveOuts, blockDefs[&block]);
287 
288  // liveIns(b) = blockUses(b) U diff(b)
289  vector<Value> tmpLiveIns = vectorUnion(blockUses[&block], diff);
290 
291  // Update blockLiveIns if new liveins found
292  if (tmpLiveIns.size() > blockLiveIns[&block].size()) {
293  blockLiveIns[&block] = tmpLiveIns;
294  change = true;
295  }
296  }
297  }
298 
299  return blockLiveIns;
300 }

308 // Insert appropriate type of Merge CMerge for control-only path,
309 // Merge for single-successor blocks, Mux otherwise
310 Operation *insertMerge(Block *block, Value val,
311  ConversionPatternRewriter &rewriter) {
312  unsigned numPredecessors = getBlockPredecessorCount(block);
313 
314  // Control-only path originates from StartOp
315  if (val.getKind() != Value::Kind::BlockArgument) {
316  if (isa<StartOp>(val.getDefiningOp())) {
317  return rewriter.create<handshake::ControlMergeOp>(block->front().getLoc(),
318  val, numPredecessors);
319  }
320  }
321 
322  // If there are no block predecessors (i.e., entry block), function argument
323  // is set as single operand
324  if (numPredecessors <= 1)
325  return rewriter.create<handshake::MergeOp>(block->front().getLoc(), val, 1);
326 
327  return rewriter.create<handshake::MuxOp>(block->front().getLoc(), val,
328  numPredecessors);

332 // Returns DenseMap of all inserted operations
333 BlockOps insertMergeOps(handshake::FuncOp f, BlockValues blockLiveIns,
334  ConversionPatternRewriter &rewriter) {
335  BlockOps blockMerges;
336 
337  for (Block &block : f) {
338  // Live-ins identified by liveness analysis
339  rewriter.setInsertionPointToStart(&block);
340 
341  for (auto &val : blockLiveIns[&block]) {
342  Operation *newOp = insertMerge(&block, val, rewriter);
343  blockMerges[&block].push_back(newOp);

350  blockMerges[&block].push_back(newOp);

369 // Get value from predBlock which will be set as operand of op (merge)
370 Value getMergeOperand(Operation *op, Block *predBlock, BlockOps blockMerges) {
371  // Helper value (defining value of merge) to identify Merges which propagate
372  // the same defining value
373  Value srcVal = op->getOperand(0);
374 
375  Block *block = op->getBlock();
376 
377  // Value comes from predecessor block (i.e., not an argument of this block)
378  if (std::find(block->getArguments().begin(), block->getArguments().end(),
379  srcVal) == block->getArguments().end()) {
380  // Value is not defined by operation in predBlock
381  if (!blockHasSrcOp(srcVal, predBlock)) {
382  // Find the corresponding Merge
383  for (Operation *predOp : blockMerges[predBlock])
384  if (predOp->getOperand(0) == srcVal)
385  return predOp->getResult(0);
386  } else
387  return srcVal;
388  }
389 
390  // Value is argument of this block
391  // Operand of terminator in predecessor block should be input to Merge
392  else {
393  unsigned index = srcVal.cast<BlockArgument>().getArgNumber();
394  Operation *termOp = predBlock->getTerminator();
395  if (mlir::CondBranchOp br = dyn_cast<mlir::CondBranchOp>(termOp)) {
396  if (block == br.getTrueDest())
397  return br.getTrueOperand(index);
398  else {
399  assert(block == br.getFalseDest());
400  return br.getFalseOperand(index);
401  }
402  } else if (isa<mlir::BranchOp>(termOp))
403  return termOp->getOperand(index);
404  }
405  return nullptr;
406 }

440 void reconnectMergeOps(handshake::FuncOp f, BlockOps blockMerges) {
441  // All merge operands are initially set to original (defining) value
442  // We here replace defining value with appropriate value from predecessor
443  // block The predecessor can either be a merge, the original defining value,
444  // or a branch Operand Operand(0) is helper defining value for identifying
445  // matching merges, it does not correspond to any predecessor block
446 
447  for (Block &block : f) {
448  for (Operation *op : blockMerges[&block]) {
449 
450  int count = 1;
451 
452  // Set appropriate operand from predecessor block
453  for (auto *predBlock : block.getPredecessors()) {
454  Value mgOperand = getMergeOperand(op, predBlock, blockMerges);
455  assert(mgOperand != nullptr);
456  op->setOperand(count, mgOperand);
457  count++;
458  }
459  // Reconnect all operands originating from livein defining value through
460  // corresponding merge of that block
461  for (Operation &opp : block)
462  if (!isa<MergeLikeOpInterface>(opp))
463  opp.replaceUsesOfWith(op->getOperand(0), op->getResult(0));
464  }
465  }
466 
467  // Disconnect original value (Operand(0), used as helper) from all merges
468  // If the original value must be a merge operand, it is still set as some
469  // subsequent operand
470  // If block has multiple predecessors, connect Muxes to ControlMerge
471  for (Block &block : f) {
472  unsigned numPredecessors = getBlockPredecessorCount(&block);
473 
474  if (numPredecessors <= 1) {
475  for (Operation *op : blockMerges[&block])
476  op->eraseOperand(0);
477  } else {
478  Operation *cntrlMg = getControlMerge(&block);
479  assert(cntrlMg != nullptr);
480  cntrlMg->eraseOperand(0);
481 
482  for (Operation *op : blockMerges[&block])
483  if (op != cntrlMg)
484  op->setOperand(0, cntrlMg->getResult(1));
485  }
486  }
487 
488  removeBlockOperands(f);
489 }

515 int getBranchCount(Value val, Block *block) {
516  int uses = 0;
517  for (int i = 0, e = block->getNumSuccessors(); i < e; ++i) {
518  int curr = 0;
519  Block *succ = block->getSuccessor(i);
520  for (auto &u : val.getUses()) {
521  if (u.getOwner()->getBlock() == succ)
522  curr++;
523  }
524  uses = (curr > uses) ? curr : uses;
525  }
526  return uses;

544 void addBranchOps(handshake::FuncOp f, ConversionPatternRewriter &rewriter) {
545 
546  BlockValues liveOuts;
547 
548  for (Block &block : f) {
549  for (Operation &op : block) {
550  for (auto result : op.getResults())
551  if (isLiveOut(result))
552  liveOuts[&block].push_back(result);
553  }
554  }
555 
556  for (Block &block : f) {
557  Operation *termOp = block.getTerminator();
558  rewriter.setInsertionPoint(termOp);
559 
560  for (Value val : liveOuts[&block]) {
561  // Count the number of branches which the liveout needs
562  int numBranches = getBranchCount(val, &block);
563 
564  // Instantiate branches and connect to Merges
565  for (int i = 0, e = numBranches; i < e; ++i) {
566  Operation *newOp = nullptr;
567 
568  if (auto condBranchOp = dyn_cast<mlir::CondBranchOp>(termOp))
569  newOp = rewriter.create<handshake::ConditionalBranchOp>(
570  termOp->getLoc(), condBranchOp.getCondition(), val);
571  else if (isa<mlir::BranchOp>(termOp))
572  newOp = rewriter.create<handshake::BranchOp>(termOp->getLoc(), val);
573 
574  if (newOp != nullptr) {
575  for (int i = 0, e = block.getNumSuccessors(); i < e; ++i) {
576  Block *succ = block.getSuccessor(i);
577  Value res = getSuccResult(termOp, newOp, succ);
578 
579  for (auto &u : val.getUses()) {
580  if (u.getOwner()->getBlock() == succ) {
581  u.getOwner()->replaceUsesOfWith(val, res);
582  break;
583  }
584  }
585  }
586  }
587  }
588  }
589  }

848 // Replaces standard memory ops with their handshake version (i.e.,
849 // ops which connect to memory/LSQ). Returns a map with an ordered
850 // list of new ops corresponding to each memref. Later, we instantiate
851 // a memory node for each memref and connect it to its load/store ops
852 MemRefToMemoryAccessOp replaceMemoryOps(handshake::FuncOp f,
853  ConversionPatternRewriter &rewriter) {
854  // Map from original memref to new load/store operations.
855  MemRefToMemoryAccessOp MemRefOps;
856 
857  vector<Operation *> opsToErase;
858 
859  // Replace load and store ops with the corresponding handshake ops
860  // Need to traverse ops in blocks to store them in MemRefOps in program order
861  for (Block &block : f)
862  for (Operation &op : block) {
863  if (isMemoryOp(&op)) {
864  rewriter.setInsertionPoint(&op);
865  Value memref = getOpMemRef(&op);
866  Operation *newOp = nullptr;
867 
868  if (mlir::LoadOp loadOp = dyn_cast<mlir::LoadOp>(op)) {
869  // Get operands which correspond to address indices
870  // This will add all operands except alloc
871  SmallVector<Value, 8> operands(loadOp.getIndices());
872 
873  newOp =
874  rewriter.create<handshake::LoadOp>(op.getLoc(), memref, operands);
875  op.getResult(0).replaceAllUsesWith(newOp->getResult(0));
876  } else if (mlir::StoreOp storeOp = dyn_cast<mlir::StoreOp>(op)) {
877  // Get operands which correspond to address indices
878  // This will add all operands except alloc and data
879  SmallVector<Value, 8> operands(storeOp.getIndices());
880 
881  // Create new op where operands are store data and address indices
882  newOp = rewriter.create<handshake::StoreOp>(
883  op.getLoc(), storeOp.getValueToStore(), operands);
884  } else if (isa<mlir::AffineReadOpInterface,
885  mlir::AffineWriteOpInterface>(op)) {
886  // Get essential memref access inforamtion.
887  MemRefAccess access(&op);
888  // The address of an affine load/store operation can be a result of an
889  // affine map, which is a linear combination of constants and
890  // parameters. Therefore, we should extract the affine map of each
891  // address and expand it into proper expressions that calculate the
892  // result.
893  AffineMap map;
894  if (auto loadOp = dyn_cast<mlir::AffineReadOpInterface>(op))
895  map = loadOp.getAffineMap();
896  else
897  map = dyn_cast<AffineWriteOpInterface>(op).getAffineMap();
898 
899  // The returned object from expandAffineMap is an optional list of the
900  // expansion results from the given affine map, which are the actual
901  // address indices that can be used as operands for handshake
902  // LoadOp/StoreOp. The following processing requires it to be a valid
903  // result.
904  auto operands =
905  expandAffineMap(rewriter, op.getLoc(), map, access.indices);
906  assert(operands &&
907  "Address operands of affine memref access cannot be reduced.");
908 
909  if (isa<mlir::AffineReadOpInterface>(op)) {
910  newOp = rewriter.create<handshake::LoadOp>(
911  op.getLoc(), access.memref, *operands);
912  op.getResult(0).replaceAllUsesWith(newOp->getResult(0));
913  } else {
914  newOp = rewriter.create<handshake::StoreOp>(
915  op.getLoc(), op.getOperand(0), *operands);
916  }
917  } else {
918  op.emitError("Load/store operation cannot be handled.");
919  }
920 
921  MemRefOps[memref].push_back(newOp);
922 
923  opsToErase.push_back(&op);
924  }
925  }
926 
927  // Erase old memory ops
928  for (unsigned i = 0, e = opsToErase.size(); i != e; ++i) {
929  auto *op = opsToErase[i];
930  for (int i = 0, e = op->getNumOperands(); i < e; ++i)
931  op->eraseOperand(0);
932  assert(op->getNumOperands() == 0);
933 
934  rewriter.eraseOp(op);
935  }
936 
937  return MemRefOps;
938 }

1151 void connectToMemory(handshake::FuncOp f, MemRefToMemoryAccessOp MemRefOps,
1152  bool lsq, ConversionPatternRewriter &rewriter) {
1153  // Add MemoryOps which represent the memory interface
1154  // Connect memory operations and control appropriately
1155  int mem_count = 0;
1156  for (auto memory : MemRefOps) {
1157  // First operand corresponds to memref (alloca or function argument)
1158  Value memrefOperand = getMemRefOperand(memory.first);
1159 
1160  vector<Value> operands;
1161  // operands.push_back(memrefOperand);
1162 
1163  // Get control values which need to connect to memory
1164  vector<Value> controlVals = getControlValues(memory.second);
1165 
1166  // In case of LSQ interface, set control values as inputs (used to trigger
1167  // allocation to LSQ)
1168  if (lsq)
1169  operands.insert(operands.end(), controlVals.begin(), controlVals.end());
1170 
1171  // Add load indices and store data+indices to memory operands
1172  // Count number of loads so that we can generate appropriate number of
1173  // memory outputs (data to load ops)
1174 
1175  // memory.second is in program order
1176  // Enforce MemoryOp port ordering as follows:
1177  // Operands: all stores then all loads (stdata1, staddr1, stdata2,...,
1178  // ldaddr1, ldaddr2,....) Outputs: all load outputs, ordered the same as
1179  // load addresses (lddata1, lddata2, ...), followed by all none outputs,
1180  // ordered as operands (stnone1, stnone2,...ldnone1, ldnone2,...)
1181  vector<int> newInd(memory.second.size(), 0);
1182  int ind = 0;
1183  for (int i = 0, e = memory.second.size(); i < e; ++i) {
1184  auto *op = memory.second[i];
1185  if (isa<handshake::StoreOp>(op)) {
1186  SmallVector<Value, 8> results = getResultsToMemory(op);
1187  operands.insert(operands.end(), results.begin(), results.end());
1188  newInd[i] = ind++;
1189  }
1190  }
1191 
1192  int ld_count = 0;
1193 
1194  for (int i = 0, e = memory.second.size(); i < e; ++i) {
1195  auto *op = memory.second[i];
1196  if (isa<handshake::LoadOp>(op)) {
1197  SmallVector<Value, 8> results = getResultsToMemory(op);
1198  operands.insert(operands.end(), results.begin(), results.end());
1199 
1200  ld_count++;
1201  newInd[i] = ind++;
1202  }
1203  }
1204 
1205  // control-only outputs for each access (indicate access completion)
1206  int cntrl_count = lsq ? 0 : memory.second.size();
1207 
1208  Block *entryBlock = &f.front();
1209  rewriter.setInsertionPointToStart(entryBlock);
1210 
1211  // Place memory op next to the alloc op
1212  Operation *newOp = rewriter.create<MemoryOp>(
1213  entryBlock->front().getLoc(), operands, ld_count, cntrl_count, lsq,
1214  mem_count++, memrefOperand);
1215 
1216  setLoadDataInputs(memory.second, newOp);
1217 
1218  if (!lsq) {
1219  // Create Joins which join done signals from memory with the
1220  // control-only network
1221  addJoinOps(rewriter, controlVals);
1222 
1223  // Connect all load/store done signals to the join of their block
1224  // Ensure that the block terminates only after all its accesses have
1225  // completed
1226  // True is default. When no sync needed, set to false (for now,
1227  // user-determined)
1228  bool control = true;
1229 
1230  if (control)
1231  setJoinControlInputs(memory.second, newOp, ld_count, newInd);
1232  else {
1233  for (int i = 0, e = cntrl_count; i < e; ++i) {
1234  rewriter.setInsertionPointAfter(newOp);
1235  rewriter.create<SinkOp>(newOp->getLoc(),
1236  newOp->getResult(ld_count + i));
1237  }
1238  }
1239 
1240  // Set control-only inputs to each memory op
1241  // Ensure that op starts only after prior blocks have completed
1242  // Ensure that op starts only after predecessor ops (with RAW, WAR, or
1243  // WAW) have completed
1244  setMemOpControlInputs(rewriter, memory.second, newOp, ld_count, newInd);
1245  }
1246  }

1273 /// Rewrite affine.for operations in a handshake.func into its representations
1274 /// as a CFG in the standard dialect. Affine expressions in loop bounds will be
1275 /// expanded to code in the standard dialect that actually computes them. We
1276 /// combine the lowering of affine loops in the following two conversions:
1277 /// [AffineToStandard](https://mlir.llvm.org/doxygen/AffineToStandard_8cpp.html),
1278 /// [SCFToStandard](https://mlir.llvm.org/doxygen/SCFToStandard_8cpp_source.html)
1279 /// into this function.
1280 /// The affine memory operations will be preserved until other rewrite
1281 /// functions, e.g.,`replaceMemoryOps`, are called. Any affine analysis, e.g.,
1282 /// getting dependence information, should be carried out before calling this
1283 /// function; otherwise, the affine for loops will be destructed and key
1284 /// information will be missing.
1285 LogicalResult rewriteAffineFor(handshake::FuncOp f,
1286  ConversionPatternRewriter &rewriter) {
1287  // Get all affine.for operations in the function body.
1288  SmallVector<mlir::AffineForOp, 8> forOps;
1289  f.walk([&](mlir::AffineForOp op) { forOps.push_back(op); });
1290 
1291  // TODO: how to deal with nested loops?
1292  for (unsigned i = 0, e = forOps.size(); i < e; i++) {
1293  auto forOp = forOps[i];
1294 
1295  // Insert lower and upper bounds right at the position of the original
1296  // affine.for operation.
1297  rewriter.setInsertionPoint(forOp);
1298  auto loc = forOp.getLoc();
1299  auto lowerBound = expandAffineMap(rewriter, loc, forOp.getLowerBoundMap(),
1300  forOp.getLowerBoundOperands());
1301  auto upperBound = expandAffineMap(rewriter, loc, forOp.getUpperBoundMap(),
1302  forOp.getUpperBoundOperands());
1303  if (!lowerBound || !upperBound)
1304  return failure();
1305  auto step = rewriter.create<mlir::ConstantIndexOp>(loc, forOp.getStep());
1306 
1307  // Build blocks for a common for loop. initBlock and initPosition are the
1308  // block that contains the current forOp, and the position of the forOp.
1309  auto *initBlock = rewriter.getInsertionBlock();
1310  auto initPosition = rewriter.getInsertionPoint();
1311 
1312  // Split the current block into several parts. `endBlock` contains the code
1313  // starting from forOp. `conditionBlock` will have the condition branch.
1314  // `firstBodyBlock` is the loop body, and `lastBodyBlock` is about the loop
1315  // iterator stepping. Here we move the body region of the AffineForOp here
1316  // and split it into `conditionBlock`, `firstBodyBlock`, and
1317  // `lastBodyBlock`.
1318  // TODO: is there a simpler API for doing so?
1319  auto *endBlock = rewriter.splitBlock(initBlock, initPosition);
1320  // Split and get the references to different parts (blocks) of the original
1321  // loop body.
1322  auto *conditionBlock = &forOp.region().front();
1323  auto *firstBodyBlock =
1324  rewriter.splitBlock(conditionBlock, conditionBlock->begin());
1325  auto *lastBodyBlock =
1326  rewriter.splitBlock(firstBodyBlock, firstBodyBlock->end());
1327  rewriter.inlineRegionBefore(forOp.region(), endBlock);
1328 
1329  // The loop IV is the first argument of the conditionBlock.
1330  auto iv = conditionBlock->getArgument(0);
1331 
1332  // Get the loop terminator, which should be the last operation of the
1333  // original loop body. And `firstBodyBlock` points to that loop body.
1334  auto terminator = dyn_cast<mlir::AffineYieldOp>(firstBodyBlock->back());
1335  if (!terminator)
1336  return failure();
1337 
1338  // First, we fill the content of the lastBodyBlock with how the loop
1339  // iterator steps.
1340  rewriter.setInsertionPointToEnd(lastBodyBlock);
1341  auto stepped = rewriter.create<mlir::AddIOp>(loc, iv, step).getResult();
1342 
1343  // Next, we get the loop carried values, which are terminator operands.
1344  SmallVector<Value, 8> loopCarried;
1345  loopCarried.push_back(stepped);
1346  loopCarried.append(terminator.operand_begin(), terminator.operand_end());
1347  rewriter.create<mlir::BranchOp>(loc, conditionBlock, loopCarried);
1348 
1349  // Then we fill in the condition block.
1350  rewriter.setInsertionPointToEnd(conditionBlock);
1351  auto comparison = rewriter.create<mlir::CmpIOp>(loc, CmpIPredicate::slt, iv,
1352  upperBound.getValue()[0]);
1353 
1354  rewriter.create<mlir::CondBranchOp>(loc, comparison, firstBodyBlock,
1355  ArrayRef<Value>(), endBlock,
1356  ArrayRef<Value>());
1357 
1358  // We also insert the branch operation at the end of the initBlock.
1359  rewriter.setInsertionPointToEnd(initBlock);
1360  // TODO: should we add more operands here?
1361  rewriter.create<mlir::BranchOp>(loc, conditionBlock,
1362  lowerBound.getValue()[0]);
1363 
1364  // Finally, setup the firstBodyBlock.
1365  rewriter.setInsertionPointToEnd(firstBodyBlock);
1366  // TODO: is it necessary to add this explicit branch operation?
1367  rewriter.create<mlir::BranchOp>(loc, lastBodyBlock);
1368 
1369  // Remove the original forOp and the terminator in the loop body.
1370  rewriter.eraseOp(terminator);
1371  rewriter.eraseOp(forOp);
1372  }
1373 
1374  return success();
1375 }

1501  /// DFS-based graph cycle detection and naive buffer insertion. Exactly one
1502  /// 2-slot non-transparent buffer will be inserted into each graph cycle.
1503  void insertBufferOp(Operation *op, OpBuilder &builder) {
1504  // Mark operation as visited and push into the stack.
1505  opVisited[op] = true;
1506  opOnStack[op] = true;
1507

1509  for (auto &operand : op->getUses()) {
1510  auto user = operand.getOwner();
1511 
1512  // If graph cycle detected, insert a BufferOp into the edge.
1513  if (opOnStack[user] && !isa<handshake::BufferOp>(op) &&
1514  !isa<handshake::BufferOp>(user)) {
1515  auto value = operand.get();
1516 
1517  builder.setInsertionPointAfter(op);
1518  auto bufferOp = builder.create<handshake::BufferOp>(
1519  op->getLoc(), value.getType(), value, /*sequential=*/true,
1520  /*control=*/value.getType().isa<NoneType>(),
1521  /*slots=*/2);
1522  value.replaceUsesWithIf(
1523  bufferOp,
1524  function_ref<bool(OpOperand &)>([](OpOperand &operand) -> bool {
1525  return !isa<handshake::BufferOp>(operand.getOwner());
1526  }));
1527  }
1528  // For unvisited operations, recursively call insertBufferOp() method.
1529  else if (!opVisited[user])
1530  insertBufferOp(user, builder);
1531  }
1532  // Pop operation out of the stack.
1533  opOnStack[op] = false;
1534  }

1539  for (auto &op : block) {
1540  opVisited[&op] = false;
1541  opOnStack[&op] = false;
1542  }
1543  }

1545  auto builder = OpBuilder(f.getContext());
1546  for (auto &arg : f.getBody().front().getArguments()) {
1547  for (auto &operand : arg.getUses()) {
1548  if (!opVisited[operand.getOwner()])
1549  insertBufferOp(operand.getOwner(), builder);
