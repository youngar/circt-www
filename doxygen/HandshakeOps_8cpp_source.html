51 bool isReadyToExecute(ArrayRef<mlir::Value> ins, ArrayRef<mlir::Value> outs,
52  llvm::DenseMap<mlir::Value, llvm::Any> &valueMap) {
53  for (auto in : ins)
54  if (valueMap.count(in) == 0)
55  return false;
56 
57  for (auto out : outs)
58  if (valueMap.count(out) > 0)
59  return false;
60 
61  return true;

65 std::vector<llvm::Any>
66 fetchValues(ArrayRef<mlir::Value> values,
67  llvm::DenseMap<mlir::Value, llvm::Any> &valueMap) {
68  std::vector<llvm::Any> ins;

85 // Update the time map after the execution
86 void updateTime(ArrayRef<mlir::Value> ins, ArrayRef<mlir::Value> outs,
87  llvm::DenseMap<mlir::Value, double> &timeMap, double latency) {
88  double time = 0;
89  for (auto &in : ins)
90  time = std::max(time, timeMap[in]);
91  time += latency;
92  for (auto &out : outs)
93  timeMap[out] = time;
94 }

96 bool tryToExecute(Operation *op,
97  llvm::DenseMap<mlir::Value, llvm::Any> &valueMap,
98  llvm::DenseMap<mlir::Value, double> &timeMap,
99  std::vector<mlir::Value> &scheduleList, double latency) {
100  auto ins = toVector(op->getOperands());
101  auto outs = toVector(op->getResults());
102 
103  if (isReadyToExecute(ins, outs, valueMap)) {
104  auto in = fetchValues(ins, valueMap);
105  std::vector<llvm::Any> out(outs.size());
106  auto generalOp = dyn_cast<handshake::GeneralOpInterface>(op);
107  if (!generalOp)
108  op->emitError("Undefined execution for the current op");
109  generalOp.execute(in, out);
110  storeValues(out, outs, valueMap);
111  updateTime(ins, outs, timeMap, latency);
112  scheduleList = outs;
113  return true;
114  } else
115  return false;
116 }

251 bool handshake::MuxOp::tryExecute(
252  llvm::DenseMap<mlir::Value, llvm::Any> &valueMap,
253  llvm::DenseMap<unsigned, unsigned> &memoryMap,
254  llvm::DenseMap<mlir::Value, double> &timeMap,

272 
273  // Consume the inputs.
274  valueMap.erase(control);
275  valueMap.erase(in);
276  scheduleList.push_back(getResult());
277  return true;

280 static LogicalResult verify(MuxOp op) {
281  unsigned numDataOperands = static_cast<int>(op.dataOperands().size());
282  if (numDataOperands < 2)
283  return op.emitError("need at least two inputs to mux");
284 
285  auto selectType = op.selectOperand().getType();
286 
287  unsigned selectBits;
288  if (auto integerType = selectType.dyn_cast<IntegerType>())
289  selectBits = integerType.getWidth();
290  else if (selectType.isIndex())
291  selectBits = IndexType::kInternalStorageBitWidth;
292  else
293  return op.emitError("unsupported type for select operand: ") << selectType;
294 
295  double maxDataOperands = std::pow(2, selectBits);
296  if (numDataOperands > maxDataOperands)
297  return op.emitError("select bitwidth was ")
298  << selectBits << ", which can mux "
299  << static_cast<int64_t>(maxDataOperands) << " operands, but found "
300  << numDataOperands << " operands";
301 
302  return success();
303 }

567 bool handshake::MemoryOp::allocateMemory(
568  llvm::DenseMap<unsigned, unsigned> &memoryMap,
569  std::vector<std::vector<llvm::Any>> &store,
570  std::vector<double> &storeTimes) {
571  unsigned id = getID();
572  if (memoryMap.count(id))
573  return false;
574 
575  auto type = getMemRefType();
576  std::vector<llvm::Any> in;
577 
578  ArrayRef<int64_t> shape = type.getShape();
579  int allocationSize = 1;
580  unsigned count = 0;
581  for (int64_t dim : shape) {
582  if (dim > 0)
583  allocationSize *= dim;
584  else {
585  assert(count < in.size());
586  allocationSize *= llvm::any_cast<APInt>(in[count++]).getSExtValue();
587  }
588  }
589  unsigned ptr = store.size();
590  store.resize(ptr + 1);
591  storeTimes.resize(ptr + 1);
592  store[ptr].resize(allocationSize);
593  storeTimes[ptr] = 0.0;
594  mlir::Type elementType = type.getElementType();
595  int width = elementType.getIntOrFloatBitWidth();
596  for (int i = 0; i < allocationSize; i++) {
597  if (elementType.isa<mlir::IntegerType>()) {
598  store[ptr][i] = APInt(width, 0);
599  } else if (elementType.isa<mlir::FloatType>()) {
600  store[ptr][i] = APFloat(0.0);
601  } else {
602  llvm_unreachable("Unknown result type!\n");
603  }
604  }
605 
606  memoryMap[id] = ptr;
607  return true;
608 }

610 bool handshake::MemoryOp::tryExecute(
611  llvm::DenseMap<mlir::Value, llvm::Any> &valueMap,
612  llvm::DenseMap<unsigned, unsigned> &memoryMap,
613  llvm::DenseMap<mlir::Value, double> &timeMap,
614  std::vector<std::vector<llvm::Any>> &store,
615  std::vector<mlir::Value> &scheduleList) {
616  auto op = getOperation();
617  int opIndex = 0;
618  bool notReady = false;
619  unsigned id = getID(); // The ID of this memory.
620  unsigned buffer = memoryMap[id];
621 
622  for (unsigned i = 0; i < getStCount().getZExtValue(); i++) {
623  mlir::Value data = op->getOperand(opIndex++);
624  mlir::Value address = op->getOperand(opIndex++);
625  mlir::Value nonceOut = op->getResult(getLdCount().getZExtValue() + i);
626  if ((!valueMap.count(data) || !valueMap.count(address))) {
627  notReady = true;
628  continue;
629  }
630  auto addressValue = valueMap[address];
631  auto addressTime = timeMap[address];
632  auto dataValue = valueMap[data];
633  auto dataTime = timeMap[data];
634

673  timeMap[nonceOut] = addressTime;
674  scheduleList.push_back(dataOut);
675  scheduleList.push_back(nonceOut);
676  // Consume the inputs.
677  valueMap.erase(address);
678  }
679  return (notReady) ? false : true;
