25 static void buildModule(OpBuilder &builder, OperationState &result,
26  StringAttr name, ArrayRef<ModulePortInfo> ports) {
27  using namespace mlir::impl;
28 
29  // Add an attribute for the name.
30  result.addAttribute(::mlir::SymbolTable::getSymbolAttrName(), name);
31 
32  SmallVector<Type, 4> argTypes;
33  SmallVector<Type, 4> resultTypes;
34  for (auto elt : ports) {
35  if (elt.isOutput())
36  resultTypes.push_back(elt.type);
37  else
38  argTypes.push_back(elt.type);
39  }
40 
41  // Record the argument and result types as an attribute.
42  auto type = builder.getFunctionType(argTypes, resultTypes);
43  result.addAttribute(getTypeAttrName(), TypeAttr::get(type));
44 
45  // Record the names of the arguments if present.
46  SmallString<8> attrNameBuf;
47  SmallString<8> attrDirBuf;
48  for (const ModulePortInfo &port : ports) {
49  SmallVector<NamedAttribute, 2> argAttrs;
50  if (!port.name.getValue().empty())
51  argAttrs.push_back(
52  NamedAttribute(builder.getIdentifier("rtl.name"), port.name));
53 
54  StringRef attrName = port.isOutput()
55  ? getResultAttrName(port.argNum, attrNameBuf)
56  : getArgAttrName(port.argNum, attrNameBuf);
57  result.addAttribute(attrName, builder.getDictionaryAttr(argAttrs));
58  }
59  result.addRegion();
60 }

138 /// Parse a function result list.
139 ///
140 /// function-result-list ::= function-result-list-parens
141 /// function-result-list-parens ::= `(` `)`
142 /// | `(` function-result-list-no-parens `)`
143 /// function-result-list-no-parens ::= function-result (`,` function-result)*
144 /// function-result ::= (percent-identifier `:`) type attribute-dict?
145 ///
146 static ParseResult
147 parseFunctionResultList(OpAsmParser &parser, SmallVectorImpl<Type> &resultTypes,
148  SmallVectorImpl<NamedAttrList> &resultAttrs) {
149  if (parser.parseLParen())
150  return failure();
151 
152  // Special case for an empty set of parens.
153  if (succeeded(parser.parseOptionalRParen()))
154  return success();
155 
156  // Parse individual function results.
157  do {
158  resultTypes.emplace_back();
159  resultAttrs.emplace_back();
160 
161  OpAsmParser::OperandType operandName;
162  auto namePresent = parser.parseOptionalOperand(operandName);
163  StringRef implicitName;
164  if (namePresent.hasValue()) {
165  if (namePresent.getValue() || parser.parseColon())
166  return failure();
167 
168  // If the name was specified, then we will use it.
169  implicitName = operandName.name.drop_front();
170  }
171 
172  if (parser.parseType(resultTypes.back()) ||
173  parser.parseOptionalAttrDict(resultAttrs.back()))
174  return failure();
175 
176  // If we have an implicit name and no explicit rtl.name attribute, then use
177  // the implicit name as the rtl.name attribute.
178  if (!implicitName.empty() && !getRTLNameAttr(resultAttrs.back())) {
179  auto nameAttr = parser.getBuilder().getStringAttr(implicitName);
180  resultAttrs.back().append("rtl.name", nameAttr);
181  }
182  } while (succeeded(parser.parseOptionalComma()));
183  return parser.parseRParen();
184 }

203 static ParseResult parseRTLModuleOp(OpAsmParser &parser, OperationState &result,
204  bool isExtModule = false) {
205  using namespace mlir::impl;
206 
207  SmallVector<OpAsmParser::OperandType, 4> entryArgs;
208  SmallVector<NamedAttrList, 4> argAttrs;
209  SmallVector<NamedAttrList, 4> resultAttrs;
210  SmallVector<Type, 4> argTypes;
211  SmallVector<Type, 4> resultTypes;
212  auto &builder = parser.getBuilder();
213 
214  // Parse the name as a symbol.
215  StringAttr nameAttr;
216  if (parser.parseSymbolName(nameAttr, ::mlir::SymbolTable::getSymbolAttrName(),
217  result.attributes))
218  return failure();
219 
220  // Parse the function signature.
221  bool isVariadic = false;
222 
223  if (parseModuleFunctionSignature(parser, /*allowVariadic=*/false, entryArgs,
224  argTypes, argAttrs, isVariadic, resultTypes,
225  resultAttrs))
226  return failure();
227 
228  // Record the argument and result types as an attribute. This is necessary
229  // for external modules.
230  auto type = builder.getFunctionType(argTypes, resultTypes);
231  result.addAttribute(getTypeAttrName(), TypeAttr::get(type));
232 
233  // If function attributes are present, parse them.
234  if (parser.parseOptionalAttrDictWithKeyword(result.attributes))
235  return failure();
236 
237  assert(argAttrs.size() == argTypes.size());
238  assert(resultAttrs.size() == resultTypes.size());
239 
240  auto *context = result.getContext();
241 
242  // Postprocess each of the arguments. If there was no 'rtl.name'
243  // attribute, and if the argument name was non-numeric, then add the
244  // rtl.name attribute with the textual name from the IR. The name in the
245  // text file is a load-bearing part of the IR, but we don't want the
246  // verbosity in dumps of including it explicitly in the attribute
247  // dictionary.
248  for (size_t i = 0, e = argAttrs.size(); i != e; ++i) {
249  auto &attrs = argAttrs[i];
250 
251  // If an explicit name attribute was present, don't add the implicit one.
252  bool hasNameAttr = false;
253  for (auto &elt : attrs)
254  if (elt.first.str() == "rtl.name")
255  hasNameAttr = true;
256  if (hasNameAttr || entryArgs.empty())
257  continue;
258 
259  auto &arg = entryArgs[i];
260 
261  // The name of an argument is of the form "%42" or "%id", and since
262  // parsing succeeded, we know it always has one character.
263  assert(arg.name.size() > 1 && arg.name[0] == '%' && "Unknown MLIR name");
264  if (isdigit(arg.name[1]))
265  continue;
266 
267  auto nameAttr = StringAttr::get(arg.name.drop_front(), context);
268  attrs.push_back({Identifier::get("rtl.name", context), nameAttr});
269  }
270 
271  // Add the attributes to the function arguments.
272  addArgAndResultAttrs(builder, result, argAttrs, resultAttrs);
273 
274  // Parse the optional function body.
275  auto *body = result.addRegion();
276  if (!isExtModule) {
277  if (parser.parseRegion(*body, entryArgs,
278  entryArgs.empty() ? ArrayRef<Type>() : argTypes))
279  return failure();
280 
281  RTLModuleOp::ensureTerminator(*body, parser.getBuilder(), result.location);
282  }
283  return success();
284 }

296 static void printModuleSignature(OpAsmPrinter &p, Operation *op,
297  ArrayRef<Type> argTypes, bool isVariadic,
298  ArrayRef<Type> resultTypes) {
299  Region &body = op->getRegion(0);
300  bool isExternal = body.empty();
301 
302  p << '(';
303  for (unsigned i = 0, e = argTypes.size(); i < e; ++i) {
304  if (i > 0)
305  p << ", ";
306 
307  Value argumentValue;
308  if (!isExternal) {
309  argumentValue = body.front().getArgument(i);
310  p.printOperand(argumentValue);
311  p << ": ";
312  }
313 
314  p.printType(argTypes[i]);
315 
316  auto argAttrs = ::mlir::impl::getArgAttrs(op, i);
317 
318  // If the argument has the rtl.name attribute, and if it was used by
319  // the printer exactly (not name mangled with a suffix etc) then we can
320  // omit the rtl.name attribute from the argument attribute dictionary.
321  ArrayRef<StringRef> elidedAttrs;
322  StringRef tmp;
323  if (argumentValue) {
324  if (auto nameAttr = getRTLNameAttr(argAttrs)) {
325 
326  // Check to make sure the asmprinter is printing it correctly.
327  SmallString<32> resultNameStr;
328  llvm::raw_svector_ostream tmpStream(resultNameStr);
329  p.printOperand(argumentValue, tmpStream);
330 
331  // If the name is the same as we would otherwise use, then we're good!
332  if (tmpStream.str().drop_front() == nameAttr.getValue()) {
333  tmp = "rtl.name";
334  elidedAttrs = tmp;
335  }
336  }
337  }
338  p.printOptionalAttrDict(argAttrs, elidedAttrs);
339  }
340 
341  if (isVariadic) {
342  if (!argTypes.empty())
343  p << ", ";
344  p << "...";
345  }
346 
347  p << ')';
348 
349  // We print result types specially since we support named arguments.
350  if (!resultTypes.empty()) {
351  auto &os = p.getStream();
352  os << " -> (";
353  for (size_t i = 0, e = resultTypes.size(); i < e; ++i) {
354  if (i != 0)
355  os << ", ";
356  auto resultAttrs = ::mlir::impl::getResultAttrs(op, i);
357  StringAttr name = getRTLNameAttr(resultAttrs);
358  if (name)
359  os << '%' << name.getValue() << ": ";
360 
361  p.printType(resultTypes[i]);
362  p.printOptionalAttrDict(resultAttrs, {"rtl.name"});
363  }
364  os << ')';
365  }
366 }

414 static LogicalResult verifyInstanceOp(InstanceOp op) {
415  // Check that this instance is inside a module.
416  auto module = dyn_cast<RTLModuleOp>(op->getParentOp());
417  if (!module) {
418  op.emitOpError("should be embedded in an 'rtl.module'");
419  return failure();
420  }
421 
422  auto referencedModule = op.getReferencedModule();
423  if (referencedModule == nullptr)
424  return op.emitError("Cannot find module definition '")
425  << op.moduleName() << "'";
426 
427  if (!isa<RTLModuleOp>(referencedModule) &&
428  !isa<RTLExternModuleOp>(referencedModule))
429  return op.emitError("Symbol resolved to '")
430  << referencedModule->getName()
431  << "' which is not a RTL[Ext]ModuleOp";
432 
433  if (auto paramDictOpt = op.parameters()) {
434  DictionaryAttr paramDict = paramDictOpt.getValue();
435  auto checkParmValue = [&](NamedAttribute elt) -> bool {
436  auto value = elt.second;
437  if (value.isa<IntegerAttr>() || value.isa<StringAttr>() ||
438  value.isa<FloatAttr>())
439  return true;
440  op.emitError() << "has unknown extmodule parameter value '" << elt.first
441  << "' = " << value;
442  return false;
443  };
444 
445  if (!llvm::all_of(paramDict, checkParmValue))
446  return failure();
447  }
448 
449  return success();
450 }

491 /// Verify that the num of operands and types fit the declared results.
492 static LogicalResult verifyOutputOp(OutputOp *op) {
493  OperandRange outputValues = op->getOperands();
494  auto opParent = (*op)->getParentOp();
495 
496  // Check that we are in the correct region. OutputOp should be directly
497  // contained by an RTLModuleOp region. We'll loosen this restriction if
498  // there's a compelling use case.
499  if (!isa<RTLModuleOp>(opParent)) {
500  op->emitOpError("operation expected to be in a RTLModuleOp.");
501  return failure();
502  }
503 
504  // Check that the we (rtl.output) have the same number of operands as our
505  // region has results.
506  FunctionType modType = getModuleType(opParent);
507  ArrayRef<Type> modResults = modType.getResults();
508  if (modResults.size() != outputValues.size()) {
509  op->emitOpError("must have same number of operands as region results.");
510  return failure();
511  }
512 
513  // Check that the types of our operands and the region's results match.
514  for (size_t i = 0, e = modResults.size(); i < e; ++i) {
515  if (modResults[i] != outputValues[i].getType()) {
516  op->emitOpError("output types must match module. In "
517  "operand ")
518  << i << ", expected " << modResults[i] << ", but got "
519  << outputValues[i].getType() << ".";
520  return failure();
521  }
522  }
523 
524  return success();
525 }
