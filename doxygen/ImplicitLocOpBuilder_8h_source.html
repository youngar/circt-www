20 /// ImplictLocOpBuilder maintains a 'current location', allowing use of the
21 /// create<> method without specifying the location. It is otherwise the same
22 /// as OpBuilder.
23 class ImplicitLocOpBuilder : public mlir::OpBuilder {
24  using OpBuilder = mlir::OpBuilder;
25  using Location = mlir::Location;
26  using Block = mlir::Block;
27  using Value = mlir::Value;
28 
29 public:
30  /// Create an ImplicitLocOpBuilder using the insertion point and listener from
31  /// an existing OpBuilder.
32  ImplicitLocOpBuilder(Location loc, const OpBuilder &builder)
33  : OpBuilder(builder), curLoc(loc) {}
34 
35  /// OpBuilder has a bunch of convenience constructors - we support them all
36  /// with the additional Location.
37  template <typename T>
38  ImplicitLocOpBuilder(Location loc, T &&operand, Listener *listener = nullptr)
39  : OpBuilder(operand, listener), curLoc(loc) {}
40 
41  ImplicitLocOpBuilder(Location loc, Block *block, Block::iterator insertPoint,
42  Listener *listener = nullptr)
43  : OpBuilder(block, insertPoint, listener), curLoc(loc) {}
44  ImplicitLocOpBuilder(mlir::Operation *op)
45  : OpBuilder(op), curLoc(op->getLoc()) {}
46 
47  /// Create a builder and set the insertion point to before the first operation
48  /// in the block but still inside the block.
49  static ImplicitLocOpBuilder atBlockBegin(Location loc, Block *block,
50  Listener *listener = nullptr) {
51  return ImplicitLocOpBuilder(loc, block, block->begin(), listener);
52  }
53 
54  /// Create a builder and set the insertion point to after the last operation
55  /// in the block but still inside the block.
56  static ImplicitLocOpBuilder atBlockEnd(Location loc, Block *block,
57  Listener *listener = nullptr) {
58  return ImplicitLocOpBuilder(loc, block, block->end(), listener);
59  }
60 
61  /// Create a builder and set the insertion point to before the block
62  /// terminator.
63  static ImplicitLocOpBuilder atBlockTerminator(Location loc, Block *block,
64  Listener *listener = nullptr) {
65  auto *terminator = block->getTerminator();
66  assert(terminator != nullptr && "the block has no terminator");
67  return ImplicitLocOpBuilder(loc, block, Block::iterator(terminator),
68  listener);
69  }
70 
71  /// Accessors for the implied location.
72  Location getLoc() const { return curLoc; }
73  void setLoc(Location loc) { curLoc = loc; }
74 
75  // We allow clients to use the explicit-loc version of create as well.
76  using OpBuilder::create;
77  using OpBuilder::createOrFold;
78 
79  /// Create an operation of specific op type at the current insertion point and
80  /// location.
81  template <typename OpTy, typename... Args>
82  OpTy create(Args &&... args) {
83  return OpBuilder::create<OpTy>(curLoc, std::forward<Args>(args)...);
84  }
85 
86  /// Create an operation of specific op type at the current insertion point,
87  /// and immediately try to fold it. This functions populates 'results' with
88  /// the results after folding the operation.
89  template <typename OpTy, typename... Args>
90  void createOrFold(llvm::SmallVectorImpl<Value> &results, Args &&... args) {
91  OpBuilder::createOrFold<OpTy>(results, curLoc, std::forward<Args>(args)...);
92  }
93 
94  /// Overload to create or fold a single result operation.
95  template <typename OpTy, typename... Args>
96  typename std::enable_if<OpTy::template hasTrait<mlir::OpTrait::OneResult>(),
97  Value>::type
98  createOrFold(Args &&... args) {
99  return OpBuilder::createOrFold<OpTy>(curLoc, std::forward<Args>(args)...);
100  }
101 
102  /// Overload to create or fold a zero result operation.
103  template <typename OpTy, typename... Args>
104  typename std::enable_if<OpTy::template hasTrait<mlir::OpTrait::ZeroResult>(),
105  OpTy>::type
106  createOrFold(Args &&... args) {
107  return OpBuilder::createOrFold<OpTy>(curLoc, std::forward<Args>(args)...);
108  }
109 
110  /// This builder can also be used to emit diagnostics to the current location.
111  mlir::InFlightDiagnostic
112  emitError(const llvm::Twine &message = llvm::Twine()) {
113  return mlir::emitError(curLoc, message);
114  }
115  mlir::InFlightDiagnostic
116  emitWarning(const llvm::Twine &message = llvm::Twine()) {
117  return mlir::emitWarning(curLoc, message);
118  }
119  mlir::InFlightDiagnostic
120  emitRemark(const llvm::Twine &message = llvm::Twine()) {
121  return mlir::emitRemark(curLoc, message);
122  }
123 
124 private:
125  Location curLoc;
126 };
