23 /// Return the element type of an ArrayType or UnpackedArrayType, or null if the
24 /// operand isn't an array.
25 Type circt::sv::getAnyRTLArrayElementType(Type type) {
26  if (!type)
27  return {};
28  if (auto array = type.dyn_cast<rtl::ArrayType>())
29  return array.getElementType();
30  if (auto array = type.dyn_cast<rtl::UnpackedArrayType>())
31  return array.getElementType();
32  return {};
33 }

81 /// Parses a type registered to this dialect
82 Type SVDialect::parseType(DialectAsmParser &parser) const {
83  llvm::StringRef mnemonic;
84  auto loc = parser.getCurrentLocation();
85  if (parser.parseKeyword(&mnemonic))
86  return Type();
87  if (auto type = generatedTypeParser(getContext(), parser, mnemonic))
88  return type;
89  parser.emitError(loc, "Failed to parse type sv.") << mnemonic << "\n";
90  return Type();
91 }
