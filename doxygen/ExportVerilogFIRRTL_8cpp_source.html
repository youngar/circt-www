50 /// Return the width of the specified FIRRTL type in bits or -1 if it isn't
51 /// supported.
52 static int getBitWidthOrSentinel(Type type) {
53  return TypeSwitch<Type, int>(type)
54  .Case<IntegerType>([](IntegerType integerType) {
55  // Turn zero-bit values into single bit ones for simplicity. This
56  // generates correct logic, even though it isn't efficient.
57  // FIXME: This actually isn't correct at all!
58  auto result = integerType.getWidth();
59  return result ? result : 1;
60  })
61  .Case<ClockType, ResetType, AsyncResetType>([](Type) { return 1; })
62  .Case<SIntType, UIntType>([](IntType intType) {
63  // Turn zero-bit values into single bit ones for simplicity. This
64  // occurs in the addr lines of mems with depth=1.
65  // FIXME: This actually isn't correct at all!
66  auto result = intType.getWidthOrSentinel();
67  return result ? result : 1;
68  })
69  .Case<AnalogType>([](AnalogType analogType) {
70  auto result = analogType.getWidthOrSentinel();
71  return result ? result : 1;
72  })
73  .Case<FlipType>([](FlipType flipType) {
74  return getBitWidthOrSentinel(flipType.getElementType());
75  })
76  .Default([](Type) { return -1; });
77 }

101 /// Return true if this is a noop cast that will emit with no syntax.
102 static bool isNoopCast(Operation *op) {
103  // These are always noop casts.
104  if (isa<AsAsyncResetPrimOp>(op) || isa<AsClockPrimOp>(op) ||
105  isa<AsUIntPrimOp>(op) || isa<AsSIntPrimOp>(op) ||
106  isa<AsPassivePrimOp>(op) || isa<AsNonPassivePrimOp>(op))
107  return true;
108 
109  // cvt from signed is noop.
110  if (isa<CvtPrimOp>(op) &&
111  op->getOperand(0).getType().cast<IntType>().isSigned())
112  return true;
113 
114  // Shift left by zero is a noop.
115  if (auto shl = dyn_cast<ShlPrimOp>(op))
116  if (shl.amount() == 0)
117  return true;
118 
119  return false;

134 /// Convert a nested bundle of fields into a flat list of fields. This is used
135 /// when working with instances and mems to flatten them.
136 static void flattenBundleTypes(Type type, StringRef suffixSoFar, bool isFlipped,
137  SmallVectorImpl<FlatBundleFieldEntry> &results) {
138  if (auto flip = type.dyn_cast<FlipType>())
139  return flattenBundleTypes(flip.getElementType(), suffixSoFar, !isFlipped,
140  results);
141 
142  // In the base case we record this field.
143  auto bundle = type.dyn_cast<BundleType>();
144  if (!bundle) {
145  results.push_back({type, suffixSoFar.str(), isFlipped});
146  return;
147  }
148 
149  SmallString<16> tmpSuffix(suffixSoFar);
150 
151  // Otherwise, we have a bundle type. Break it down.
152  for (auto &elt : bundle.getElements()) {
153  // Construct the suffix to pass down.
154  tmpSuffix.resize(suffixSoFar.size());
155  tmpSuffix.push_back('_');
156  tmpSuffix.append(elt.name.strref());
157  // Recursively process subelements.
158  flattenBundleTypes(elt.type, tmpSuffix, isFlipped, results);
159  }

447 /// Add the specified name to the name table, auto-uniquing the name if
448 /// required. If the name is empty, then this creates a unique temp name.
449 StringRef ModuleEmitter::addName(Value value, StringRef name) {
450  if (name.empty())
451  name = "_T";
452 
453  // Check to see if this name is valid. The first character cannot be a
454  // number of other weird thing. If it is, start with an underscore.
455  if (!isalpha(name.front()) && name.front() != '_') {
456  SmallString<16> tmpName("_");
457  tmpName += name;
458  return addName(value, tmpName);
459  }
460 
461  auto isValidVerilogCharacter = [](char ch) -> bool {
462  return isalpha(ch) || isdigit(ch) || ch == '_';
463  };
464 
465  // Check to see if the name consists of all-valid identifiers. If not, we
466  // need to escape them.
467  for (char ch : name) {
468  if (isValidVerilogCharacter(ch))
469  continue;
470 
471  // Otherwise, we need to escape it.
472  SmallString<16> tmpName;
473  for (char ch : name) {
474  if (isValidVerilogCharacter(ch))
475  tmpName += ch;
476  else if (ch == ' ')
477  tmpName += '_';
478  else {
479  tmpName += llvm::utohexstr((unsigned char)ch);
480  }
481  }
482  return addName(value, tmpName);
483  }
484 
485  // Get the list of reserved words we need to avoid. We could prepopulate this
486  // into the used words cache, but it is large and immutable, so we just query
487  // it when needed.
488  auto &reservedWords = getReservedWords();
489 
490  // Check to see if this name is available - if so, use it.
491  if (!reservedWords.count(name)) {
492  auto insertResult = usedNames.insert(name);
493  if (insertResult.second) {
494  nameTable[value] = &*insertResult.first;
495  return insertResult.first->getKey();
496  }
497  }
498 
499  // If not, we need to auto-unique it.
500  SmallVector<char, 16> nameBuffer(name.begin(), name.end());
501  nameBuffer.push_back('_');
502  auto baseSize = nameBuffer.size();
503 
504  // Try until we find something that works.
505  while (1) {
506  auto suffix = llvm::utostr(nextGeneratedNameID++);
507  nameBuffer.append(suffix.begin(), suffix.end());
508  name = StringRef(nameBuffer.data(), nameBuffer.size());
509 
510  if (!reservedWords.count(name)) {
511  auto insertResult = usedNames.insert(name);
512  if (insertResult.second) {
513  nameTable[value] = &*insertResult.first;
514  return insertResult.first->getKey();
515  }
516  }
517 
518  // Chop off the suffix and try again.
519  nameBuffer.resize(baseSize);
520  }
521 }

523 /// Return the location information as a (potentially empty) string.
524 std::string
525 ModuleEmitter::getLocationInfoAsString(const SmallPtrSet<Operation *, 8> &ops) {
526  std::string resultStr;
527  llvm::raw_string_ostream sstr(resultStr);
528 
529  // Multiple operations may come from the same location or may not have useful
530  // location info. Unique it now.
531  SmallPtrSet<Attribute, 8> locations;
532  for (auto *op : ops) {
533  if (auto loc = op->getLoc().dyn_cast<FileLineColLoc>())
534  locations.insert(loc);
535  }
536 
537  auto printLoc = [&](FileLineColLoc loc) {
538  sstr << loc.getFilename();
539  if (auto line = loc.getLine()) {
540  sstr << ':' << line;
541  if (auto col = loc.getColumn())
542  sstr << ':' << col;
543  }
544  };
545 
546  switch (locations.size()) {
547  case 1:
548  printLoc((*locations.begin()).cast<FileLineColLoc>());
549  LLVM_FALLTHROUGH;
550  case 0:
551  return sstr.str();
552  default:
553  break;
554  }
555 
556  // Sort the entries.
557  SmallVector<FileLineColLoc, 8> locVector;
558  locVector.reserve(locations.size());
559  for (auto loc : locations)
560  locVector.push_back(loc.cast<FileLineColLoc>());
561 
562  llvm::array_pod_sort(
563  locVector.begin(), locVector.end(),
564  [](const FileLineColLoc *lhs, const FileLineColLoc *rhs) -> int {
565  if (auto fn = lhs->getFilename().compare(rhs->getFilename()))
566  return fn;
567  if (lhs->getLine() != rhs->getLine())
568  return lhs->getLine() < rhs->getLine() ? -1 : 1;
569  return lhs->getColumn() < rhs->getColumn() ? -1 : 1;
570  });
571 
572  // The entries are sorted by filename, line, col. Try to merge together
573  // entries to reduce verbosity on the column info.
574  StringRef lastFileName;
575  for (size_t i = 0, e = locVector.size(); i != e;) {
576  if (i != 0)
577  sstr << ", ";
578 
579  // Print the filename if it changed.
580  auto first = locVector[i];
581  if (first.getFilename() != lastFileName) {
582  lastFileName = first.getFilename();
583  sstr << lastFileName;
584  }
585 
586  // Scan for entires with the same file/line.
587  size_t end = i + 1;
588  while (end != e && first.getFilename() == locVector[end].getFilename() &&
589  first.getLine() == locVector[end].getLine())
590  ++end;
591 
592  // If we have one entry, print it normally.
593  if (end == i + 1) {
594  if (auto line = first.getLine()) {
595  sstr << ':' << line;
596  if (auto col = first.getColumn())
597  sstr << ':' << col;
598  }
599  ++i;
600  continue;
601  }
602 
603  // Otherwise print a brace enclosed list.
604  sstr << ':' << first.getLine() << ":{";
605  while (i != end) {
606  sstr << locVector[i++].getColumn();
607 
608  if (i != end)
609  sstr << ',';
610  }
611  sstr << '}';
612  }
613 
614  return sstr.str();
615 }

672 /// This builds a recursively nested expression from an SSA use-def graph. This
673 /// uses a post-order walk, but it needs to obey precedence and signedness
674 /// constraints that depend on the behavior of the child nodes. To handle this,
675 /// we emit the characters to a SmallVector which allows us to emit a bunch of

876 SubExprInfo ExprEmitter::emitBinary(Operation *op, VerilogPrecedence prec,
877  const char *syntax, bool opSigned) {
878  if (opSigned)
879  os << "$signed(";
880  auto lhsInfo = emitSubExpr(op->getOperand(0), prec);
881  if (opSigned)
882  os << ")";
883  os << ' ' << syntax << ' ';
884 
885  // The precedence of the RHS operand must be tighter than this operator if
886  // they have a different opcode in order to handle things like "x-(a+b)".
887  // This isn't needed on the LHS, because the relevant Verilog operators are
888  // left-associative.
889  //
890  auto *rhsOperandOp = lookThroughNoopCasts(op->getOperand(1)).getDefiningOp();
891  auto rhsPrec = VerilogPrecedence(prec - 1);
892  if (rhsOperandOp && op->getName() == rhsOperandOp->getName())
893  rhsPrec = prec;
894 
895  if (opSigned)
896  os << "$signed(";
897  auto rhsInfo = emitSubExpr(op->getOperand(1), rhsPrec);
898  if (opSigned)
899  os << ")";
900 
901  // If we have a strict sign, then match the firrtl operation sign.
902  // Otherwise, the result is signed if both operands are signed.
903  SubExprSignedness signedness;
904  if (opSigned ||
905  (lhsInfo.signedness == IsSigned && rhsInfo.signedness == IsSigned))
906  signedness = IsSigned;
907  else
908  signedness = IsUnsigned;
909 
910  return {prec, signedness};
911 }

976 /// Emit the specified value as a subexpression to the stream.
977 SubExprInfo ExprEmitter::emitSubExpr(Value exp,
978  VerilogPrecedence parenthesizeIfLooserThan,
979  bool forceExpectedSign, bool opForceSign) {
980  auto *op = exp.getDefiningOp();
981  bool shouldEmitInlineExpr = op && isVerilogExpression(op);
982 
983  // Don't emit this expression inline if it has multiple uses.
984  if (shouldEmitInlineExpr && parenthesizeIfLooserThan != ForceEmitMultiUse &&
985  emitter.outOfLineExpressions.count(op))
986  shouldEmitInlineExpr = false;
987 
988  // If this is a non-expr or shouldn't be done inline, just refer to its
989  // name.
990  if (!shouldEmitInlineExpr) {
991  if ((forceExpectedSign && getSignednessOf(exp.getType()) != IsUnsigned) ||
992  opForceSign) {
993  os << "$signed(" << emitter.getName(exp) << ')';
994  return {Unary, IsSigned};
995  }
996  os << emitter.getName(exp);
997  return {Symbol, IsUnsigned};
998  }
999 
1000  unsigned subExprStartIndex = resultBuffer.size();
1001 
1002  // Okay, this is an expression we should emit inline. Do this through our
1003  // visitor.
1004  auto expInfo = dispatchExprVisitor(exp.getDefiningOp());
1005 
1006  // Check cases where we have to insert things before the expression now that
1007  // we know things about it.
1008  if (forceExpectedSign &&
1009  getSignednessOf(exp.getType()) != expInfo.signedness) {
1010  // If the sign of the result matters and we emitted something with the
1011  // wrong sign, correct it.
1012  StringRef cast = expInfo.signedness == IsSigned ? "$unsigned(" : "$signed(";
1013  resultBuffer.insert(resultBuffer.begin() + subExprStartIndex, cast.begin(),
1014  cast.end());
1015  os << ')';
1016 
1017  } else if (expInfo.precedence > parenthesizeIfLooserThan) {
1018  // If this subexpression would bind looser than the expression it is bound
1019  // into, then we need to parenthesize it. Insert the parentheses
1020  // retroactively.
1021  resultBuffer.insert(resultBuffer.begin() + subExprStartIndex, '(');
1022  os << ')';
1023  }
1024 
1025  // Remember that we emitted this.
1026  emittedExprs.insert(exp.getDefiningOp());
1027  return expInfo;
1028 }

1030 static void collectCatValues(Value operand, SmallVectorImpl<Value> &catValues,
1031  ExprEmitter &emitter) {
1032  operand = emitter.lookThroughNoopCasts(operand);
1033 
1034  // If the operand is a single-use cat, flatten it into the vector we're
1035  // building.
1036  if (auto *op = operand.getDefiningOp())
1037  if (auto cat = dyn_cast<CatPrimOp>(op))
1038  if (!emitter.emitter.outOfLineExpressions.count(cat)) {
1039  collectCatValues(cat.lhs(), catValues, emitter);
1040  collectCatValues(cat.rhs(), catValues, emitter);
1041  return;
1042  }
1043  catValues.push_back(operand);
1044 }

1046 /// Emit a verilog concatenation of the specified values. If the before or
1047 /// after strings are specified, they are included as prefix/postfix elements
1048 /// of the concatenation, respectively.
1049 SubExprInfo ExprEmitter::emitCat(ArrayRef<Value> values, StringRef before,
1050  StringRef after) {
1051  os << '{';
1052  if (!before.empty()) {
1053  os << before;
1054  if (!values.empty() || !after.empty())
1055  os << ", ";
1056  }
1057 
1058  // Recursively flatten any nested cat-like expressions.
1059  SmallVector<Value, 8> elements;
1060  for (auto v : values)
1061  collectCatValues(v, elements, *this);
1062 
1063  llvm::interleaveComma(elements, os,
1064  [&](Value v) { emitSubExpr(v, LowestPrecedence); });
1065 
1066  if (!after.empty()) {
1067  if (!values.empty())
1068  os << ", ";
1069  os << after;
1070  }
1071  os << '}';
1072  return {Unary, IsUnsigned};
1073 }

1075 /// Emit a verilog bit selection operation like x[4:0], the bit numbers are
1076 /// inclusive like verilog.
1077 ///
1078 /// Note that anything that emits a BitSelect must be handled in the
1079 /// isExpressionUnableToInline.
1080 SubExprInfo ExprEmitter::emitBitSelect(Value operand, unsigned hiBit,
1081  unsigned loBit) {
1082  auto x = emitSubExpr(operand, LowestPrecedence);
1083  assert(x.precedence == Symbol &&
1084  "should be handled by isExpressionUnableToInline");
1085 
1086  // If we're extracting the whole input, just return it. This is valid but
1087  // non-canonical IR, and we don't want to generate invalid Verilog.
1088  if (loBit == 0 &&
1089  unsigned(getBitWidthOrSentinel(operand.getType())) == hiBit + 1)
1090  return x;
1091 
1092  os << '[' << hiBit;
1093  if (hiBit != loBit) // Emit x[4] instead of x[4:4].
1094  os << ':' << loBit;
1095  os << ']';
1096  return {Unary, IsUnsigned};
1097 }

1173 SubExprInfo ExprEmitter::visitExpr(PadPrimOp op) {
1174  auto inType = getTypeOf<IntType>(op.getOperand());
1175  auto inWidth = inType.getWidthOrSentinel();
1176  if (inWidth == -1)
1177  return visitUnhandledExpr(op);
1178  auto destWidth = op.amount();
1179 
1180  // If the destination width is smaller than the input width, then this is a
1181  // truncation.
1182  if (destWidth == unsigned(inWidth))
1183  return emitNoopCast(op);
1184  if (destWidth < unsigned(inWidth))
1185  return emitBitSelect(op.getOperand(), destWidth - 1, 0);
1186 
1187  // If this is unsigned, it is a zero extension.
1188  if (!inType.isSigned()) {
1189  os << "{{" << (destWidth - inWidth) << "'d0}, ";
1190  emitSubExpr(op.getOperand(), LowestPrecedence);
1191  os << "}";
1192  return {Unary, IsUnsigned};
1193  }
1194 
1195  // Handle sign extend from a single bit in a pretty way.
1196  if (inWidth == 1) {
1197  os << '{' << destWidth << '{';
1198  emitSubExpr(op.getOperand(), LowestPrecedence);
1199  os << "}}";
1200  return {Unary, IsUnsigned};
1201  }
1202 
1203  // Otherwise, this is a sign extension of a general expression.
1204  // TODO(QoI): Instead of emitting the expression multiple times, we should
1205  // emit it to a known name.
1206  os << "{{" << (destWidth - inWidth) << '{';
1207  emitSubExpr(op.getOperand(), Unary);
1208  os << '[' << (inWidth - 1) << "]}}, ";
1209  emitSubExpr(op.getOperand(), LowestPrecedence);
1210  os << "}";
1211  return {Unary, IsUnsigned};
1212 }

1540 void ModuleEmitter::emitDecl(MemOp op) {
1541  // Check that the MemOp has been properly lowered before this.
1542  if (op.readLatency() != 0 || op.writeLatency() != 1) {
1543  // FIXME: This should be an error.
1544  op.emitWarning("FIXME: need to support mem read/write latency correctly");
1545  }
1546  SmallPtrSet<Operation *, 8> ops;
1547  ops.insert(op);
1548 
1549  auto memName = op.name().getValue().str();
1550  uint64_t depth = op.depth();
1551  auto locInfo = getLocationInfoAsString(ops);
1552 
1553  // If we haven't already emitted a declaration of initvar, do so.
1554  if (!emittedInitVar && depth > 1) {
1555  assert(depth < (1ULL << 31) &&
1556  "FIXME: This doesn't support mems greater than 2^32");
1557  addInitial("integer initvar;", "", /*ppCond*/ "RANDOMIZE_MEM_INIT",
1558  /*cond*/ "", /*partialOrder: initVar decl*/ 10);
1559  emittedInitVar = true;
1560  }
1561 
1562  // Aggregate mems may declare multiple reg's. We need to random initialize
1563  // them all.
1564  SmallVector<FlatBundleFieldEntry, 8> fieldTypes;
1565  flattenBundleTypes(op.getDataType(), "", false, fieldTypes);
1566 
1567  emitRandomizeProlog();
1568 
1569  // On initialization, fill the mem with random bits if RANDOMIZE_MEM_INIT
1570  // is set.
1571  if (depth == 1) { // Don't emit a for loop for one element.
1572  for (const auto &elt : fieldTypes) {
1573  std::string action = memName + elt.suffix + "[0] = `RANDOM;";
1574  addInitial(action, locInfo, /*ppCond*/ "RANDOMIZE_MEM_INIT",
1575  /*cond*/ "", /*partialOrder: After initVar decl*/ 11);
1576  }
1577 
1578  } else if (!fieldTypes.empty()) {
1579  auto initvar = memName + "_initvar";
1580  std::string action = "for (" + initvar + " = 0; " + initvar + " < " +
1581  llvm::utostr(depth) + "; " + initvar + " = " +
1582  initvar + "+1)";
1583 
1584  if (fieldTypes.size() > 1)
1585  action += " begin";
1586 
1587  for (const auto &elt : fieldTypes)
1588  action += "\n " + memName + elt.suffix + "[" + initvar + "] = `RANDOM;";
1589 
1590  if (fieldTypes.size() > 1)
1591  action += "\nend";
1592  addInitial(action, locInfo, /*ppCond*/ "RANDOMIZE_MEM_INIT",
1593  /*cond*/ "", /*partialOrder: After initVar decl*/ 11);
1594  }
1595 
1596  // Keep track of whether this mem is an even power of two or not.
1597  bool isPowerOfTwo = llvm::isPowerOf2_64(depth);
1598

1600  // them.
1601  SmallVector<std::pair<Identifier, MemOp::PortKind>, 2> ports;
1602  op.getPorts(ports);
1603 
1604  for (auto &port : ports) {
1605  auto portName = port.first;
1606 
1607  // Return the identifier emitted to Verilog that refers to the specified
1608  // field of the specified read/write port. This corresponds to the naming
1609  // of subfield's.
1610  // TODO(verilog dialect): eliminate subfields.
1611  auto getPortName = [&](StringRef fieldName) -> std::string {
1612  return memName + "_" + portName.str() + "_" + fieldName.str();
1613  };
1614

1616  case MemOp::PortKind::ReadWrite:
1617  op.emitOpError("readwrite ports should be lowered into separate read and "
1618  "write ports by previous passes");
1619  continue;
1620 
1621  case MemOp::PortKind::Read:
1622  // Emit an assign to the read port, using the address.
1623  // TODO(firrtl-spec): It appears that the clock signal on the read port is
1624  // ignored, why does it exist?
1625  if (!isPowerOfTwo) {
1626  indent() << "`ifndef RANDOMIZE_GARBAGE_ASSIGN\n";
1627  addIndent();
1628  }
1629

1631  indent() << "assign " << getPortName("data") << elt.suffix << " = "
1632  << memName << elt.suffix << '[' << getPortName("addr") << "];";
1633  emitLocationInfoAndNewLine(ops);
1634  }
1635

1637  reduceIndent();
1638  indent() << "`else\n";
1639  addIndent();
1640 
1641  for (const auto &elt : fieldTypes) {
1642  indent() << "assign " << getPortName("data") << elt.suffix << " = ";
1643  os << getPortName("addr") << " < " << depth << " ? ";
1644  os << memName << elt.suffix << '[' << getPortName("addr") << "]";
1645  os << " : `RANDOM;";
1646  emitLocationInfoAndNewLine(ops);
1647  }
1648 
1649  reduceIndent();
1650  indent() << "`endif // RANDOMIZE_GARBAGE_ASSIGN\n";
1651  }
1652 
1653  break;
1654 
1655  case MemOp::PortKind::Write:
1656  auto clockExpr = getPortName("clk");

1681 /// Most expressions are invalid to bit-select from in Verilog, but some things
1682 /// are ok. Return true if it is ok to inline bitselect from the result of this
1683 /// expression. It is conservatively correct to return false.
1684 static bool isOkToBitSelectFrom(Value v) {
1685  // Module ports are always ok to bit select from.
1686  auto *op = v.getDefiningOp();
1687  if (!op)
1688  return true;
1689 
1690  if (isa<SubfieldOp>(op))
1691  return true;
1692 
1693  // As{Non}PassivePrimOp is transparent.
1694  if (auto cast = dyn_cast<AsPassivePrimOp>(op))
1695  return isOkToBitSelectFrom(cast.getOperand());
1696  if (auto cast = dyn_cast<AsNonPassivePrimOp>(op))
1697  return isOkToBitSelectFrom(cast.getOperand());
1698 
1699  // TODO: We could handle concat and other operators here.
1700  return false;
1701 }
1702 
1703 /// Return true if we are unable to ever inline the specified operation. This
1704 /// happens because not all Verilog expressions are composable, notably you can
1705 /// only use bit selects like x[4:6] on simple expressions.
1706 static bool isExpressionUnableToInline(Operation *op) {
1707  // Scan the users of the operation to see if any of them need this to be
1708  // emitted out-of-line.
1709  for (auto user : op->getUsers()) {
1710  // Verilog bit selection is required by the standard to be:
1711  // "a vector, packed array, packed structure, parameter or concatenation".
1712  // It cannot be an arbitrary expression.
1713  if (isa<HeadPrimOp>(user) || isa<TailPrimOp>(user) ||
1714  isa<ShrPrimOp>(user) || isa<BitsPrimOp>(user))
1715  if (!isOkToBitSelectFrom(op->getResult(0)))
1716  return true;
1717 
1718  // Signed pad is emits a bit extract since it is a sign extend.
1719  if (auto pad = dyn_cast<PadPrimOp>(user)) {
1720  auto inType = getTypeOf<IntType>(pad.getOperand());
1721  auto inWidth = inType.getWidthOrSentinel();
1722  if (unsigned(inWidth) > pad.amount() &&
1723  !isOkToBitSelectFrom(op->getResult(0)))
1724  return true;
1725  }
1726  }
1727  return false;
1728 }

1730 /// Return true for operations that are always inlined.
1731 static bool isExpressionAlwaysInline(Operation *op) {
1732  if (isa<firrtl::ConstantOp>(op) || isa<SubfieldOp>(op))
1733  return true;
1734 
1735  // If this is a noop cast and the operand is always inlined, then the noop
1736  // cast is always inlined.
1737  if (isNoopCast(op))
1738  if (auto *operandOp = op->getOperand(0).getDefiningOp())
1739  return isExpressionAlwaysInline(operandOp);
1740 
1741  return false;

1744 /// Return true if this expression should be emitted inline into any statement
1745 /// that uses it.
1746 static bool isExpressionEmittedInline(Operation *op) {
1747  // If it isn't structurally possible to inline this expression, emit it out of
1748  // line.
1749  if (isExpressionUnableToInline(op))
1750  return false;

1753  if (isExpressionAlwaysInline(op))

1755

1757  return op->getResult(0).hasOneUse();

1768  return "reg";

1817  // base name may be unique but the suffixed names may not be. The right
1818  // way to solve this is to change the instances and mems in a new Verilog
1819  // dialect to use multiple return values, exposing the independent
