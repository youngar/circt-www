27 void FIRRTLType::print(raw_ostream &os) const {
28  auto printWidthQualifier = [&](Optional<int32_t> width) {
29  if (width)
30  os << '<' << width.getValue() << '>';
31  };
32 
33  TypeSwitch<FIRRTLType>(*this)
34  .Case<ClockType>([&](Type) { os << "clock"; })
35  .Case<ResetType>([&](Type) { os << "reset"; })
36  .Case<AsyncResetType>([&](Type) { os << "asyncreset"; })
37  .Case<SIntType>([&](SIntType sIntType) {
38  os << "sint";
39  printWidthQualifier(sIntType.getWidth());
40  })
41  .Case<UIntType>([&](UIntType uIntType) {
42  os << "uint";
43  printWidthQualifier(uIntType.getWidth());
44  })
45  .Case<AnalogType>([&](AnalogType analogType) {
46  os << "analog";
47  printWidthQualifier(analogType.getWidth());
48  })
49  .Case<FlipType>([&](FlipType flipType) {
50  os << "flip<";
51  flipType.getElementType().print(os);
52  os << '>';
53  })
54  .Case<BundleType>([&](BundleType bundleType) {
55  os << "bundle<";
56  llvm::interleaveComma(bundleType.getElements(), os,
57  [&](BundleType::BundleElement element) {
58  os << element.name << ": ";
59  element.type.print(os);
60  });
61  os << '>';
62  })
63  .Case<FVectorType>([&](FVectorType vectorType) {
64  os << "vector<";
65  vectorType.getElementType().print(os);
66  os << ", " << vectorType.getNumElements() << '>';
67  })
68  .Default([](Type) { assert(0 && "unknown dialect type to print"); });
69 }

75 /// type
76 /// ::= clock
77 /// ::= reset
78 /// ::= asyncreset
79 /// ::= sint ('<' int '>')?
80 /// ::= uint ('<' int '>')?
81 /// ::= analog ('<' int '>')?
82 /// ::= flip '<' type '>'
83 /// ::= bundle '<' (bundle-elt (',' bundle-elt)*)? '>'
84 /// ::= vector '<' type ',' int '>'
85 ///
86 /// bundle-elt ::= identifier ':' type
87 ///
88 static ParseResult parseType(FIRRTLType &result, DialectAsmParser &parser) {
89  StringRef name;
90  if (parser.parseKeyword(&name))
91  return failure();
92 
93  auto *context = parser.getBuilder().getContext();
94 
95  if (name.equals("clock")) {
96  return result = ClockType::get(context), success();
97  } else if (name.equals("reset")) {
98  return result = ResetType::get(context), success();
99  } else if (name.equals("asyncreset")) {
100  return result = AsyncResetType::get(context), success();
101  } else if (name.equals("sint") || name.equals("uint") ||
102  name.equals("analog")) {
103  // Parse the width specifier if it exists.
104  int32_t width = -1;
105  if (!parser.parseOptionalLess()) {
106  if (parser.parseInteger(width) || parser.parseGreater())
107  return failure();
108 
109  if (width < 0)
110  return parser.emitError(parser.getNameLoc(), "unknown width"),
111  failure();
112  }
113 
114  if (name.equals("sint"))
115  result = SIntType::get(context, width);
116  else if (name.equals("uint"))
117  result = UIntType::get(context, width);
118  else {
119  assert(name.equals("analog"));
120  result = AnalogType::get(context, width);
121  }
122  return success();
123  } else if (name.equals("flip")) {
124  FIRRTLType element;
125  if (parser.parseLess() || parseType(element, parser) ||
126  parser.parseGreater())
127  return failure();
128  return result = FlipType::get(element), success();
129  } else if (name.equals("bundle")) {
130  if (parser.parseLess())
131  return failure();
132 
133  SmallVector<BundleType::BundleElement, 4> elements;
134  if (parser.parseOptionalGreater()) {
135  // Parse all of the bundle-elt's.
136  do {
137  std::string nameStr;
138  StringRef name;
139  FIRRTLType type;
140 
141  // The 'name' can be an identifier or an integer.
142  auto parseIntOrStringName = [&]() -> ParseResult {
143  uint32_t fieldIntName;
144  auto intName = parser.parseOptionalInteger(fieldIntName);
145  if (intName.hasValue()) {
146  nameStr = llvm::utostr(fieldIntName);
147  name = nameStr;
148  return intName.getValue();
149  }
150 
151  // Otherwise must be an identifier.
152  return parser.parseKeyword(&name);
153  return success();
154  };
155 
156  if (parseIntOrStringName() || parser.parseColon() ||
157  parseType(type, parser))
158  return failure();
159 
160  elements.push_back({Identifier::get(name, context), type});
161  } while (!parser.parseOptionalComma());
162 
163  if (parser.parseGreater())
164  return failure();
165  }
166 
167  return result = BundleType::get(elements, context), success();
168  } else if (name.equals("vector")) {
169  FIRRTLType elementType;
170  unsigned width = 0;
171 
172  if (parser.parseLess() || parseType(elementType, parser) ||
173  parser.parseComma() || parser.parseInteger(width) ||
174  parser.parseGreater())
175  return failure();
176 
177  return result = FVectorType::get(elementType, width), success();
178  }
179 
180  return parser.emitError(parser.getNameLoc(), "unknown firrtl type"),
181  failure();
182 }

189  return result;

315 /// Helper to implement the equivalence logic for a pair of bundle elements.
316 /// Note that the FIRRTL spec requires bundle elements to have the same
317 /// orientation, but this only compares their passive types. The FIRRTL dialect
318 /// differs from the spec in how it uses flip types for module output ports and
319 /// canonicalizes flips in bundles, so only passive types can be compared here.
320 static bool areBundleElementsEquivalent(BundleType::BundleElement destElement,
321  BundleType::BundleElement srcElement) {
322  if (destElement.name != srcElement.name)
323  return false;
324 
325  return areTypesEquivalent(destElement.type, srcElement.type);
326 }
327 
328 /// Returns whether the two types are equivalent. See the FIRRTL spec for the
329 /// full definition of type equivalence. This predicate differs from the spec in
330 /// that it only compares passive types. Because of how the FIRRTL dialect uses
331 /// flip types in module ports and aggregates, this definition, unlike the spec,
332 /// ignores flips.
333 bool firrtl::areTypesEquivalent(FIRRTLType destType, FIRRTLType srcType) {
334  // Ensure we are comparing passive types.
335  if (!destType.isPassive())
336  destType = destType.getPassiveType();
337  if (!srcType.isPassive())
338  srcType = srcType.getPassiveType();
339 
340  // Reset types can be driven by UInt<1>, AsyncReset, or Reset types.
341  if (destType.isa<ResetType>())
342  return srcType.isResetType();
343 
344  // Reset types can drive UInt<1>, AsyncReset, or Reset types.
345  if (srcType.isa<ResetType>())
346  return destType.isResetType();
347 
348  // Vector types can be connected if they have the same size and element type.
349  auto destVectorType = destType.dyn_cast<FVectorType>();
350  auto srcVectorType = srcType.dyn_cast<FVectorType>();
351  if (destVectorType && srcVectorType)
352  return destVectorType.getNumElements() == srcVectorType.getNumElements() &&
353  areTypesEquivalent(destVectorType.getElementType(),
354  srcVectorType.getElementType());
355 
356  // Bundle types can be connected if they have the same size, element names,
357  // and element types.
358  auto destBundleType = destType.dyn_cast<BundleType>();
359  auto srcBundleType = srcType.dyn_cast<BundleType>();
360  if (destBundleType && srcBundleType) {
361  auto destElements = destBundleType.getElements();
362  auto srcElements = srcBundleType.getElements();
363  size_t numDestElements = destElements.size();
364  if (numDestElements != srcElements.size())
365  return false;
366 
367  for (size_t i = 0; i < numDestElements; ++i) {
368  auto destElement = destElements[i];
369  auto srcElement = srcElements[i];
370  if (!areBundleElementsEquivalent(destElement, srcElement))
371  return false;
372  }
373  }
374 
375  // Ground types can be connected if their passive, widthless versions
376  // are equal.
377  return destType.getWidthlessType() == srcType.getWidthlessType();
378 }

510 FIRRTLType FlipType::get(FIRRTLType element) {
511  // We maintain a canonical form for flip types, where we prefer to have the
512  // flip as far outward from an otherwise passive type as possible. If a flip
513  // is being used with an aggregate type that contains non-passive or analog
514  // elements, then it is forced into the elements to get the canonical form.
515  return TypeSwitch<FIRRTLType, FIRRTLType>(element)
516  .Case<ClockType, ResetType, AsyncResetType, SIntType, UIntType>(
517  [&](Type) {
518  // TODO: This should maintain a canonical form, digging any flips
519  // out of sub-types.
520  auto *context = element.getContext();
521  return Base::get(context, element);
522  })
523  .Case<FlipType>([](FlipType flipType) {
524  // flip(flip(x)) -> x
525  return flipType.getElementType();
526  })
527  .Case<AnalogType>([](AnalogType analogType) {
528  // flip(analog) -> analog.
529  return analogType;
530  })
531  .Case<BundleType>([&](BundleType bundleType) {
532  // If the bundle is passive and doesn't contain analog elements, then
533  // we're done because the flip will be at the outer level. Otherwise, it
534  // contains flip types recursively within itself that we should
535  // canonicalize.
536  auto properties = bundleType.getRecursiveTypeProperties();
537  if (properties.first && !properties.second) {
538  return Base::get(element.getContext(), element).cast<FIRRTLType>();
539  }
540 
541  return getFlippedBundleType(bundleType.getElements());
542  })
543  .Case<FVectorType>([&](FVectorType vectorType) {
544  // If the vector is passive and doesn't contain analog elements, then
545  // we're done because the flip will be at the outer level. Otherwise, it
546  // contains flip types recursively within itself that we should
547  // canonicalize.
548  auto properties = vectorType.getRecursiveTypeProperties();
549  if (properties.first && !properties.second) {
550  auto *context = element.getContext();
551  return Base::get(context, element).cast<FIRRTLType>();
552  }
553 
554  return FVectorType::get(get(vectorType.getElementType()),
555  vectorType.getNumElements());
556  })
557  .Default([](Type) {
558  llvm_unreachable("unknown FIRRTL type");
559  return FIRRTLType();
560  });
561 }

587 struct BundleTypeStorage : mlir::TypeStorage {
588  using KeyTy = ArrayRef<BundleType::BundleElement>;
589 
590  BundleTypeStorage(KeyTy elements)
591  : elements(elements.begin(), elements.end()) {
592  bool isPassive = true, containsAnalog = false;
593  for (auto &element : elements) {
594  auto type = element.type;
595  auto eltInfo = type.getRecursiveTypeProperties();
596  isPassive &= eltInfo.first;
597  containsAnalog |= eltInfo.second;
598  }
599  unsigned flags = 0;
600  if (isPassive)
601  flags |= IsPassiveBitMask;
602  if (containsAnalog)
603  flags |= ContainsAnalogBitMask;
604  passiveContainsAnalogTypeInfo.setInt(flags);
605  }
606 
607  bool operator==(const KeyTy &key) const { return key == KeyTy(elements); }
608 
609  static llvm::hash_code hashKey(const KeyTy &key) {
610  return llvm::hash_combine_range(key.begin(), key.end());
611  }
612 
613  static BundleTypeStorage *construct(TypeStorageAllocator &allocator,
614  KeyTy key) {
615  return new (allocator.allocate<BundleTypeStorage>()) BundleTypeStorage(key);
616  }
617 
618  SmallVector<BundleType::BundleElement, 4> elements;
619 
620  /// This holds two bits indicating whether the current type is passive and
621  /// if it contains an analog type, and can hold a pointer to a passive type if
622  /// not.
623  llvm::PointerIntPair<Type, 2, unsigned> passiveContainsAnalogTypeInfo;
624 };

630 FIRRTLType BundleType::get(ArrayRef<BundleElement> elements,
631  MLIRContext *context) {
632  // If all of the elements are flip types, then we canonicalize the flips to
633  // the outer level.
634  if (!elements.empty() &&
635  llvm::all_of(elements, [&](const BundleElement &elt) -> bool {
636  return elt.type.isa<FlipType>();
637  })) {
638  return FlipType::get(getFlippedBundleType(elements));
639  }
640 
641  return Base::get(context, elements);

655 /// Return this type with any flip types recursively removed from itself.
656 FIRRTLType BundleType::getPassiveType() {
657  auto *impl = getImpl();
658 
659  // If we've already determined and cached the passive type, use it.
660  if (auto passiveType = impl->passiveContainsAnalogTypeInfo.getPointer())
661  return passiveType.cast<FIRRTLType>();
662 
663  // If this type is already passive, use it and remember for next time.
664  if (impl->passiveContainsAnalogTypeInfo.getInt() & IsPassiveBitMask) {
665  impl->passiveContainsAnalogTypeInfo.setPointer(*this);
666  return *this;
667  }
668 
669  // Otherwise at least one element is non-passive, rebuild a passive version.
670  SmallVector<BundleType::BundleElement, 16> newElements;
671  newElements.reserve(impl->elements.size());
672  for (auto &elt : impl->elements) {
673  newElements.push_back({elt.name, elt.type.getPassiveType()});
674  }
675 
676  auto passiveType = BundleType::get(newElements, getContext());
677  impl->passiveContainsAnalogTypeInfo.setPointer(passiveType);
678  return passiveType;
679 }
