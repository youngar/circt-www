57 OpFoldResult DivPrimOp::fold(ArrayRef<Attribute> operands) {
58  APInt value;
59 
60  /// div(x, x) -> 1
61  ///
62  /// Division by zero is undefined in the FIRRTL specification. This
63  /// fold exploits that fact to optimize self division to one.
64  if (lhs() == rhs()) {
65  auto width = getType().getWidthOrSentinel();
66  if (width == -1)
67  width = 2;
68  if (width != 0)
69  return IntegerAttr::get(IntegerType::get(getContext(), width), 1);
70  }
71 
72  /// div(x, 1) -> x : (uint, uint) -> uint
73  ///
74  /// UInt division by one returns the numerator. SInt division can't
75  /// be folded here because it increases the return type bitwidth by
76  /// one and requires sign extension (a new op).
77  if (matchPattern(rhs(), m_FConstant(value)) && value.isOneValue() &&
78  lhs().getType() == getType())
79  return lhs();
80 
81  return {};
82 }

86  APInt value;
87 
88  /// and(x, 0) -> 0
89  if (matchPattern(rhs(), m_FConstant(value)) && value.isNullValue() &&
90  rhs().getType() == getType())
91  return rhs();
92 
93  /// and(x, -1) -> x
94  if (matchPattern(rhs(), m_FConstant(value)) && value.isAllOnesValue() &&
95  lhs().getType() == getType() && rhs().getType() == getType())
96  return lhs();
97 
98  /// and(x, x) -> x
99  if (lhs() == rhs() && rhs().getType() == getType())
100  return rhs();
101 
102  return constFoldBinaryOp<IntegerAttr>(operands,
103  [](APInt a, APInt b) { return a & b; });

146 OpFoldResult EQPrimOp::fold(ArrayRef<Attribute> operands) {
147  APInt value;
148 
149  if (matchPattern(rhs(), m_FConstant(value))) {
150  APInt lhsCst;
151  // Constant fold.
152  if (matchPattern(lhs(), m_FConstant(lhsCst)) &&
153  value.getBitWidth() == lhsCst.getBitWidth()) {
154  return getIntAttr(APInt(1, value == lhsCst), getContext());
155  }
156 
157  /// eq(x, 1) -> x when x is 1 bit.
158  /// TODO: Support SInt<1> on the LHS etc.
159  if (value.isAllOnesValue() && lhs().getType() == getType() &&
160  rhs().getType() == getType())
161  return lhs();
162 
163  /// TODO: eq(x, 0) -> not(x) when x is 1 bit.
164  /// TODO: eq(x, 0) -> not(orr(x)) when x is >1 bit
165  /// TODO: eq(x, ~0) -> andr(x)) when x is >1 bit
166  }
167 
168  return {};
169 }

216 void CatPrimOp::getCanonicalizationPatterns(OwningRewritePatternList &results,
217  MLIRContext *context) {
218  struct Folder final : public OpRewritePattern<CatPrimOp> {
219  using OpRewritePattern::OpRewritePattern;
220 
221  LogicalResult matchAndRewrite(CatPrimOp op,
222  PatternRewriter &rewriter) const override {
223  // cat(bits(x, ...), bits(x, ...)) -> bits(x ...) when the two ...'s are
224  // consequtive in the input.
225  if (auto lhsBits =
226  dyn_cast_or_null<BitsPrimOp>(op.lhs().getDefiningOp())) {
227  if (auto rhsBits =
228  dyn_cast_or_null<BitsPrimOp>(op.rhs().getDefiningOp())) {
229  if (lhsBits.input() == rhsBits.input() &&
230  lhsBits.lo() - 1 == rhsBits.hi()) {
231  rewriter.replaceOpWithNewOp<BitsPrimOp>(
232  op, op.getType(), lhsBits.input(), lhsBits.hi(), rhsBits.lo());
233  return success();
234  }
235  }
236  }
237  return failure();
238  }
239  };
240 
241  results.insert<Folder>(context);
242 }

244 OpFoldResult BitsPrimOp::fold(ArrayRef<Attribute> operands) {
245  auto inputType = input().getType().cast<FIRRTLType>();
246  // If we are extracting the entire input, then return it.
247  if (inputType == getType() &&
248  inputType.cast<IntType>().getWidthOrSentinel() != -1)
249  return input();
250 
251  // Constant fold.
252  APInt value;
253  if (inputType.cast<IntType>().hasWidth() &&
254  matchPattern(input(), m_FConstant(value)))
255  return getIntAttr(value.lshr(lo()).truncOrSelf(hi() - lo() + 1),
256  getContext());
257 
258  return {};
259 }

261 void BitsPrimOp::getCanonicalizationPatterns(OwningRewritePatternList &results,
262  MLIRContext *context) {
263  struct Folder final : public OpRewritePattern<BitsPrimOp> {
264  using OpRewritePattern::OpRewritePattern;
265 
266  LogicalResult matchAndRewrite(BitsPrimOp op,
267  PatternRewriter &rewriter) const override {
268 
269  auto inputOp = op.input().getDefiningOp();
270  // bits(bits(x, ...), ...) -> bits(x, ...).
271  if (auto innerBits = dyn_cast_or_null<BitsPrimOp>(inputOp)) {
272  auto newLo = op.lo() + innerBits.lo();
273  auto newHi = newLo + op.hi() - op.lo();
274  rewriter.replaceOpWithNewOp<BitsPrimOp>(op, innerBits.input(), newHi,
275  newLo);
276  return success();
277  }
278  return failure();
279  }
280  };

282  results.insert<Folder>(context);

285 /// Replace the specified operation with a 'bits' op from the specified hi/lo
286 /// bits. Insert a cast to handle the case where the original operation
287 /// returned a signed integer.
288 static void replaceWithBits(Operation *op, Value value, unsigned hiBit,
289  unsigned loBit, PatternRewriter &rewriter) {
290  auto resType = op->getResult(0).getType().cast<IntType>();
291  if (value.getType().cast<IntType>().getWidth() != resType.getWidth())
292  value = rewriter.create<BitsPrimOp>(op->getLoc(), value, hiBit, loBit);
293 
294  if (resType.isSigned() && !value.getType().cast<IntType>().isSigned()) {
295  value = rewriter.createOrFold<AsSIntPrimOp>(op->getLoc(), resType, value);
296  } else if (resType.isUnsigned() &&
297  !value.getType().cast<IntType>().isUnsigned()) {
298  value = rewriter.createOrFold<AsUIntPrimOp>(op->getLoc(), resType, value);
299  }
300  rewriter.replaceOp(op, value);
301 }

327 OpFoldResult MuxPrimOp::fold(ArrayRef<Attribute> operands) {
328  APInt value;
329 
330  /// mux(0/1, x, y) -> x or y
331  if (matchPattern(sel(), m_FConstant(value))) {
332  if (value.isNullValue() && low().getType() == getType())
333  return low();
334  if (!value.isNullValue() && high().getType() == getType())
335  return high();
336  }
337 
338  // mux(cond, x, x) -> x
339  if (high() == low())
340  return high();
341 
342  // mux(cond, x, cst)
343  if (matchPattern(low(), m_FConstant(value))) {
344  APInt c1;
345  // mux(cond, c1, c2)
346  if (matchPattern(high(), m_FConstant(c1))) {
347  // mux(cond, 1, 0) -> cond
348  if (c1.isOneValue() && value.isNullValue() &&
349  getType() == sel().getType())
350  return sel();
351 
352  // TODO: x ? ~0 : 0 -> sext(x)
353  // TODO: "x ? c1 : c2" -> many tricks
354  }
355  // TODO: "x ? a : 0" -> sext(x) & a
356  }
357 
358  // TODO: "x ? c1 : y" -> "~x ? y : c1"
359 
360  return {};
361 }

363 OpFoldResult PadPrimOp::fold(ArrayRef<Attribute> operands) {
364  auto input = this->input();
365 
366  // pad(x) -> x if the width doesn't change.
367  if (input.getType() == getType())
368  return input;
369 
370  // Need to know the input width.
371  auto inputType = input.getType().cast<IntType>();
372  int32_t width = inputType.getWidthOrSentinel();
373  if (width == -1)
374  return {};
375 
376  // Constant fold.
377  APInt value;
378  if (matchPattern(input, m_FConstant(value))) {
379  auto destWidth = getType().getWidthOrSentinel();
380  if (destWidth == -1)
381  return {};
382 
383  if (inputType.isSigned())
384  return getIntAttr(value.sext(destWidth), getContext());
385  return getIntAttr(value.zext(destWidth), getContext());
386  }
387 
388  return {};
389 }

391 OpFoldResult ShlPrimOp::fold(ArrayRef<Attribute> operands) {
392  auto input = this->input();
393  auto inputType = input.getType().cast<IntType>();
394  int shiftAmount = amount();
395 
396  // shl(x, 0) -> x
397  if (shiftAmount == 0)
398  return input;
399 
400  // Constant fold.
401  APInt value;
402  if (matchPattern(input, m_FConstant(value))) {
403  auto inputWidth = inputType.getWidthOrSentinel();
404  if (inputWidth != -1) {
405  auto resultWidth = inputWidth + shiftAmount;
406  shiftAmount = std::min(shiftAmount, resultWidth);
407  return getIntAttr(value.zext(resultWidth).shl(shiftAmount), getContext());
408  }
409  }
410  return {};
411 }

413 OpFoldResult ShrPrimOp::fold(ArrayRef<Attribute> operands) {
414  auto input = this->input();
415  auto inputType = input.getType().cast<IntType>();
416  int shiftAmount = amount();
417 
418  // shl(x, 0) -> x
419  if (shiftAmount == 0)
420  return input;
421 
422  auto inputWidth = inputType.getWidthOrSentinel();
423  if (inputWidth == -1)
424  return {};
425 
426  // shr(x, cst) where cst is all of x's bits and x is unsigned is 0.
427  // If x is signed, it is the sign bit.
428  if (shiftAmount >= inputWidth && inputType.isUnsigned())
429  return getIntAttr(APInt(1, 0), getContext());
430 
431  // Constant fold.
432  APInt value;
433  if (matchPattern(input, m_FConstant(value))) {
434  if (!inputType.isSigned())
435  value = value.lshr(std::min(shiftAmount, inputWidth));
436  else
437  value = value.ashr(std::min(shiftAmount, inputWidth - 1));
438  auto resultWidth = std::max(inputWidth - shiftAmount, 1);
439  return getIntAttr(value.truncOrSelf(resultWidth), getContext());
440  }
441  return {};
442 }

444 void ShrPrimOp::getCanonicalizationPatterns(OwningRewritePatternList &results,
445  MLIRContext *context) {
446  struct Folder final : public OpRewritePattern<ShrPrimOp> {
447  using OpRewritePattern::OpRewritePattern;
448 
449  LogicalResult matchAndRewrite(ShrPrimOp op,
450  PatternRewriter &rewriter) const override {
451  auto inputWidth =
452  op.input().getType().cast<IntType>().getWidthOrSentinel();
453  if (inputWidth == -1)
454  return failure();
455 
456  // If we know the input width, we can canonicalize this into a BitsPrimOp.
457  unsigned shiftAmount = op.amount();
458  if (int(shiftAmount) >= inputWidth) {
459  // shift(x, 32) => 0 when x has 32 bits. This is handled by fold().
460  if (op.getType().isUnsigned())
461  return failure();
462 
463  // Shifting a signed value by the full width is actually taking the sign
464  // bit. If the shift amount is greater than the input width, it is
465  // equivalent to shifting by the input width.
466  shiftAmount = inputWidth - 1;
467  }
468 
469  replaceWithBits(op, op.input(), inputWidth - 1, shiftAmount, rewriter);
470  return success();
471  }
472  };
473 
474  results.insert<Folder>(context);
475 }
