40 // This is a common base class for all FIRRTL types.
41 class FIRRTLType : public Type {
42 public:
43  void print(raw_ostream &os) const;
44 
45  /// Return true if this is a "passive" type - one that contains no "flip"
46  /// types recursively within itself.
47  bool isPassive() { return getRecursiveTypeProperties().first; }
48 
49  /// Return true if this is or contains an Analog type.
50  bool containsAnalog() { return getRecursiveTypeProperties().second; }
51 
52  /// Return a pair with the 'isPassive' and 'containsAnalog' bits.
53  std::pair<bool, bool> getRecursiveTypeProperties();
54 
55  /// Return this type with any flip types recursively removed from itself.
56  FIRRTLType getPassiveType();
57 
58  /// Return this type with all ground types replaced with UInt<1>. This is
59  /// used for `mem` operations.
60  FIRRTLType getMaskType();
61 
62  /// Return this type with widths of all ground types removed. This
63  /// enables two types to be compared by structure and name ignoring
64  /// widths.
65  FIRRTLType getWidthlessType();
66 
67  /// If this is an IntType, AnalogType, or sugar type for a single bit (Clock,
68  /// Reset, etc) then return the bitwidth. Return -1 if the is one of these
69  /// types but without a specified bitwidth. Return -2 if this isn't a simple
70  /// type.
71  int32_t getBitWidthOrSentinel() const;
72 
73  /// Support method to enable LLVM-style type casting.
74  static bool classof(Type type) {
75  return llvm::isa<FIRRTLDialect>(type.getDialect());
76  }
77 
78  /// Return true if this is a valid "reset" type.
79  bool isResetType();
80 
81 protected:
82  using Type::Type;
83 };

128 template <typename ConcreteType, typename ParentType>
129 class WidthQualifiedType
130  : public FIRRTLType::TypeBase<ConcreteType, ParentType,
131  detail::WidthTypeStorage> {
132 public:
133  using FIRRTLType::TypeBase<ConcreteType, ParentType,
134  detail::WidthTypeStorage>::Base::Base;
135 
136  /// Return the width of this type, or -1 if it has none specified.
137  int32_t getWidthOrSentinel() {
138  auto width = static_cast<ConcreteType *>(this)->getWidth();
139  return width.hasValue() ? width.getValue() : -1;
140  }
141 };

146 /// This is the common base class between SIntType and UIntType.
147 class IntType : public FIRRTLType {
148 public:
149  using FIRRTLType::FIRRTLType;
150 
151  /// Return a SIntType or UInt type with the specified signedness and width.
152  static IntType get(MLIRContext *context, bool isSigned, int32_t width = -1);
153 
154  bool isSigned() { return isa<SIntType>(); }
155  bool isUnsigned() { return isa<UIntType>(); }
156 
157  /// Return true if this integer type has a known width.
158  bool hasWidth() { return getWidth().hasValue(); }
159 
160  /// Return the bitwidth of this type or None if unknown.
161  Optional<int32_t> getWidth();
162 
163  /// Return the width of this type, or -1 if it has none specified.
164  int32_t getWidthOrSentinel() {
165  auto width = getWidth();
166  return width.hasValue() ? width.getValue() : -1;
167  }
168 
169  static bool classof(Type type) {
170  return type.isa<SIntType>() || type.isa<UIntType>();
171  }
172 };

228 /// BundleType is an aggregate of named elements. This is effectively a struct
229 /// for FIRRTL.
230 class BundleType : public FIRRTLType::TypeBase<BundleType, FIRRTLType,
231  detail::BundleTypeStorage> {
232 public:
233  using Base::Base;
234 
235  // Each element of a bundle, which is a name and type.
236  struct BundleElement {
237  Identifier name;
238  FIRRTLType type;
239 
240  BundleElement(Identifier name, FIRRTLType type) : name(name), type(type) {}
241 
242  bool operator==(const BundleElement &rhs) const {
243  return name == rhs.name && type == rhs.type;
244  }
245  bool operator!=(const BundleElement &rhs) const { return !operator==(rhs); }
246  };
247 
248  static FIRRTLType get(ArrayRef<BundleElement> elements, MLIRContext *context);
