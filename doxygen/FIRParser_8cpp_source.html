36 /// Return true if this is a useless temporary name produced by FIRRTL. We
37 /// drop these as they don't convey semantic meaning.
38 static bool isUselessName(StringRef name) {
39  // Ignore _T and _T_123
40  if (name.startswith("_T")) {
41  if (name.size() == 2)
42  return true;
43  return name.size() > 3 && name[2] == '_' && llvm::isDigit(name[3]);
44  }
45 
46  // Ignore _GEN and _GEN_123, these are produced by Namespace.scala.
47  if (name.startswith("_GEN")) {
48  if (name.size() == 4)
49  return true;
50  return name.size() > 5 && name[4] == '_' && llvm::isDigit(name[5]);
51  }
52  return false;
53 }

308 /// Parse an @info marker if present. If so, apply the symbolic location
309 /// specified it to all of the operations listed in subOps.
310 ///
311 /// info ::= FileInfo
312 ///
313 ParseResult FIRParser::parseOptionalInfo(LocWithInfo &result,
314  ArrayRef<Operation *> subOps) {
315  if (getToken().isNot(FIRToken::fileinfo))
316  return success();
317 
318  auto loc = getToken().getLoc();
319 
320  // See if we can parse this token into a File/Line/Column record. If not,
321  // just ignore it with a warning.
322  auto unknownFormat = [&]() -> ParseResult {
323  mlir::emitWarning(translateLocation(loc),
324  "ignoring unknown @ info record format");
325  return success();
326  };
327 
328  auto spelling = getTokenSpelling();
329  consumeToken(FIRToken::fileinfo);
330 
331  // The spelling of the token looks something like "@[Decoupled.scala 221:8]".
332  if (!spelling.startswith("@[") || !spelling.endswith("]"))
333  return unknownFormat();
334 
335  spelling = spelling.drop_front(2).drop_back(1);
336 
337  // Decode the locator in "spelling", returning the filename and filling in
338  // lineNo and colNo on success. On failure, this returns an empty filename.
339  auto decodeLocator = [&](StringRef input, unsigned &resultLineNo,
340  unsigned &resultColNo) -> StringRef {
341  // Split at the last space.
342  auto spaceLoc = input.find_last_of(' ');
343  if (spaceLoc == StringRef::npos)
344  return {};
345 
346  auto filename = input.take_front(spaceLoc);
347  auto lineAndColumn = input.drop_front(spaceLoc + 1);
348 
349  // Decode the line/column. If the colon is missing, then it will be empty
350  // here.
351  StringRef lineStr, colStr;
352  std::tie(lineStr, colStr) = lineAndColumn.split(':');
353 
354  // Decode the line number and the column number if present.
355  if (lineStr.getAsInteger(10, resultLineNo))
356  return {};
357  if (!colStr.empty() && colStr.getAsInteger(10, resultColNo))
358  return {};
359  return filename;
360  };
361 
362  // Decode the locator spelling, reporting an error if it is malformed.
363  unsigned lineNo = 0, columnNo = 0;
364  StringRef filename = decodeLocator(spelling, lineNo, columnNo);
365  if (filename.empty())
366  return unknownFormat();
367 
368  // If info locators are ignored, don't actually apply them. We still do all
369  // the verification above though.
370  if (state.options.ignoreInfoLocators)
371  return success();
372 
373  // Compound locators will be combined with spaces, like:
374  // @[Foo.scala 123:4 Bar.scala 309:14]
375  // and at this point will be parsed as a-long-string-with-two-spaces at
376  // 309:14. We'd like to parse this into two things and represent it as an
377  // MLIR fused locator, but we want to be conservatively safe for filenames
378  // that have a space in it. As such, we are careful to make sure we can
379  // decode the filename/loc of the result. If so, we accumulate results,
380  // backward, in this vector.
381  SmallVector<Location> extraLocs;
382  auto spaceLoc = filename.find_last_of(' ');
383  while (spaceLoc != StringRef::npos) {
384  // Try decoding the thing before the space. Validates that there is another
385  // space and that the file/line can be decoded in that substring.
386  unsigned nextLineNo = 0, nextColumnNo = 0;
387  auto nextFilename =
388  decodeLocator(filename.take_front(spaceLoc), nextLineNo, nextColumnNo);
389 
390  // On failure we didn't have a joined locator.
391  if (nextFilename.empty())
392  break;
393 
394  // On success, remember what we already parsed (Bar.Scala / 309:14), and
395  // move on to the next chunk.
396  auto loc = FileLineColLoc::get(filename.drop_front(spaceLoc + 1), lineNo,
397  columnNo, getContext());
398  extraLocs.push_back(loc);
399  filename = nextFilename;
400  lineNo = nextLineNo;
401  columnNo = nextColumnNo;
402  spaceLoc = filename.find_last_of(' ');
403  }
404 
405  Location resultLoc =
406  FileLineColLoc::get(filename, lineNo, columnNo, getContext());
407  if (!extraLocs.empty()) {
408  extraLocs.push_back(resultLoc);
409  std::reverse(extraLocs.begin(), extraLocs.end());
410  resultLoc = FusedLoc::get(extraLocs, getContext());
411  }
412  result.setInfoLocation(resultLoc);
413 
414  // Now that we have a symbolic location, apply it to any subOps specified.
415  for (auto *op : subOps) {
416  op->setLoc(resultLoc);
417  }
418 
419  return success();
420 }

426 /// intLit ::= UnsignedInt
427 /// ::= SignedInt
428 /// ::= HexLit
429 /// ::= OctalLit
430 /// ::= BinaryLit
431 /// HexLit ::= '"' 'h' ( '+' | '-' )? ( HexDigit )+ '"'
432 /// OctalLit ::= '"' 'o' ( '+' | '-' )? ( OctalDigit )+ '"'
433 /// BinaryLit ::= '"' 'b' ( '+' | '-' )? ( BinaryDigit )+ '"'
434 ///
435 ParseResult FIRParser::parseIntLit(APInt &result, const Twine &message) {
436  auto spelling = getTokenSpelling();
437  switch (getToken().getKind()) {
438  case FIRToken::integer:
439  if (spelling.getAsInteger(10, result))
440  return emitError(message), failure();
441 
442  // Make sure that the returned APInt has a zero at the top so clients don't
443  // confuse it with a negative number.
444  if (result.isNegative())
445  result = result.zext(result.getBitWidth() + 1);
446 
447  consumeToken(FIRToken::integer);
448  return success();
449 
450  case FIRToken::signed_integer:
451  assert(spelling[0] == '+' || spelling[0] == '-');
452  if (spelling.drop_front().getAsInteger(10, result))
453  return emitError(message), failure();
454 
455  // Make sure that the returned APInt has a zero at the top so clients don't
456  // confuse it with a negative number.
457  if (result.isNegative())
458  result = result.zext(result.getBitWidth() + 1);
459 
460  if (spelling[0] == '-')
461  result = -result;
462  consumeToken(FIRToken::signed_integer);
463  return success();
464  case FIRToken::string: {
465  // Drop the quotes.
466  assert(spelling.front() == '"' && spelling.back() == '"');
467  spelling = spelling.drop_back().drop_front();
468 
469  // Decode the base.
470  unsigned base;
471  switch (spelling.empty() ? ' ' : spelling.front()) {
472  case 'h':
473  base = 16;
474  break;
475  case 'o':
476  base = 8;
477  break;
478  case 'b':
479  base = 2;
480  break;
481  default:
482  return emitError("expected base specifier (h/o/b) in integer literal"),
483  failure();
484  }
485  spelling = spelling.drop_front();
486 
487  // Handle the optional sign.
488  bool isNegative = false;
489  if (!spelling.empty() && spelling.front() == '+')
490  spelling = spelling.drop_front();
491  else if (!spelling.empty() && spelling.front() == '-') {
492  isNegative = true;
493  spelling = spelling.drop_front();
494  }
495 
496  // Parse the digits.
497  if (spelling.empty())
498  return emitError("expected digits in integer literal"), failure();
499 
500  if (spelling.getAsInteger(base, result))
501  return emitError("invalid character in integer literal"), failure();
502 
503  // We just parsed the positive version of this number. Make sure it has
504  // a zero at the top so clients don't confuse it with a negative number and
505  // so the negation (in the case of a negative sign) doesn't overflow.
506  if (result.isNegative())
507  result = result.zext(result.getBitWidth() + 1);
508 
509  if (isNegative)
510  result = -result;
511 
512  consumeToken(FIRToken::string);
513  return success();
514  }
515 
516  default:
517  return emitError("expected integer literal"), failure();
518  }
519 }

611 /// ::= type '[' intLit ']'
612 ///

614 ///
615 ParseResult FIRParser::parseType(FIRRTLType &result, const Twine &message) {
616  switch (getToken().getKind()) {
617  default:
618  return emitError(message), failure();

621  consumeToken(FIRToken::kw_Clock);
622  result = ClockType::get(getContext());
623  break;

625  case FIRToken::kw_Reset:
626  consumeToken(FIRToken::kw_Reset);
627  result = ResetType::get(getContext());
628  break;
629 
630  case FIRToken::kw_AsyncReset:
631  consumeToken(FIRToken::kw_AsyncReset);
632  result = AsyncResetType::get(getContext());
633  break;
634 
635  case FIRToken::kw_UInt:
636  case FIRToken::kw_SInt:
637  case FIRToken::kw_Analog: {
638  auto kind = getToken().getKind();
639  consumeToken();

642  int32_t width;

646  if (kind == FIRToken::kw_SInt)
647  result = SIntType::get(getContext(), width);
648  else if (kind == FIRToken::kw_UInt)
649  result = UIntType::get(getContext(), width);
650  else {

652  result = AnalogType::get(getContext(), width);
653  }
654  break;

657  case FIRToken::l_brace: {

936 /// Parse the 'exp' grammar, returning all of the suboperations in the
937 /// specified vector, and the ultimate SSA value in value.
938 ///
939 /// exp ::= id // Ref
940 /// ::= prim
941 /// ::= integer-literal-exp
942 /// ::= exp '.' fieldId
943 /// ::= exp '[' intLit ']'
944 /// XX ::= exp '.' DoubleLit // TODO Workaround for #470
945 /// ::= exp '[' exp ']'
946 ///
947 ParseResult FIRStmtParser::parseExp(Value &result, SubOpVector &subOps,
948  const Twine &message) {
949  switch (getToken().getKind()) {
950 
951  // Handle all the primitive ops: primop exp* intLit* ')'
952 #define TOK_LPKEYWORD(SPELLING) case FIRToken::lp_##SPELLING:
953 #include "FIRTokenKinds.def"
954  if (parsePrimExp(result, subOps))
955  return failure();
956  break;
957 
958  case FIRToken::kw_UInt:
959  case FIRToken::kw_SInt:
960  if (parseIntegerLiteralExp(result, subOps))
961  return failure();
962  break;
963 
964  // Otherwise there are a bunch of keywords that are treated as identifiers
965  // try them.
966  case FIRToken::identifier: // exp ::= id
967  default: {
968  StringRef name;
969  auto loc = getToken().getLoc();
970  SymbolValueEntry symtabEntry;
971  if (parseId(name, message) || lookupSymbolEntry(symtabEntry, name, loc))
972  return failure();
973 
974  // If we looked up a normal value, then we're done.
975  if (auto val = symtabEntry.dyn_cast<Value>()) {
976  result = val;
977  break;
978  }
979 
980  // Otherwise we referred to an implicitly bundled value. We *must* be in
981  // the midst of processing a field ID reference. If not, this is an error.
982  StringRef fieldName;
983  if (parseToken(FIRToken::period, "expected '.' in field reference") ||
984  parseFieldId(fieldName, "expected field name"))
985  return failure();
986 
987  result = Value();
988 
989  // Look up the fieldName in the unbundled info record. Indices are biased
990  // by one intentially to avoid using index #0.
991  unsigned unbundledId = symtabEntry.get<UnbundledID>() - 1;
992  assert(unbundledId < unbundledValues.size());
993  UnbundledValueEntry &entry = unbundledValues[unbundledId];
994  for (auto elt : entry) {
995  if (elt.first.cast<StringAttr>().getValue() == fieldName) {
996  result = elt.second;
997  break;
998  }
999  }
1000  if (!result) {
1001  emitError(loc, "use of invalid field name '")
1002  << fieldName << "' on bundle value";
1003  return failure();
1004  }
1005  break;
1006  }
1007  }
1008 
1009  return parseOptionalExpPostscript(result, subOps);
1010 }

1012 /// Parse the postfix productions of expression after the leading expression
1013 /// has been parsed.
1014 ///
1015 /// exp ::= exp '.' fieldId
1016 /// ::= exp '[' intLit ']'
1017 /// XX ::= exp '.' DoubleLit // TODO Workaround for #470
1018 /// ::= exp '[' exp ']'
1019 ParseResult FIRStmtParser::parseOptionalExpPostscript(Value &result,
1020  SubOpVector &subOps) {
1021 
1022  // Handle postfix expressions.
1023  while (1) {
1024  // Subfield: exp ::= exp '.' fieldId
1025  if (consumeIf(FIRToken::period)) {
1026  if (parsePostFixFieldId(result, subOps))
1027  return failure();
1028 
1029  continue;
1030  }
1031 
1032  // Subindex: exp ::= exp '[' intLit ']' | exp '[' exp ']'
1033  if (consumeIf(FIRToken::l_square)) {
1034  if (getToken().isAny(FIRToken::integer, FIRToken::string)) {
1035  if (parsePostFixIntSubscript(result, subOps))
1036  return failure();
1037  continue;
1038  }
1039  if (parsePostFixDynamicSubscript(result, subOps))
1040  return failure();
1041 
1042  continue;
1043  }
1044 
1045  return success();
1046  }
1047 }

1142 /// prim ::= primop exp* intLit* ')'
1143 ParseResult FIRStmtParser::parsePrimExp(Value &result, SubOpVector &subOps) {
1144  auto kind = getToken().getKind();
1145  auto loc = getToken().getLoc();
1146  consumeToken();
1147 
1148  // Parse the operands and constant integer arguments.
1149  SmallVector<Value, 4> operands;
1150  SmallVector<int32_t, 4> integers;
1151  if (parseListUntil(FIRToken::r_paren, [&]() -> ParseResult {
1152  // Handle the integer constant case if present.
1153  if (getToken().isAny(FIRToken::integer, FIRToken::signed_integer,
1154  FIRToken::string)) {
1155  integers.push_back(0);
1156  return parseIntLit(integers.back(), "expected integer");
1157  }
1158 
1159  // Otherwise it must be a value operand. These must all come before the
1160  // integers.
1161  if (!integers.empty())
1162  return emitError("expected more integer constants"), failure();
1163 
1164  Value operand;
1165  if (parseExp(operand, subOps,
1166  "expected expression in primitive operand"))
1167  return failure();
1168 
1169  // If the operand contains a flip, strip it out with an asPassive op.
1170  if (!operand.getType().cast<FIRRTLType>().isPassive())
1171  operand = convertToPassive(operand, translateLocation(loc));
1172 
1173  operands.push_back(operand);
1174  return success();
1175  }))
1176  return failure();
1177 
1178  SmallVector<FIRRTLType, 4> opTypes;
1179  for (auto v : operands)
1180  opTypes.push_back(v.getType().cast<FIRRTLType>());
1181 
1182  SmallVector<StringRef, 2> attrNames;
1183  switch (kind) {
1184  default:
1185  break;
1186  case FIRToken::lp_bits:
1187  attrNames.push_back("hi");
1188  attrNames.push_back("lo");
1189  break;
1190  case FIRToken::lp_head:
1191  case FIRToken::lp_pad:
1192  case FIRToken::lp_shl:
1193  case FIRToken::lp_shr:
1194  case FIRToken::lp_tail:
1195  attrNames.push_back("amount");
1196  break;
1197  }
1198 
1199  if (integers.size() != attrNames.size()) {
1200  emitError(loc,
1201  "expected " + Twine(attrNames.size()) + " constant arguments");
1202  return failure();
1203  }
1204 
1205  NamedAttrList attrs;
1206  for (size_t i = 0, e = attrNames.size(); i != e; ++i)
1207  attrs.append(attrNames[i], builder.getI32IntegerAttr(integers[i]));
1208 
1209  switch (kind) {
1210  default:
1211  emitError(loc, "primitive not supported yet");
1212  return failure();
1213 
1214 #define TOK_LPKEYWORD_PRIM(SPELLING, CLASS) \
1215  case FIRToken::lp_##SPELLING: { \
1216  auto resultTy = \
1217  CLASS::getResultType(opTypes, integers, translateLocation(loc)); \
1218  if (!resultTy) \
1219  return failure(); \
1220  result = builder.create<CLASS>(translateLocation(loc), resultTy, \
1221  ValueRange(operands), attrs); \
1222  break; \
1223  }
1224 #include "FIRTokenKinds.def"
1225  }
1226 
1227  subOps.push_back(result.getDefiningOp());
1228  return success();
1229 }

1231 /// integer-literal-exp ::= 'UInt' optional-width '(' intLit ')'
1232 /// ::= 'SInt' optional-width '(' intLit ')'
1233 ParseResult FIRStmtParser::parseIntegerLiteralExp(Value &result,
1234  SubOpVector &subOps) {
1235  bool isSigned = getToken().is(FIRToken::kw_SInt);
1236  auto loc = getToken().getLoc();
1237  consumeToken();
1238 
1239  // Parse a width specifier if present.
1240  int32_t width;
1241  APInt value;
1242  if (parseOptionalWidth(width) ||
1243  parseToken(FIRToken::l_paren, "expected '(' in integer expression") ||
1244  parseIntLit(value, "expected integer value") ||
1245  parseToken(FIRToken::r_paren, "expected ')' in integer expression"))
1246  return failure();
1247 
1248  if (width == 0)
1249  return emitError(loc, "zero bit constants are not allowed"), failure();
1250 
1251  // Construct an integer attribute of the right width.
1252  auto type = IntType::get(builder.getContext(), isSigned, width);
1253 
1254  IntegerType::SignednessSemantics signedness;
1255  if (type.isSigned()) {
1256  signedness = IntegerType::Signed;
1257  if (width != -1) {
1258  // Check for overlow if we are truncating bits.
1259  if (unsigned(width) < value.getBitWidth() &&
1260  value.getNumSignBits() <= value.getBitWidth() - width) {
1261  return emitError(loc, "initializer too wide for declared width"),
1262  failure();
1263  }
1264 
1265  value = value.sextOrTrunc(width);
1266  }
1267  } else {
1268  signedness = IntegerType::Unsigned;
1269  if (width != -1) {
1270  // Check for overlow if we are truncating bits.
1271  if (unsigned(width) < value.getBitWidth() &&
1272  value.countLeadingZeros() < value.getBitWidth() - width) {
1273  return emitError(loc, "initializer too wide for declared width"),
1274  failure();
1275  }
1276  value = value.zextOrTrunc(width);
1277  }
1278  }
1279 
1280  Type attrType =
1281  IntegerType::get(type.getContext(), value.getBitWidth(), signedness);
1282  auto attr = builder.getIntegerAttr(attrType, value);
1283 
1284  // Check to see if we've already created this constant. If so, reuse it.
1285  auto &entry = moduleContext.constantCache[{attr, type}];
1286  if (entry) {
1287  // If we already had an entry, reuse it.
1288  result = entry;
1289  return success();
1290  }
1291 
1292  // Make sure to insert constants at the top level of the module to maintain
1293  // dominance.
1294  OpBuilder::InsertPoint savedIP;
1295 
1296  auto parentOp = builder.getInsertionBlock()->getParentOp();
1297  if (!isa<FModuleOp>(parentOp)) {
1298  savedIP = builder.saveInsertionPoint();
1299  while (!isa<FModuleOp>(parentOp)) {
1300  builder.setInsertionPoint(parentOp);
1301  parentOp = builder.getInsertionBlock()->getParentOp();
1302  }
1303  }
1304 
1305  auto op = builder.create<ConstantOp>(translateLocation(loc), type, value);
1306  entry = op;
1307  subOps.push_back(op);
1308  result = op;
1309 
1310  if (savedIP.isSet())
1311  builder.setInsertionPoint(savedIP.getBlock(), savedIP.getPoint());
1312  return success();
1313 }

1315 /// The .fir grammar has the annoying property where:
1316 /// 1) some statements start with keywords
1317 /// 2) some start with an expression
1318 /// 3) it allows the 'reference' expression to either be an identifier or a
1319 /// keyword.
1320 ///
1321 /// One example of this is something like, where this is not a register decl:
1322 /// reg <- thing
1323 ///
1324 /// Solving this requires lookahead to the second token. We handle it by
1325 /// factoring the lookahead inline into the code to keep the parser fast.
1326 ///
1327 /// As such, statements that start with a leading keyword call this method to
1328 /// check to see if the keyword they consumed was actually the start of an
1329 /// expression. If so, they parse the expression-based statement and return the
1330 /// parser result. If not, they return None and the statement is parsed like
1331 /// normal.
1332 Optional<ParseResult>
1333 FIRStmtParser::parseExpWithLeadingKeyword(StringRef keyword,
1334  const LocWithInfo &info) {
1335 
1336  switch (getToken().getKind()) {
1337  default:
1338  // This isn't part of an expression, and isn't part of a statement.
1339  return None;
1340 
1341  case FIRToken::period: // exp `.` identifier
1342  case FIRToken::l_square: // exp `[` index `]`
1343  case FIRToken::kw_is: // exp is invalid
1344  case FIRToken::less_equal: // exp <= thing
1345  case FIRToken::less_minus: // exp <- thing
1346  break;
1347  }
1348 
1349  Value lhs;
1350  SmallVector<Operation *, 8> subOps;
1351  if (lookupSymbolEntry(lhs, keyword, info.getFIRLoc()) ||
1352  parseOptionalExpPostscript(lhs, subOps))

1485  consumeToken();

1488  // that.

1493  StringRef memName;
1494  Value memory, indexExp, clock;
1495  SmallVector<Operation *, 8> subOps;
1496  if (parseToken(FIRToken::kw_mport, "expected 'mport' in memory port") ||
1497  parseId(resultValue, "expected result name") ||
1498  parseToken(FIRToken::equal, "expected '=' in memory port") ||
1499  parseId(memName, "expected memory name") ||
1500  lookupSymbolEntry(memory, memName, info.getFIRLoc()) ||
1501  parseToken(FIRToken::l_square, "expected '[' in memory port") ||
1502  parseExp(indexExp, subOps, "expected index expression") ||
1503  parseToken(FIRToken::r_square, "expected ']' in memory port") ||
1504  parseExp(clock, subOps, "expected clock expression") ||
1505  parseOptionalInfo(info, subOps))

1507 
1508  auto memVType = memory.getType().dyn_cast<FVectorType>();
1509  if (!memVType)
1510  return emitError(info.getFIRLoc(), "memory should have vector type");

1514  indexExp, clock, direction,
1515  filterUselessName(resultValue));
1516 
1517  // TODO(firrtl scala bug): If the next operation is a skip, just eat it if it
1518  // is at the same indent level as us. This is a horrible hack on top of the
1519  // following hack to work around a Scala bug.

1522  nextIndent.hasValue() && mdirIndent.getValue() == nextIndent.getValue()) {

1524  return failure();

1526  nextIndent = getIndentation();

1531  // is a problem for us, because the index expression and clock may be defined
1532  // in a nested expression. We can't really tell when this is happening
1533  // without unbounded lookahead either.
1534  //
1535  // Fortunately, this seems to happen in very idiomatic cases, where the
1536  // mport happens at the end of a when block. We detect this situation by
1537  // seeing if the next statement is indented less than our memport - if so,
1538  // this is the last statement at the end of the 'when' block. Trigger a
1539  // hacky workaround just in this case.
1540  if (mdirIndent.hasValue() && nextIndent.hasValue() &&
1541  mdirIndent.getValue() > nextIndent.getValue()) {
1542  auto memNameId = Identifier::get(memName, getContext());

1544  // To make this even more gross, we have no efficient way to figure out
1545  // what scope a value lives in our scoped hash table. We keep a shadow
1546  // table to track this.

1550  return failure();
1551  }
1552

1554  // some IR hackery. Create a wire for the resultValue name right before
1555  // the mem in question, inject its name into that scope, then connect
1556  // the output of the mport to it.
1557  if (scopeAndOperation.first != symbolTable.getCurScope()) {
1558  OpBuilder memOpBuilder(scopeAndOperation.second);
1559 
1560  auto wireHack = memOpBuilder.create<WireOp>(
1561  info.getLoc(), result.getType(), StringAttr());
1562  builder.create<ConnectOp>(info.getLoc(), wireHack, result);
1563 
1564  // Inject this the wire's name into the same scope as the memory.
1565  symbolTable.insertIntoScope(
1566  scopeAndOperation.first,
1567  Identifier::get(resultValue.getValue(), getContext()),
1568  {info.getFIRLoc(), SymbolValueEntry(wireHack)});
1569  return success();
1570  }
1571  }
1572 
1573  return addSymbolEntry(resultValue.getValue(), result, info.getFIRLoc());
1574 }

1717 /// when ::= 'when' exp ':' info? suite? ('else' ( when | ':' info? suite?)
1718 /// )? suite ::= simple_stmt | INDENT simple_stmt+ DEDENT
1719 ParseResult FIRStmtParser::parseWhen(unsigned whenIndent) {
1720  LocWithInfo info(getToken().getLoc(), this);
1721  consumeToken(FIRToken::kw_when);
1722 
1723  // If this was actually the start of a connect or something else handle
1724  // that.
1725  if (auto isExpr = parseExpWithLeadingKeyword("when", info))
1726  return isExpr.getValue();
1727 
1728  Value condition;
1729  SmallVector<Operation *, 8> subOps;
1730  if (parseExp(condition, subOps, "expected condition in 'when'") ||
1731  parseToken(FIRToken::colon, "expected ':' in when") ||
1732  parseOptionalInfo(info, subOps))
1733  return failure();
1734 
1735  condition = convertToPassive(condition, info.getLoc());
1736 
1737  // Create the IR representation for the when.
1738  auto whenStmt = builder.create<WhenOp>(info.getLoc(), condition,
1739  /*createElse*/ false);
1740 
1741  // This is a function to parse a suite body.
1742  auto parseSuite = [&](OpBuilder builder) -> ParseResult {
1743  // Declarations within the suite are scoped to within the suite.
1744  SymbolTable::ScopeTy suiteScope(symbolTable);
1745  MemoryScopeTable::ScopeTy suiteMemoryScope(memoryScopeTable);
1746 
1747  // After parsing the when region, we can release any new entries in
1748  // unbundledValues since the symbol table entries that refer to them will be
1749  // gone.
1750  struct UnbundledValueRestorer {
1751  UnbundledValuesList &list;
1752  size_t startingSize;
1753  UnbundledValueRestorer(UnbundledValuesList &list) : list(list) {
1754  startingSize = list.size();
1755  }
1756  ~UnbundledValueRestorer() { list.resize(startingSize); }
1757  } x(unbundledValues);
1758 
1759  // We parse the substatements into their own parser, so they get inserted
1760  // into the specified 'when' region.
1761  FIRStmtParser subParser(builder, *this, moduleContext);
1762 
1763  // Figure out whether the body is a single statement or a nested one.
1764  auto stmtIndent = getIndentation();
1765 
1766  // Parsing a single statment is straightforward.
1767  if (!stmtIndent.hasValue())
1768  return subParser.parseSimpleStmt(whenIndent);
1769 
1770  if (stmtIndent.getValue() <= whenIndent)
1771  return emitError("statement must be indented more than 'when'"),
1772  failure();
1773 
1774  // Parse a block of statements that are indented more than the when.
1775  return subParser.parseSimpleStmtBlock(whenIndent);
1776  };
1777 
1778  // Parse the 'then' body into the 'then' region.
1779  if (parseSuite(whenStmt.getThenBodyBuilder()))
1780  return failure();
1781 
1782  // If the else is present, handle it otherwise we're done.
1783  if (getToken().isNot(FIRToken::kw_else))
1784  return success();
1785 
1786  // If the 'else' is less indented than the when, then it must belong to some
1787  // containing 'when'.
1788  auto elseIndent = getIndentation();
1789  if (elseIndent.hasValue() && elseIndent.getValue() < whenIndent)
1790  return success();
1791 
1792  LocWithInfo elseInfo(getToken().getLoc(), this);
1793  consumeToken(FIRToken::kw_else);
1794 
1795  // Create an else block to parse into.
1796  whenStmt.createElseRegion();
1797 
1798  // If we have the ':' form, then handle it.
1799  if (getToken().is(FIRToken::kw_when)) {
1800  // TODO(completeness): Handle the 'else when' syntactic sugar when we
1801  // care.
1802  return emitError("'else when' syntax not supported yet"), failure();
1803  }
1804 
1805  // Parse the 'else' body into the 'else' region.
1806  if (parseToken(FIRToken::colon, "expected ':' after 'else'") ||
1807  parseOptionalInfo(elseInfo) || parseSuite(whenStmt.getElseBodyBuilder()))
1808  return failure();
1809 
1810  // TODO(firrtl spec): There is no reason for the 'else :' grammar to take an
1811  // info. It doesn't appear to be generated either.
1812  return success();
1813 }

1815 /// leading-exp-stmt ::= exp '<=' exp info?
1816 /// ::= exp '<-' exp info?
1817 /// ::= exp 'is' 'invalid' info?
1818 ParseResult FIRStmtParser::parseLeadingExpStmt(Value lhs, SubOpVector &subOps) {
1819  // Figure out which kind of statement it is.
1820  LocWithInfo info(getToken().getLoc(), this);
1821 
1822  // If 'is' grammar is special.
1823  if (consumeIf(FIRToken::kw_is)) {
1824  if (parseToken(FIRToken::kw_invalid, "expected 'invalid'") ||
1825  parseOptionalInfo(info, subOps))
1826  return failure();
1827 
1828  // The FIRRTL specification describes Invalidates as a statement with
1829  // implicit connect semantics. The FIRRTL dialect models it as a primitive
1830  // that returns an "Invalid Value", followed by an explicit connect to make
1831  // the representation simpler and more consistent.
1832  auto invalidType = lhs.getType().cast<FIRRTLType>();
1833  if (invalidType.isa<AnalogType>()) {
1834  auto val = builder.create<InvalidValuePrimOp>(info.getLoc(), invalidType);
1835  builder.create<AttachOp>(info.getLoc(), ValueRange{lhs, val});
1836  } else {
1837  auto val = builder.create<InvalidValuePrimOp>(
1838  info.getLoc(), invalidType.getPassiveType());
1839  builder.create<ConnectOp>(info.getLoc(), lhs, val);
1840  }
1841  return success();
1842  }
1843 
1844  auto kind = getToken().getKind();
1845  if (getToken().isNot(FIRToken::less_equal, FIRToken::less_minus))
1846  return emitError("expected '<=', '<-', or 'is' in statement"), failure();
1847  consumeToken();
1848 
1849  Value rhs;
1850  if (parseExp(rhs, subOps, "unexpected token in statement") ||
1851  parseOptionalInfo(info, subOps))
1852  return failure();
1853 
1854  if (kind == FIRToken::less_equal) {
1855  // Some operations, dshl for example, have implicit truncations, even in lo
1856  // firrtl. Chisel will also use connects as partial connects to do
1857  // truncation. Handle truncations as partial connects, which allow
1858  // truncation.
1859  auto lhsPType = lhs.getType().cast<FIRRTLType>().getPassiveType();
1860  auto rhsPType = rhs.getType().cast<FIRRTLType>().getPassiveType();
1861  if (lhsPType != rhsPType && lhsPType.getBitWidthOrSentinel() >= 0 &&
1862  lhsPType.getBitWidthOrSentinel() < rhsPType.getBitWidthOrSentinel()) {
1863  builder.create<PartialConnectOp>(info.getLoc(), lhs, rhs);
1864  } else {
1865  builder.create<ConnectOp>(info.getLoc(), lhs, rhs);
1866  }
1867  } else {
1868  assert(kind == FIRToken::less_minus && "unexpected kind");
1869  builder.create<PartialConnectOp>(info.getLoc(), lhs, rhs);
1870  }
1871  return success();
1872 }

1877 /// instance ::= 'inst' id 'of' id info?
1878 ParseResult FIRStmtParser::parseInstance() {
1879  LocWithInfo info(getToken().getLoc(), this);
1880  consumeToken(FIRToken::kw_inst);
1881 
1882  // If this was actually the start of a connect or something else handle
1883  // that.
1884  if (auto isExpr = parseExpWithLeadingKeyword("inst", info))
1885  return isExpr.getValue();
1886 
1887  StringAttr id;
1888  StringRef moduleName;
1889  if (parseId(id, "expected instance name") ||
1890  parseToken(FIRToken::kw_of, "expected 'of' in instance") ||
1891  parseId(moduleName, "expected module name") || parseOptionalInfo(info))
1892  return failure();
1893 
1894  // Look up the module that is being referenced.
1895  auto circuit =
1896  builder.getBlock()->getParentOp()->getParentOfType<CircuitOp>();
1897  auto referencedModule = circuit.lookupSymbol(moduleName);
1898  if (!referencedModule) {
1899  emitError(info.getFIRLoc(),
1900  "use of undefined module name '" + moduleName + "' in instance");
1901  return failure();
1902  }
1903 
1904  SmallVector<ModulePortInfo, 4> modulePorts;
1905  getModulePortInfo(referencedModule, modulePorts);
1906 
1907  // Make a bundle of the inputs and outputs of the specified module.
1908  SmallVector<Type, 4> resultTypes;
1909  SmallVector<Attribute, 4> resultNames;
1910  resultTypes.reserve(modulePorts.size());
1911  resultNames.reserve(modulePorts.size());
1912 
1913  for (auto port : modulePorts) {
1914  resultTypes.push_back(FlipType::get(port.type));
1915  resultNames.push_back(port.name);
1916  }
1917  auto result = builder.create<InstanceOp>(
1918  info.getLoc(), resultTypes, builder.getSymbolRefAttr(moduleName),
1919  builder.getArrayAttr(resultNames), filterUselessName(id));
1920 
1921  // Since we are implicitly unbundling the instance results, we need to keep
1922  // track of the mapping from bundle fields to results in the unbundledValues
1923  // data structure. Build our entry now.
1924  UnbundledValueEntry unbundledValueEntry;
1925  unbundledValueEntry.reserve(modulePorts.size());
1926  for (size_t i = 0, e = modulePorts.size(); i != e; ++i)
1927  unbundledValueEntry.push_back({resultNames[i], result.getResult(i)});
1928 
1929  // Add it to unbundledValues and add an entry to the symbol table to remember
1930  // it.
1931  unbundledValues.push_back(std::move(unbundledValueEntry));
1932  auto entryId = UnbundledID(unbundledValues.size());
1933  return addSymbolEntry(id.getValue(), entryId, info.getFIRLoc());
1934 }

2132  builder.getArrayAttr(resultNames), filterUselessName(id));
