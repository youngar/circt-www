27 /// This represents a specific token for .fir files.
28 class FIRToken {
29 public:
30  enum Kind {
31 #define TOK_MARKER(NAME) NAME,
32 #define TOK_IDENTIFIER(NAME) NAME,
33 #define TOK_LITERAL(NAME) NAME,
34 #define TOK_PUNCTUATION(NAME, SPELLING) NAME,
35 #define TOK_KEYWORD(SPELLING) kw_##SPELLING,
36 #define TOK_LPKEYWORD(SPELLING) lp_##SPELLING,
37 #include "FIRTokenKinds.def"
38  };
39 
40  FIRToken(Kind kind, StringRef spelling) : kind(kind), spelling(spelling) {}
41 
42  // Return the bytes that make up this token.
43  StringRef getSpelling() const { return spelling; }
44 
45  // Token classification.
46  Kind getKind() const { return kind; }
47  bool is(Kind K) const { return kind == K; }
48 
49  bool isAny(Kind k1, Kind k2) const { return is(k1) || is(k2); }
50 
51  /// Return true if this token is one of the specified kinds.
52  template <typename... T>
53  bool isAny(Kind k1, Kind k2, Kind k3, T... others) const {
54  if (is(k1))
55  return true;
56  return isAny(k2, k3, others...);
57  }
58 
59  bool isNot(Kind k) const { return kind != k; }
60 
61  /// Return true if this token isn't one of the specified kinds.
62  template <typename... T>
63  bool isNot(Kind k1, Kind k2, T... others) const {
64  return !isAny(k1, k2, others...);
65  }
66 
67  /// Return true if this is one of the keyword token kinds (e.g. kw_wire).
68  bool isKeyword() const;
69 
70  /// Given a token containing a string literal, return its value, including
71  /// removing the quote characters and unescaping the contents of the string.
72  /// The lexer has already verified that this token is valid.
73  std::string getStringValue() const;
74  static std::string getStringValue(StringRef spelling);
75 
76  // Location processing.
77  llvm::SMLoc getLoc() const;
78  llvm::SMLoc getEndLoc() const;
79  llvm::SMRange getLocRange() const;
80 
81 private:
82  /// Discriminator that indicates the sort of token this is.
83  Kind kind;
84 
85  /// A reference to the entire token contents; this is always a pointer into
86  /// a memory buffer owned by the source manager.
87  StringRef spelling;
88 };
