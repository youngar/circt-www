24 // If the specified attribute set contains the firrtl.name attribute, return it.
25 StringAttr firrtl::getFIRRTLNameAttr(ArrayRef<NamedAttribute> attrs) {
26  for (auto &argAttr : attrs) {
27  // FIXME: We currently use firrtl.name instead of name because this makes
28  // the FunctionLike handling in MLIR core happier. It otherwise doesn't
29  // allow attributes on module parameters.
30  if (argAttr.first != "firrtl.name")
31  continue;
32 
33  return argAttr.second.dyn_cast<StringAttr>();
34  }
35 
36  return StringAttr();
37 }

47  /// Get a special name to use when printing the given operation. See
48  /// OpAsmInterface.td#getAsmResultNames for usage details and documentation.
49  void getAsmResultNames(Operation *op,
50  OpAsmSetValueNameFn setNameFn) const override {
51  if (auto instance = dyn_cast<InstanceOp>(op)) {
52  StringRef base;
53  if (auto nameAttr = op->getAttrOfType<StringAttr>("name"))
54  base = nameAttr.getValue();
55  if (base.empty())
56  base = "inst";
57 
58  for (size_t i = 0, e = op->getNumResults(); i != e; ++i) {
59  setNameFn(instance.getResult(i),
60  (base + "_" + instance.getPortNameStr(i)).str());
61  }
62  return;
63  }
64 
65  if (auto memory = dyn_cast<MemOp>(op)) {
66  StringRef base;
67  if (auto nameAttr = op->getAttrOfType<StringAttr>("name"))
68  base = nameAttr.getValue();
69  if (base.empty())
70  base = "mem";
71 
72  for (size_t i = 0, e = op->getNumResults(); i != e; ++i) {
73  setNameFn(memory.getResult(i),
74  (base + "_" + memory.getPortNameStr(i)).str());
75  }
76  return;
77  }
78 
79  // Many firrtl dialect operations have an optional 'name' attribute. If
80  // present, use it.
81  if (op->getNumResults() == 1)
82  if (auto nameAttr = op->getAttrOfType<StringAttr>("name"))
83  setNameFn(op->getResult(0), nameAttr.getValue());
84 
85  // For constants in particular, propagate the value into the result name to
86  // make it easier to read the IR.
87  if (auto constant = dyn_cast<ConstantOp>(op)) {
88  auto intTy = constant.getType().dyn_cast<IntType>();
89 
90  // Otherwise, build a complex name with the value and type.
91  SmallString<32> specialNameBuffer;
92  llvm::raw_svector_ostream specialName(specialNameBuffer);
93  specialName << 'c';
94  if (intTy) {
95  constant.value().print(specialName, /*isSigned:*/ intTy.isSigned());
96 
97  specialName << (intTy.isSigned() ? "_si" : "_ui");
98  auto width = intTy.getWidthOrSentinel();
99  if (width != -1)
100  specialName << width;
101  } else {
102  constant.value().print(specialName, /*isSigned:*/ false);
103  }
104  setNameFn(constant.getResult(), specialName.str());
105  }
106  }

108  /// Get a special name to use when printing the entry block arguments of the
109  /// region contained by an operation in this dialect.
110  void getAsmBlockArgumentNames(Block *block,
111  OpAsmSetValueNameFn setNameFn) const override {
112  // Check to see if the operation containing the arguments has 'firrtl.name'
113  // attributes for them. If so, use that as the name.
114  auto *parentOp = block->getParentOp();
115 
116  for (size_t i = 0, e = block->getNumArguments(); i != e; ++i) {
117  // Scan for a 'firrtl.name' attribute.
118  if (auto str = getFIRRTLNameAttr(impl::getArgAttrs(parentOp, i)))
119  setNameFn(block->getArgument(i), str.getValue());
120  }
121  }
122 };
123 } // end anonymous namespace

150 /// Registered hook to materialize a single constant operation from a given
151 /// attribute value with the desired resultant type. This method should use
152 /// the provided builder to create the operation without changing the
153 /// insertion position. The generated operation is expected to be constant
154 /// like, i.e. single result, zero operands, non side-effecting, etc. On
155 /// success, this hook should return the value generated to represent the
156 /// constant value. Otherwise, it should return null on failure.
157 Operation *FIRRTLDialect::materializeConstant(OpBuilder &builder,
158  Attribute value, Type type,
159  Location loc) {
160  // Integer constants.
161  if (auto intType = type.dyn_cast<IntType>())
162  if (auto attrValue = value.dyn_cast<IntegerAttr>())
163  return builder.create<ConstantOp>(loc, type, attrValue);
164 
165  return nullptr;
166 }
