27 /// Given a FIRRTL type, return the corresponding type for the RTL dialect.
28 /// This returns a null type if it cannot be lowered.
29 static Type lowerType(Type type) {
30  auto firType = type.dyn_cast<FIRRTLType>();
31  if (!firType)
32  return {};
33 
34  // Ignore flip types.
35  firType = firType.getPassiveType();
36 
37  if (BundleType bundle = firType.dyn_cast<BundleType>()) {
38  mlir::SmallVector<rtl::StructType::FieldInfo, 8> rtlfields;
39  for (auto element : bundle.getElements()) {
40  Type etype = lowerType(element.type);
41  if (!etype)
42  return {};
43  rtlfields.push_back(rtl::StructType::FieldInfo{element.name, etype});
44  }
45  return rtl::StructType::get(type.getContext(), rtlfields);
46  }
47 
48  auto width = firType.getBitWidthOrSentinel();
49  if (width >= 0) // IntType, analog with known width, clock, etc.
50  return IntegerType::get(type.getContext(), width);
51 
52  return {};
53 }

61 /// Cast from a standard type to a FIRRTL type, potentially with a flip.
62 static Value castToFIRRTLType(Value val, Type type,
63  ImplicitLocOpBuilder &builder) {
64  auto firType = type.cast<FIRRTLType>();
65 
66  // If this was an Analog type, it will be converted to an InOut type.
67  if (type.isa<AnalogType>())
68  return builder.createOrFold<AnalogInOutCastOp>(firType, val);
69 
70  if (BundleType bundle = type.dyn_cast<BundleType>()) {
71  val = builder.createOrFold<RTLStructCastOp>(firType.getPassiveType(), val);
72  } else {
73  val = builder.createOrFold<StdIntCastOp>(firType.getPassiveType(), val);
74  }
75 
76  // Handle the flip type if needed.
77  if (type != val.getType())
78  val = builder.createOrFold<AsNonPassivePrimOp>(firType, val);
79  return val;
80 }

82 /// Cast from a FIRRTL type (potentially with a flip) to a standard type.
83 static Value castFromFIRRTLType(Value val, Type type,
84  ImplicitLocOpBuilder &builder) {
85  // Strip off Flip type if needed.
86  val = builder.createOrFold<AsPassivePrimOp>(val);
87  if (rtl::StructType structTy = type.dyn_cast<rtl::StructType>())
88  return builder.createOrFold<RTLStructCastOp>(type, val);

130 /// Run on the firrtl.circuit operation, lowering any firrtl.module operations
131 /// it contains.
132 void FIRRTLModuleLowering::runOnOperation() {
133  // We run on the top level modules in the IR blob. Start by finding the
134  // firrtl.circuit within it. If there is none, then there is nothing to do.
135  auto *moduleBody = getOperation().getBody();
136 
137  // Find the single firrtl.circuit in the module.
138  CircuitOp circuit;
139  for (auto &op : *moduleBody) {
140  if ((circuit = dyn_cast<CircuitOp>(&op)))
141  break;
142  }
143 
144  if (!circuit)
145  return;
146 
147  // Emit all the macros and preprocessor gunk at the start of the file.
148  lowerFileHeader(circuit);
149 
150  auto *circuitBody = circuit.getBody();
151 
152  // Keep track of the mapping from old to new modules. The result may be null
153  // if lowering failed.
154  DenseMap<Operation *, Operation *> oldToNewModuleMap;
155 
156  // Iterate through each operation in the circuit body, transforming any
157  // FModule's we come across.
158  for (auto &op : circuitBody->getOperations()) {
159  if (auto module = dyn_cast<FModuleOp>(op)) {
160  oldToNewModuleMap[&op] = lowerModule(module, moduleBody);
161  continue;
162  }
163 
164  if (auto extModule = dyn_cast<FExtModuleOp>(op)) {
165  oldToNewModuleMap[&op] = lowerExtModule(extModule, moduleBody);
166  continue;
167  }
168 
169  if (isa<DoneOp>(op))
170  continue;
171 
172  // Otherwise we don't know what this is. We are just going to drop it,
173  // but emit an error so the client has some chance to know that this is
174  // going to happen.
175  op.emitError("unexpected operation '")
176  << op.getName() << "' in a firrtl.circuit";
177  }
178 
179  // Now that we've lowered all of the modules, move the bodies over and update
180  // any instances that refer to the old modules. Only rtl.instance can refer
181  // to an rtl.module, not a firrtl.instance.
182  //
183  // TODO: This is a trivially parallelizable for loop. We should be able to
184  // process each module in parallel.
185  for (auto &op : circuitBody->getOperations()) {
186  if (auto module = dyn_cast<FModuleOp>(op))
187  lowerModuleBody(module, oldToNewModuleMap);
188  }
189 
190  // Finally delete all the old modules.
191  for (auto oldNew : oldToNewModuleMap)
192  oldNew.first->erase();
193 
194  // Now that the modules are moved over, remove the Circuit. We pop the 'main
195  // module' specified in the Circuit into an attribute on the top level module.
196  getOperation()->setAttr(
197  "firrtl.mainModule",
198  StringAttr::get(circuit.name(), circuit.getContext()));
199  circuit.erase();
200 }

271 LogicalResult
272 FIRRTLModuleLowering::lowerPorts(ArrayRef<ModulePortInfo> firrtlPorts,
273  SmallVectorImpl<rtl::ModulePortInfo> &ports,
274  Operation *moduleOp) {
275 
276  ports.reserve(firrtlPorts.size());
277  size_t numArgs = 0;
278  size_t numResults = 0;
279  for (auto firrtlPort : firrtlPorts) {
280  rtl::ModulePortInfo rtlPort;
281  rtlPort.name = firrtlPort.name;
282  rtlPort.type = lowerType(firrtlPort.type);
283 
284  // We can't lower all types, so make sure to cleanly reject them.
285  if (!rtlPort.type) {
286  moduleOp->emitError("cannot lower this port type to RTL");
287  return failure();
288  }
289 
290  // If this is a zero bit port, just drop it. It doesn't matter if it is
291  // input, output, or inout. We don't want these at the RTL level.
292  if (rtlPort.type.isInteger(0))
293  continue;
294 
295  // Figure out the direction of the port.
296  if (firrtlPort.isOutput()) {
297  rtlPort.direction = rtl::PortDirection::OUTPUT;
298  rtlPort.argNum = numResults++;
299  } else if (firrtlPort.isInput()) {
300  rtlPort.direction = rtl::PortDirection::INPUT;
301  rtlPort.argNum = numArgs++;
302  } else {
303  // If the port is an inout bundle or contains an analog type, then it is
304  // implicitly inout.
305  rtlPort.type = rtl::InOutType::get(rtlPort.type);
306  rtlPort.direction = rtl::PortDirection::INOUT;
307  rtlPort.argNum = numArgs++;
308  }
309  ports.push_back(rtlPort);
310  }
311  return success();
312 }

352 /// If the value dominates the marker, just return. Otherwise add it to ops and
353 /// recursively process its operands.
354 ///
355 /// Return true if we can't move an operation, false otherwise.
356 static bool
357 collectOperationTreeBelowMarker(Value value, Operation *marker,
358  SmallVector<Operation *, 8> &ops,
359  SmallPtrSet<Operation *, 8> &visited) {
360  // If the value is a BB argument or if the op is in a containing block, then
361  // it must dominate the marker.
362  auto *op = value.getDefiningOp();
363  if (!op || op->getBlock() != marker->getBlock())
364  return false;
365 
366  // We can't move the marker itself.
367  if (op == marker)
368  return true;
369 
370  // Otherwise if it is an op in the same block as the marker, see if it is
371  // already at or above the marker.
372  if (op->isBeforeInBlock(marker))
373  return false;
374 
375  // Pull the computation tree into the set. If it was already added, then
376  // don't reprocess it.
377  if (!visited.insert(op).second)
378  return false;
379 
380  // Otherwise recursively process the operands.
381  for (auto operand : op->getOperands())
382  if (collectOperationTreeBelowMarker(operand, marker, ops, visited))
383  return true;
384 
385  // Add ops in post order so we make sure they get moved in a coherent order.
386  ops.push_back(op);
387  return false;
388 }

390 /// Given a value of flip type, check to see if all of the uses of it are
391 /// connects. If so, remove the connects and return the value being connected
392 /// to it, converted to an RTL type. If this isn't a situation we can handle,
393 /// just return null.
394 ///
395 /// This can happen when there are no connects to the value, or if
396 /// firrtl.invalid is used. The 'mergePoint' location is where a 'rtl.merge'
397 /// operation should be inserted if needed.
398 static Value tryEliminatingConnectsToValue(Value flipValue,
399  Operation *insertPoint) {
400  SmallVector<ConnectOp, 2> connects;
401  for (auto *use : flipValue.getUsers()) {
402  // We only know about 'connect' uses, where this is the destination.
403  auto connect = dyn_cast<ConnectOp>(use);
404  if (!connect || connect.src() == flipValue)
405  return {};
406 
407  connects.push_back(connect);
408  }
409 
410  // We don't have an RTL equivalent of "poison" so just don't special case the
411  // case where there are no connects other uses of an output.
412  if (connects.empty())
413  return {};
414 
415  // Don't special case zero-bit results.
416  auto loweredType = lowerType(flipValue.getType());
417  if (loweredType.isInteger(0))
418  return {};
419 
420  // We need to see if we can move all of the computation that feeds the
421  // connects to be "above" the insertion point to avoid introducing cycles that
422  // will break LowerToRTL. Consider optimizing away a wire for inputs on an
423  // instance like this:
424  //
425  // %input1, %input2, %output = firrtl.instance (...)
426  // %value1 = computation1()
427  // firrtl.connect %input1, %value1
428  //
429  // %value2 = computation2(%output)
430  // firrtl.connect %input2, %value2
431  //
432  // We can elide the wire for %input1, but have to move the computation1 ops
433  // above the firrtl.instance. However, there are cases like the second one
434  // where we *cannot* move the computation. In these sorts of cases, we just
435  // fall back to inserting a wire conservatively, which breaks the cycle.
436  //
437  // We don't have to do this check for insertion points that are at the
438  // terminator in the module, because we know that everything is above it by
439  // definition.
440  if (!insertPoint->isKnownTerminator()) {
441  // On success, these are the ops that we need to move up above the insertion
442  // point. We keep track of a visited set because each compute subgraph is
443  // a dag (not a tree), and we want to only want to visit each subnode once.
444  SmallVector<Operation *, 8> opsToMove;
445  SmallPtrSet<Operation *, 8> visited;
446 
447  // Collect the computation tree feeding the source operations. We build the
448  // list of ops to move in post-order to ensure that we provide a valid DAG
449  // ordering of the result.
450  for (auto connect : connects) {
451  if (collectOperationTreeBelowMarker(connect.src(), insertPoint, opsToMove,
452  visited))
453  return {};
454  }
455 
456  // Since it looks like all the operations can be moved, actually do it.
457  for (auto *op : opsToMove)
458  op->moveBefore(insertPoint);
459  }
460 
461  // Convert each connect into an extended version of its operand being output.
462  SmallVector<Value, 2> results;
463  ImplicitLocOpBuilder builder(insertPoint->getLoc(), insertPoint);
464 
465  for (auto connect : connects) {
466  auto connectSrc = connect.src();
467 
468  // Convert fliped sources to passive sources.
469  if (!connectSrc.getType().cast<FIRRTLType>().isPassive())
470  connectSrc = builder.createOrFold<AsPassivePrimOp>(connectSrc);
471 
472  // We know it must be the destination operand due to the types, but the
473  // source may not match the destination width.
474  auto destTy = flipValue.getType().cast<FIRRTLType>().getPassiveType();
475  if (destTy != connectSrc.getType()) {
476  // The only type mismatch we can have is due to integer width differences.
477  auto destWidth = destTy.getBitWidthOrSentinel();
478  assert(destWidth != -1 && "must know integer widths");
479  connectSrc =
480  builder.createOrFold<PadPrimOp>(destTy, connectSrc, destWidth);
481  }
482 
483  // Remove the connect and use its source as the value for the output.
484  connect.erase();
485  results.push_back(connectSrc);
486  }
487 
488  // Convert from FIRRTL type to builtin type to do the merge.
489  for (auto &result : results)
490  result = castFromFIRRTLType(result, loweredType, builder);
491 
492  // Folding merge of one value just returns the value.
493  return builder.createOrFold<rtl::MergeOp>(results);
494 }

496 /// Now that we have the operations for the rtl.module's corresponding to the
497 /// firrtl.module's, we can go through and move the bodies over, updating the
498 /// ports and instances.
499 void FIRRTLModuleLowering::lowerModuleBody(
500  FModuleOp oldModule,
501  DenseMap<Operation *, Operation *> &oldToNewModuleMap) {
502  auto newModule =
503  dyn_cast_or_null<rtl::RTLModuleOp>(oldToNewModuleMap[oldModule]);
504  // Don't touch modules if we failed to lower ports.
505  if (!newModule)
506  return;
507 
508  ImplicitLocOpBuilder bodyBuilder(oldModule.getLoc(), newModule.body());
509 
510  // Use a placeholder instruction be a cursor that indicates where we want to
511  // move the new function body to. This is important because we insert some
512  // ops at the start of the function and some at the end, and the body is
513  // currently empty to avoid iterator invalidation.
514  auto cursor = bodyBuilder.create<rtl::ConstantOp>(APInt(1, 1));
515  bodyBuilder.setInsertionPoint(cursor);
516 
517  // Insert argument casts, and re-vector users in the old body to use them.
518  SmallVector<ModulePortInfo, 8> ports;
519  oldModule.getPortInfo(ports);
520  assert(oldModule.body().getNumArguments() == ports.size() &&
521  "port count mismatch");
522 
523  size_t nextNewArg = 0;
524  size_t firrtlArg = 0;
525  SmallVector<Value, 4> outputs;
526 
527  // This is the terminator in the new module.
528  auto outputOp = newModule.getBodyBlock()->getTerminator();
529  ImplicitLocOpBuilder outputBuilder(oldModule.getLoc(), outputOp);
530 
531  for (auto &port : ports) {
532  // Inputs and outputs are both modeled as arguments in the FIRRTL level.
533  auto oldArg = oldModule.body().getArgument(firrtlArg++);
534 
535  bool isZeroWidth =
536  port.type.cast<FIRRTLType>().getBitWidthOrSentinel() == 0;
537 
538  if (!port.isOutput() && !isZeroWidth) {
539  // Inputs and InOuts are modeled as arguments in the result, so we can
540  // just map them over. We model zero bit outputs as inouts.
541  Value newArg = newModule.body().getArgument(nextNewArg++);
542 
543  // Cast the argument to the old type, reintroducing sign information in
544  // the rtl.module body.
545  newArg = castToFIRRTLType(newArg, oldArg.getType(), bodyBuilder);
546  // Switch all uses of the old operands to the new ones.
547  oldArg.replaceAllUsesWith(newArg);
548  continue;
549  }
550 
551  // We lower zero width inout and outputs to a wire that isn't connected to
552  // anything outside the module. Inputs are lowered to zero.
553  if (isZeroWidth && port.isInput()) {
554  Value newArg = bodyBuilder.create<WireOp>(FlipType::get(port.type),
555  "." + port.getName().str() +
556  ".0width_input");
557 
558  newArg = bodyBuilder.create<AsPassivePrimOp>(newArg);
559  oldArg.replaceAllUsesWith(newArg);
560  continue;
561  }
562 
563  if (auto value = tryEliminatingConnectsToValue(oldArg, outputOp)) {
564  // If we were able to find the value being connected to the output,
565  // directly use it!
566  outputs.push_back(value);
567  assert(oldArg.use_empty() && "should have removed all uses of oldArg");
568  continue;
569  }
570 
571  // Outputs need a temporary wire so they can be connect'd to, which we
572  // then return.
573  Value newArg = bodyBuilder.create<WireOp>(
574  port.type, "." + port.getName().str() + ".output");
575  // Switch all uses of the old operands to the new ones.
576  oldArg.replaceAllUsesWith(newArg);
577 
578  // Don't output zero bit results or inouts.
579  auto resultRTLType = lowerType(port.type);
580  if (!resultRTLType.isInteger(0)) {
581  auto output = castFromFIRRTLType(newArg, resultRTLType, outputBuilder);
582  outputs.push_back(output);
583  }
584  }
585 
586  // Update the rtl.output terminator with the list of outputs we have.
587  outputOp->setOperands(outputs);
588 
589  // Finally splice the body over, don't move the old terminator over though.
590  auto &oldBlockInstList = oldModule.getBodyBlock()->getOperations();
591  auto &newBlockInstList = newModule.getBodyBlock()->getOperations();
592  newBlockInstList.splice(Block::iterator(cursor), oldBlockInstList,
593  oldBlockInstList.begin(),
594  std::prev(oldBlockInstList.end()));
595 
596  // Now that we're all over into the new module, update all the
597  // firrtl.instance's to be rtl.instance's. Lowering an instance will also
598  // delete a bunch of firrtl.subfield and firrtl.connect operations, so we have
599  // to be careful about iterator invalidation.
600  for (auto opIt = newBlockInstList.begin(), opEnd = newBlockInstList.end();
601  opIt != opEnd;) {
602  auto instance = dyn_cast<InstanceOp>(&*opIt);
603  if (!instance) {
604  ++opIt;
605  continue;
606  }
607 
608  // Remember a position above the current op. New things will get put before
609  // the current op (including other instances!) and we want to make sure to
610  // revisit them.
611  cursor->moveBefore(instance);
612 
613  // We found an instance - lower it. On successful return there will be
614  // zero uses and we can remove the operation.
615  lowerInstance(instance, oldModule->getParentOfType<CircuitOp>(),
616  oldToNewModuleMap);
617  opIt = Block::iterator(cursor);
618  }
619 
620  // We are done with our cursor op.
621  cursor.erase();
622 }

624 /// Lower a firrtl.instance operation to an rtl.instance operation. This is a
625 /// bit more involved than it sounds because we have to clean up the subfield
626 /// operations that are hanging off of it, handle the differences between FIRRTL
627 /// and RTL approaches to module parameterization and output ports.
628 ///
629 /// On success, this returns with the firrtl.instance op having no users,
630 /// letting the caller erase it.
631 void FIRRTLModuleLowering::lowerInstance(
632  InstanceOp oldInstance, CircuitOp circuitOp,
633  DenseMap<Operation *, Operation *> &oldToNewModuleMap) {
634 
635  auto *oldModule = circuitOp.lookupSymbol(oldInstance.moduleName());
636  auto newModule = oldToNewModuleMap[oldModule];
637  if (!newModule) {
638  oldInstance->emitOpError("could not find module referenced by instance");
639  return;
640  }
641 
642  // If this is a referenced to a parameterized extmodule, then bring the
643  // parameters over to this instance.
644  DictionaryAttr parameters;
645  if (auto oldExtModule = dyn_cast<FExtModuleOp>(oldModule))
646  if (auto paramsOptional = oldExtModule.parameters())
647  parameters = paramsOptional.getValue();
648 
649  // Decode information about the input and output ports on the referenced
650  // module.
651  SmallVector<ModulePortInfo, 8> portInfo;
652  getModulePortInfo(oldModule, portInfo);
653 
654  // Build an index from the name attribute to an index into portInfo, so we can
655  // do efficient lookups.
656  llvm::SmallDenseMap<Attribute, unsigned> portIndicesByName;
657  for (unsigned portIdx = 0, e = portInfo.size(); portIdx != e; ++portIdx)
658  portIndicesByName[portInfo[portIdx].name] = portIdx;
659 
660  // Ok, get ready to create the new instance operation. We need to prepare
661  // input operands and results.
662  ImplicitLocOpBuilder builder(oldInstance.getLoc(), oldInstance);
663  SmallVector<Type, 8> resultTypes;
664  SmallVector<Value, 8> operands;
665  for (size_t portIndex = 0, e = portInfo.size(); portIndex != e; ++portIndex) {
666  auto &port = portInfo[portIndex];
667  auto portType = lowerType(port.type);
668  if (!portType) {
669  oldInstance->emitOpError("could not lower type of port ") << port.name;
670  return;
671  }
672 
673  if (port.isOutput()) {
674  // Drop zero bit results.
675  if (!portType.isInteger(0))
676  resultTypes.push_back(portType);
677  continue;
678  }
679 
680  // If we can find the connects to this port, then we can directly
681  // materialize it.
682  auto portResult = oldInstance.getPortNamed(port.name);
683  assert(portResult && "invalid IR, couldn't find port");
684  if (auto value = tryEliminatingConnectsToValue(portResult, oldInstance)) {
685  // If we got a value connecting to the input port, then we can pass it
686  // into the RTL instance without a temporary wire.
687  operands.push_back(value);
688  continue;
689  }
690 
691  // Otherwise, create a wire for each input/inout operand, so there is
692  // something to connect to.
693  auto name = builder.getStringAttr("." + port.getName().str() + ".wire");
694  auto wire = builder.create<WireOp>(port.type, name);
695 
696  // Drop zero bit input/inout ports.
697  if (!portType.isInteger(0))
698  operands.push_back(castFromFIRRTLType(wire, portType, builder));
699 
700  portResult.replaceAllUsesWith(wire);
701  }
702 
703  // Use the symbol from the module we are referencing.
704  FlatSymbolRefAttr symbolAttr = builder.getSymbolRefAttr(newModule);
705 
706  // Create the new rtl.instance operation.
707  StringAttr instanceName;
708  if (oldInstance.name().hasValue())
709  instanceName = oldInstance.nameAttr();
710 
711  auto newInst = builder.create<rtl::InstanceOp>(
712  resultTypes, instanceName, symbolAttr, operands, parameters);
713 
714  // Now that we have the new rtl.instance, we need to remap all of the users
715  // of the outputs/results to the values returned by the instance.
716  unsigned resultNo = 0;
717  for (size_t portIndex = 0, e = portInfo.size(); portIndex != e; ++portIndex) {
718  auto &port = portInfo[portIndex];
719  if (!port.isOutput())
720  continue;
721 
722  auto resultType = FlipType::get(port.type);
723  Value resultVal;
724  if (port.type.getPassiveType().getBitWidthOrSentinel() != 0) {
725  // Cast the value to the right signedness and flippedness.
726  resultVal = newInst.getResult(resultNo++);
727  resultVal = castToFIRRTLType(resultVal, resultType, builder);
728  } else {
729  // Zero bit results are just replaced with a wire.
730  resultVal = builder.create<WireOp>(
731  resultType, "." + port.getName().str() + ".0width_result");
732  }
733 
734  // Replace uses of the old output port with the returned value directly.
735  auto portResult = oldInstance.getPortNamed(port.name);
736  assert(portResult && "invalid IR, couldn't find port");
737  portResult.replaceAllUsesWith(resultVal);
738  }
739 
740  // Done with the oldInstance!
741  oldInstance.erase();
742 }

907 // This is the main entrypoint for the lowering pass.
908 void FIRRTLLowering::runOnOperation() {
909  // FIRRTL FModule is a single block because FIRRTL ops are a DAG. Walk
910  // through each operation, lowering each in turn if we can, introducing casts
911  // if we cannot.
912  auto *body = getOperation().getBodyBlock();
913 
914  randomizePrologEmitted = false;
915 
916  SmallVector<Operation *, 16> opsToRemove;
917 
918  // Iterate through each operation in the module body, attempting to lower
919  // each of them. We maintain 'builder' for each invocation.
920  ImplicitLocOpBuilder theBuilder(getOperation().getLoc(), &getContext());
921  builder = &theBuilder;
922  for (auto &op : body->getOperations()) {
923  builder->setInsertionPoint(&op);
924  builder->setLoc(op.getLoc());
925  if (succeeded(dispatchVisitor(&op))) {
926  opsToRemove.push_back(&op);
927  } else {
928  switch (handleUnloweredOp(&op)) {
929  case AlreadyLowered:
930  break; // Something like rtl.output, which is already lowered.
931  case NowLowered: // Something handleUnloweredOp removed.
932  opsToRemove.push_back(&op);
933  break;
934  case LoweringFailure:
935  // If lowering failed, don't remove *anything* we've lowered so far,
936  // there may be uses, and the pass will fail anyway.
937  opsToRemove.clear();
938  }
939  }
940  }
941  builder = nullptr;
942 
943  // Now that all of the operations that can be lowered are, remove the original
944  // values. We know that any lowered operations will be dead (if removed in
945  // reverse order) at this point - any users of them from unremapped operations
946  // will be changed to use the newly lowered ops.
947  while (!opsToRemove.empty()) {
948  assert(opsToRemove.back()->use_empty() &&
949  "Should remove ops in reverse order of visitation");
950  opsToRemove.pop_back_val()->erase();
951  }
952 
953  // Clear out the value mapping for next time, so we don't have dangling keys.
954  valueMapping.clear();
955 }

1001 /// Return the lowered value corresponding to the specified original value and
1002 /// then extend it to match the width of destType if needed.
1003 ///
1004 /// This returns a null value for FIRRTL values that cannot be lowered, e.g.
1005 /// unknown width integers.
1006 Value FIRRTLLowering::getLoweredAndExtendedValue(Value value, Type destType) {
1007  assert(value.getType().isa<FIRRTLType>() && destType.isa<FIRRTLType>() &&
1008  "input/output value should be FIRRTL");
1009 
1010  // We only know how to extend integer types with known width.
1011  auto destWidth = destType.cast<FIRRTLType>().getBitWidthOrSentinel();
1012  if (destWidth == -1)
1013  return {};
1014 
1015  auto result = getLoweredValue(value);
1016  if (!result) {
1017  // If this was a zero bit operand being extended, then produce a zero of the
1018  // right result type. If it is just a failure, fail.
1019  if (!isZeroBitFIRRTLType(value.getType()))
1020  return {};
1021  // Zero bit results have to be returned as null. The caller can handle
1022  // this if they want to.
1023  if (destWidth == 0)
1024  return {};
1025  // Otherwise, FIRRTL semantics is that an extension from a zero bit value
1026  // always produces a zero value in the destination width.
1027  return builder->create<rtl::ConstantOp>(APInt(destWidth, 0));
1028  }
1029 
1030  auto srcWidth = result.getType().cast<IntegerType>().getWidth();
1031  if (srcWidth == unsigned(destWidth))
1032  return result;
1033 
1034  if (srcWidth > unsigned(destWidth)) {
1035  builder->emitError("operand should not be a truncation");
1036  return {};
1037  }
1038 
1039  auto resultType = builder->getIntegerType(destWidth);
1040 
1041  // Extension follows the sign of the source value, not the destination.
1042  auto valueFIRType = value.getType().cast<FIRRTLType>().getPassiveType();
1043  if (valueFIRType.cast<IntType>().isSigned())
1044  return builder->createOrFold<rtl::SExtOp>(resultType, result);
1045 
1046  auto zero = builder->create<rtl::ConstantOp>(APInt(destWidth - srcWidth, 0));
1047  return builder->createOrFold<rtl::ConcatOp>(zero, result);
1048 }

1146 /// Handle the case where an operation wasn't lowered. When this happens, the
1147 /// operands should just be unlowered non-FIRRTL values. If the operand was
1148 /// not lowered then leave it alone, otherwise we have a problem with lowering.
1149 ///
1150 FIRRTLLowering::UnloweredOpResult
1151 FIRRTLLowering::handleUnloweredOp(Operation *op) {
1152  // Scan the operand list for the operation to see if none were lowered. In
1153  // that case the operation must be something lowered to RTL already, e.g. the
1154  // rtl.output operation. This is success for us because it is already
1155  // lowered.
1156  if (llvm::all_of(op->getOpOperands(), [&](auto &operand) -> bool {
1157  return !valueMapping.count(operand.get());
1158  })) {
1159  return AlreadyLowered;
1160  }
1161 
1162  // Ok, at least one operand got lowered, so this operation is using a FIRRTL
1163  // value, but wasn't itself lowered. This is because the lowering is
1164  // incomplete. This is either a bug or incomplete implementation.
1165  //
1166  // There is one aspect of incompleteness we intentionally expect: we allow
1167  // primitive operations that produce a zero bit result to be ignored by the
1168  // lowering logic. They don't have side effects, and handling this corner
1169  // case just complicates each of the lowering hooks. Instead, we just handle
1170  // them all right here.
1171  if (op->getNumResults() == 1) {

1338 /// Convert a nested bundle of fields into a flat list of fields. This is used
1339 /// when working with mems to flatten them. Return true if we fail to lower
1340 /// any of the element types, or false if successul.
1341 static bool flattenBundleTypes(Type type, StringRef nameSoFar,
1342  SmallVectorImpl<FlatBundleFieldEntry> &results) {
1343  // Ignore flips.
1344  if (auto flip = type.dyn_cast<FlipType>())
1345  return flattenBundleTypes(flip.getElementType(), nameSoFar, results);
1346 
1347  // In the base case we record this field.
1348  auto bundle = type.dyn_cast<BundleType>();
1349  if (!bundle) {
1350  auto rtlType = lowerType(type);
1351  if (rtlType && !rtlType.isInteger(0))
1352  results.push_back({rtlType, nameSoFar.str()});
1353  return rtlType == Type();
1354  }
1355 
1356  SmallString<16> tmpName(nameSoFar);
1357 
1358  // Otherwise, we have a bundle type. Break it down.
1359  for (auto &elt : bundle.getElements()) {
1360  // Construct the suffix to pass down.
1361  tmpName.resize(nameSoFar.size());
1362  tmpName.push_back('_');
1363  tmpName.append(elt.name.strref());
1364  // Recursively process subelements.
1365  if (flattenBundleTypes(elt.type, tmpName, results))
1366  return true;
1367  }
1368  return false;
1369 }

1371 LogicalResult FIRRTLLowering::visitDecl(MemOp op) {
1372  if (op.readLatency() != 0 || op.writeLatency() != 1) {
1373  // FIXME: This should be an error.
1374  op.emitWarning("FIXME: need to support mem read/write latency correctly");
1375  }
1376 
1377  StringRef memName = "mem";
1378  if (op.name().hasValue())
1379  memName = op.name().getValue();
1380 
1381  // Aggregate mems may declare multiple reg's. We need to declare and random
1382  // initialize them all.
1383  SmallVector<FlatBundleFieldEntry, 8> fieldTypes;
1384  if (flattenBundleTypes(op.getDataType(), memName, fieldTypes))
1385  return op.emitError("could not lower mem element type");
1386 
1387  uint64_t depth = op.depth();
1388 
1389  // Add one reg declaration for each field of the mem.
1390  SmallVector<Value> regs;
1391  for (const auto &field : fieldTypes) {
1392  auto resultType = rtl::UnpackedArrayType::get(field.type, depth);
1393  auto name = builder->getStringAttr(field.name);
1394  regs.push_back(builder->create<sv::RegOp>(resultType, name));
1395  }
1396 
1397  // Emit the initializer expression for simulation that fills it with random
1398  // value.
1399  builder->create<sv::IfDefOp>("!SYNTHESIS", [&]() {
1400  builder->create<sv::InitialOp>([&]() {
1401  emitRandomizePrologIfNeeded();
1402 
1403  builder->create<sv::IfDefOp>("RANDOMIZE_MEM_INIT", [&]() {
1404  if (depth == 1) { // Don't emit a for loop for one element.
1405  for (Value reg : regs) {
1406  auto type = sv::getInOutElementType(reg.getType());
1407  type = sv::getAnyRTLArrayElementType(type);
1408  auto randomVal =
1409  builder->create<sv::TextualValueOp>(type, "`RANDOM");
1410  auto zero = builder->create<rtl::ConstantOp>(APInt(1, 0));
1411  auto subscript = builder->create<sv::ArrayIndexInOutOp>(reg, zero);
1412  builder->create<sv::BPAssignOp>(subscript, randomVal);
1413  }
1414  } else if (!regs.empty()) {
1415  assert(depth < (1ULL << 31) && "FIXME: Our initialization logic uses "
1416  "'integer' which doesn't support "
1417  "mems greater than 2^32");
1418 
1419  std::string action = "integer {{0}}_initvar;\n";
1420  action += "for ({{0}}_initvar = 0; {{0}}_initvar < " +
1421  llvm::utostr(depth) + "; {{0}}_initvar = {{0}}_initvar+1)";
1422  if (regs.size() != 1)
1423  action += " begin";
1424 
1425  for (size_t i = 0, e = regs.size(); i != e; ++i)
1426  action +=
1427  "\n {{" + llvm::utostr(i) + "}}[{{0}}_initvar] = `RANDOM;";
1428 
1429  if (regs.size() != 1)
1430  action += "\nend";
1431  builder->create<sv::VerbatimOp>(action, regs);
1432  }
1433  });
1434  });
1435  });
1436 
1437  // Keep track of whether this mem is an even power of two or not.
1438  bool isPowerOfTwo = llvm::isPowerOf2_64(depth);
1439 
1440  // Lower all of the read/write ports. Each port is a separate
1441  // return value of the memory.
1442  auto namesArray = op.portNames();
1443  for (size_t i = 0, e = namesArray.size(); i != e; ++i) {
1444 
1445  auto portName = namesArray[i].cast<StringAttr>().getValue();
1446  auto port = op.getPortNamed(portName);
1447 
1448  // Do not lower ports if they aren't used.
1449  if (port.use_empty())
1450  continue;
1451 
1452  auto portBundleType =
1453  port.getType().cast<FIRRTLType>().getPassiveType().cast<BundleType>();
1454 
1455  SmallVector<std::pair<Identifier, Value>> portWires;
1456  for (BundleType::BundleElement elt : portBundleType.getElements()) {
1457  auto fieldType = lowerType(elt.type);
1458  if (!fieldType)
1459  return op.emitOpError("port " + elt.name.str() +
1460  " has unexpected field");
1461 
1462  if (fieldType.isInteger(0)) {
1463  portWires.push_back({elt.name, Value()});
1464  continue;
1465  }
1466  auto name =
1467  (Twine(memName) + "_" + portName + "_" + elt.name.str()).str();
1468  auto fieldWire = builder->create<sv::WireOp>(fieldType, name);
1469  portWires.push_back({elt.name, fieldWire});
1470  }
1471 
1472  // Return the inout wire corresponding to a port field.
1473  auto getPortFieldWire = [&](StringRef portName) -> Value {
1474  for (auto entry : portWires) {
1475  if (entry.first.strref() == portName)
1476  return entry.second;
1477  }
1478  llvm_unreachable("unknown port wire!");
1479  };
1480 
1481  // Now that we have the wires for each element, rewrite any subfields to use
1482  // them instead of the subfields.
1483  while (!port.use_empty()) {
1484  auto portField = cast<SubfieldOp>(*port.user_begin());
1485  portField->dropAllReferences();
1486  setLowering(portField, getPortFieldWire(portField.fieldname()));
1487  }
1488 
1489  // Return the value corresponding to a port field.
1490  auto getPortFieldValue = [&](StringRef portName) -> Value {
1491  return builder->create<sv::ReadInOutOp>(getPortFieldWire(portName));
1492  };
1493 
1494  switch (op.getPortKind(portName).getValue()) {
1495  case MemOp::PortKind::ReadWrite:
1496  op.emitOpError("readwrite ports should be lowered into separate read and "
1497  "write ports by previous passes");
1498  continue;
1499  case MemOp::PortKind::Read: {
1500  auto emitReads = [&](bool masked) {
1501  // TODO: not handling bundle elements correctly yet.
1502  if (regs.size() > 1)
1503  op.emitOpError("don't support bundle elements yet");
1504 
1505  // Emit an assign to the read port, using the address.
1506  // TODO(firrtl-spec): It appears that the clock signal on the read port
1507  // is ignored, why does it exist?
1508  for (auto reg : regs) {
1509  auto addr = getPortFieldValue("addr");
1510  Value value = builder->create<sv::ArrayIndexInOutOp>(reg, addr);
1511  value = builder->create<sv::ReadInOutOp>(value);
1512 
1513  // If we're masking, emit "addr < Depth ? mem[addr] : `RANDOM".
1514  if (masked) {
1515  auto addrWidth = addr.getType().getIntOrFloatBitWidth();
1516  auto depthCst =
1517  builder->create<rtl::ConstantOp>(APInt(addrWidth, depth));
1518  auto cmp = builder->create<rtl::ICmpOp>(ICmpPredicate::ult, addr,
1519  depthCst);
1520  auto randomVal =
1521  builder->create<sv::TextualValueOp>(value.getType(), "`RANDOM");
1522  value = builder->create<rtl::MuxOp>(cmp, value, randomVal);
1523  }
1524 
1525  // FIXME: This isn't right for multi-slot data's.
1526  builder->create<sv::ConnectOp>(getPortFieldWire("data"), value);
1527  }
1528  };
1529 
1530  // If the memory size is a power of two, then we can just unconditionally
1531  // read from it, otherwise we emit #ifdef'd masking logic.
1532  if (isPowerOfTwo) {
1533  emitReads(false);
1534  } else {
1535  builder->create<sv::IfDefOp>(
1536  "!RANDOMIZE_GARBAGE_ASSIGN", [&]() { emitReads(false); },
1537  [&]() { emitReads(true); });
1538  }
1539  break;
1540  }
1541 
1542  case MemOp::PortKind::Write: {
1543  // TODO: not handling bundle elements correctly yet.
1544  if (regs.size() > 1)
1545  op.emitOpError("don't support bundle elements yet");
1546 
1547  // Emit something like:
1548  // always @(posedge _M_write_clk) begin
1549  // if (_M_write_en & _M_write_mask)
1550  // _M[_M_write_addr] <= _M_write_data;
1551  // end
1552  auto clock = getPortFieldValue("clk");
1553  builder->create<sv::AlwaysOp>(EventControl::AtPosEdge, clock, [&]() {
1554  auto enable = getPortFieldValue("en");
1555  auto mask = getPortFieldValue("mask");
1556  auto cond = builder->create<rtl::AndOp>(enable, mask);
1557  builder->create<sv::IfOp>(cond, [&]() {
1558  // FIXME: This isn't right for multi-slot data mems.
1559  auto data = getPortFieldValue("data");
1560  auto addr = getPortFieldValue("addr");
1561 
1562  for (auto reg : regs) {
1563  auto slot = builder->create<sv::ArrayIndexInOutOp>(reg, addr);
1564  builder->create<sv::BPAssignOp>(slot, data);
1565  }
1566  });
1567  });
1568 
1569  break;
1570  }
1571  }
1572  }
1573  return success();
1574 }
