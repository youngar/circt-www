30 // These macros are used to implement verifier hooks that check that the result
31 // type of a primitive matches what is returned by the getResultType() static
32 // method. This should go away when/if this bug is implemented:
33 // https://bugs.llvm.org/show_bug.cgi?id=48645
34 #define VERIFY_RESULT_TYPE(...) \
35  { \
36  auto resType = getResultType(__VA_ARGS__, getLoc()); \
37  if (!resType) \
38  return failure(); /*already diagnosed the error*/ \
39  if (resType != getType()) \
40  return emitOpError("result type should be ") << resType; \
41  }

44 // matches. This is useful as the last thing in a verify hook.
45 #define VERIFY_RESULT_TYPE_RET(...) \
46  VERIFY_RESULT_TYPE(__VA_ARGS__); \
47  return success();

96 static LogicalResult verifyCircuitOp(CircuitOp circuit) {
97  StringRef main = circuit.name();
98 
99  // Check that the circuit has a non-empty name.
100  if (main.empty()) {
101  circuit.emitOpError("must have a non-empty name");
102  return failure();
103  }
104 
105  // Check that a module matching the "main" module exists in the circuit.
106  if (!circuit.lookupSymbol(main)) {
107  circuit.emitOpError("must contain one module that matches main name '" +
108  main + "'");
109  return failure();
110  }
111 
112  // Store a mapping of defname to either the first external module
113  // that defines it or, preferentially, the first external module
114  // that defines it and has no parameters.
115  llvm::DenseMap<Attribute, FExtModuleOp> defnameMap;
116 
117  // Verify external modules.
118  for (auto &op : *circuit.getBody()) {
119  auto extModule = dyn_cast<FExtModuleOp>(op);
120  if (!extModule)
121  continue;
122 
123  auto defname = extModule.defnameAttr();
124  if (!defname)
125  continue;
126 
127  // Check that this extmodule's defname does not conflict with
128  // the symbol name of any module.
129  auto collidingModule = circuit.lookupSymbol(defname.getValue());
130  if (isa_and_nonnull<FModuleOp>(collidingModule)) {
131  auto diag =
132  op.emitOpError()
133  << "attribute 'defname' with value " << defname
134  << " conflicts with the name of another module in the circuit";
135  diag.attachNote(collidingModule->getLoc())
136  << "previous module declared here";
137  return failure();
138  }
139 
140  // Find an optional extmodule with a defname collision. Update
141  // the defnameMap if this is the first extmodule with that
142  // defname or if the current extmodule takes no parameters and
143  // the collision does. The latter condition improves later
144  // extmodule verification as checking against a parameterless
145  // module is stricter.
146  FExtModuleOp collidingExtModule;
147  if (auto &value = defnameMap[defname]) {
148  collidingExtModule = value;
149  if (value.parameters() && !extModule.parameters())
150  value = extModule;
151  } else {
152  value = extModule;
153  // Go to the next extmodule if no extmodule with the same
154  // defname was found.
155  continue;
156  }
157 
158  // Check that the number of ports is exactly the same.
159  SmallVector<ModulePortInfo, 8> ports;
160  SmallVector<ModulePortInfo, 8> collidingPorts;
161  extModule.getPortInfo(ports);
162  collidingExtModule.getPortInfo(collidingPorts);
163  if (ports.size() != collidingPorts.size()) {
164  auto diag = op.emitOpError()
165  << "with 'defname' attribute " << defname << " has "
166  << ports.size()
167  << " ports which is different from a previously defined "
168  "extmodule with the same 'defname' which has "
169  << collidingPorts.size() << " ports";
170  diag.attachNote(collidingExtModule.getLoc())
171  << "previous extmodule definition occurred here";
172  return failure();
173  }
174 
175  // Check that ports match for name and type. Since parameters
176  // *might* affect widths, ignore widths if either module has
177  // parameters. Note that this allows for misdetections, but
178  // has zero false positives.
179  for (auto p : llvm::zip(ports, collidingPorts)) {
180  StringAttr aName = std::get<0>(p).name, bName = std::get<1>(p).name;
181  FIRRTLType aType = std::get<0>(p).type, bType = std::get<1>(p).type;
182 
183  if (extModule.parameters() || collidingExtModule.parameters()) {
184  aType = aType.getWidthlessType();
185  bType = bType.getWidthlessType();
186  }
187  if (aName != bName) {
188  auto diag = op.emitOpError()
189  << "with 'defname' attribute " << defname
190  << " has a port with name " << aName
191  << " which does not match the name of the port "
192  << "in the same position of a previously defined "
193  << "extmodule with the same 'defname', expected port "
194  "to have name "
195  << bName;
196  diag.attachNote(collidingExtModule.getLoc())
197  << "previous extmodule definition occurred here";
198  return failure();
199  }
200  if (aType != bType) {
201  auto diag = op.emitOpError()
202  << "with 'defname' attribute " << defname
203  << " has a port with name " << aName
204  << " which has a different type " << aType
205  << " which does not match the type of the port in "
206  "the same position of a previously defined "
207  "extmodule with the same 'defname', expected port "
208  "to have type "
209  << bType;
210  diag.attachNote(collidingExtModule.getLoc())
211  << "previous extmodule definition occurred here";
212  return failure();
213  }
214  }
215  }
216 
217  return success();
218 }

246  StringAttr name, ArrayRef<ModulePortInfo> ports) {
247  using namespace mlir::impl;
248 
249  // Add an attribute for the name.
250  result.addAttribute(::mlir::SymbolTable::getSymbolAttrName(), name);
251 
252  SmallVector<Type, 4> argTypes;
253  for (auto elt : ports)
254  argTypes.push_back(elt.type);
255 
256  // Record the argument and result types as an attribute.
257  auto type = builder.getFunctionType(argTypes, /*resultTypes*/ {});
258  result.addAttribute(getTypeAttrName(), TypeAttr::get(type));
259 
260  // Record the names of the arguments if present.
261  SmallString<8> attrNameBuf;
262  for (size_t i = 0, e = ports.size(); i != e; ++i) {
263  if (ports[i].getName().empty())
264  continue;
265 
266  auto argAttr =
267  NamedAttribute(builder.getIdentifier("firrtl.name"), ports[i].name);
268 
269  result.addAttribute(getArgAttrName(i, attrNameBuf),
270  builder.getDictionaryAttr(argAttr));
271  }
272 
273  result.addRegion();
274 }

303  ArrayRef<Type> argTypes, bool isVariadic,
304  ArrayRef<Type> resultTypes) {
305  Region &body = op->getRegion(0);
306  bool isExternal = body.empty();
307 
308  p << '(';
309  for (unsigned i = 0, e = argTypes.size(); i < e; ++i) {
310  if (i > 0)
311  p << ", ";
312 
313  Value argumentValue;
314  if (!isExternal) {
315  argumentValue = body.front().getArgument(i);
316  p.printOperand(argumentValue);
317  p << ": ";
318  }
319 
320  p.printType(argTypes[i]);
321 
322  auto argAttrs = ::mlir::impl::getArgAttrs(op, i);
323 
324  // If the argument has the firrtl.name attribute, and if it was used by
325  // the printer exactly (not name mangled with a suffix etc) then we can
326  // omit the firrtl.name attribute from the argument attribute dictionary.
327  ArrayRef<StringRef> elidedAttrs;
328  StringRef tmp;
329  if (argumentValue) {
330  if (auto nameAttr = getFIRRTLNameAttr(argAttrs)) {
331 
332  // Check to make sure the asmprinter is printing it correctly.
333  SmallString<32> resultNameStr;
334  llvm::raw_svector_ostream tmpStream(resultNameStr);
335  p.printOperand(argumentValue, tmpStream);
336 
337  // If the name is the same as we would otherwise use, then we're good!
338  if (tmpStream.str().drop_front() == nameAttr.getValue()) {
339  tmp = "firrtl.name";
340  elidedAttrs = tmp;
341  }
342  }
343  }
344  p.printOptionalAttrDict(argAttrs, elidedAttrs);
345  }
346 
347  if (isVariadic) {
348  if (!argTypes.empty())
349  p << ", ";
350  p << "...";
351  }
352 
353  p << ')';
354 }

391 static ParseResult parseFModuleOp(OpAsmParser &parser, OperationState &result,
392  bool isExtModule = false) {
393  using namespace mlir::impl;
394 
395  // TODO: Should refactor mlir::impl::parseFunctionLikeOp to allow these
396  // customizations for implicit argument names. Need to not print the
397  // terminator.
398 
399  SmallVector<OpAsmParser::OperandType, 4> entryArgs;
400  SmallVector<NamedAttrList, 4> argAttrs;
401  SmallVector<NamedAttrList, 4> resultAttrs;
402  SmallVector<Type, 4> argTypes;
403  SmallVector<Type, 4> resultTypes;
404  auto &builder = parser.getBuilder();
405 
406  // Parse the name as a symbol.
407  StringAttr nameAttr;
408  if (parser.parseSymbolName(nameAttr, ::mlir::SymbolTable::getSymbolAttrName(),
409  result.attributes))
410  return failure();
411 
412  // Parse the function signature.
413  bool isVariadic = false;
414  if (parseFunctionSignature(parser, /*allowVariadic*/ false, entryArgs,
415  argTypes, argAttrs, isVariadic, resultTypes,
416  resultAttrs))
417  return failure();
418 
419  // Record the argument and result types as an attribute. This is necessary
420  // for external modules.
421  auto type = builder.getFunctionType(argTypes, resultTypes);
422  result.addAttribute(getTypeAttrName(), TypeAttr::get(type));
423 
424  // If function attributes are present, parse them.
425  if (parser.parseOptionalAttrDictWithKeyword(result.attributes))
426  return failure();
427 
428  assert(argAttrs.size() == argTypes.size());
429  assert(resultAttrs.size() == resultTypes.size());
430 
431  auto *context = result.getContext();
432 
433  // Postprocess each of the arguments. If there was no 'firrtl.name'
434  // attribute, and if the argument name was non-numeric, then add the
435  // firrtl.name attribute with the textual name from the IR. The name in the
436  // text file is a load-bearing part of the IR, but we don't want the
437  // verbosity in dumps of including it explicitly in the attribute
438  // dictionary.
439  for (size_t i = 0, e = argAttrs.size(); i != e; ++i) {
440  auto &attrs = argAttrs[i];
441 
442  // If an explicit name attribute was present, don't add the implicit one.
443  bool hasNameAttr = false;
444  for (auto &elt : attrs)
445  if (elt.first.str() == "firrtl.name")
446  hasNameAttr = true;
447  if (hasNameAttr || entryArgs.empty())
448  continue;
449 
450  auto &arg = entryArgs[i];
451 
452  // The name of an argument is of the form "%42" or "%id", and since
453  // parsing succeeded, we know it always has one character.
454  assert(arg.name.size() > 1 && arg.name[0] == '%' && "Unknown MLIR name");
455  if (isdigit(arg.name[1]))
456  continue;
457 
458  auto nameAttr = StringAttr::get(arg.name.drop_front(), context);
459  attrs.push_back({Identifier::get("firrtl.name", context), nameAttr});
460  }
461 
462  // Add the attributes to the function arguments.
463  addArgAndResultAttrs(builder, result, argAttrs, resultAttrs);
464 
465  // Parse the optional function body.
466  auto *body = result.addRegion();
467  if (!isExtModule) {
468  if (parser.parseRegion(*body, entryArgs,
469  entryArgs.empty() ? ArrayRef<Type>() : argTypes))
470  return failure();
471 
472  FModuleOp::ensureTerminator(*body, parser.getBuilder(), result.location);
473  }
474  return success();
475 }

565 /// Verify the correctness of an InstanceOp.
566 static LogicalResult verifyInstanceOp(InstanceOp instance) {
567 
568  // Check that this instance is inside a module.
569  auto module = instance->getParentOfType<FModuleOp>();
570  if (!module) {
571  instance.emitOpError("should be embedded in a 'firrtl.module'");
572  return failure();
573  }
574 
575  auto *referencedModule = instance.getReferencedModule();
576  if (!referencedModule) {
577  instance.emitOpError("invalid symbol reference");
578  return failure();
579  }
580 
581  // Check that this instance doesn't recursively instantiate its wrapping
582  // module.
583  if (referencedModule == module) {
584  auto diag = instance.emitOpError()
585  << "is a recursive instantiation of its containing module";
586  diag.attachNote(module.getLoc()) << "containing module declared here";
587  return failure();
588  }
589 
590  SmallVector<ModulePortInfo> modulePorts;
591  getModulePortInfo(referencedModule, modulePorts);
592 
593  // Check that result types are consistent with the referenced module's ports.
594  size_t numResults = instance.getNumResults();
595  if (numResults != modulePorts.size()) {
596  auto diag = instance.emitOpError()
597  << "has a wrong number of results; expected "
598  << modulePorts.size() << " but got " << numResults;
599  diag.attachNote(referencedModule->getLoc())
600  << "original module declared here";
601  return failure();
602  }
603 
604  // Check that the names array is the right length.
605  if (instance.portNames().size() != instance.getNumResults()) {
606  instance.emitOpError("incorrect number of port names");
607  return failure();
608  }
609 
610  for (size_t i = 0; i != numResults; i++) {
611  auto result = instance.getPortNamed(modulePorts[i].name);
612  if (!result) {
613  auto diag = instance.emitOpError()
614  << "is missing a port named '" << modulePorts[i].name
615  << "' expected by referenced module";
616  diag.attachNote(referencedModule->getLoc())
617  << "original module declared here";
618  return failure();
619  }
620 
621  auto resultType = result.getType();
622  auto expectedType = FlipType::get(modulePorts[i].type);
623  if (resultType != expectedType) {
624  auto diag = instance.emitOpError()
625  << "result type for " << modulePorts[i].name << " must be "
626  << expectedType << ", but got " << resultType;
627 
628  diag.attachNote(referencedModule->getLoc())
629  << "original module declared here";
630  return failure();
631  }
632  }
633 
634  return success();
635 }

637 BundleType MemOp::getTypeForPort(uint64_t depth, FIRRTLType dataType,
638  PortKind portKind) {
639 
640  auto *context = dataType.getContext();
641 
642  auto getId = [&](StringRef name) -> Identifier {
643  return Identifier::get(name, context);
644  };
645 
646  SmallVector<BundleType::BundleElement, 7> portFields;
647 
648  auto addressType =
649  UIntType::get(context, std::max(1U, llvm::Log2_64_Ceil(depth)));
650 
651  portFields.push_back({getId("addr"), addressType});
652  portFields.push_back({getId("en"), UIntType::get(context, 1)});
653  portFields.push_back({getId("clk"), ClockType::get(context)});
654 
655  switch (portKind) {
656  case PortKind::Read:
657  portFields.push_back({getId("data"), FlipType::get(dataType)});
658  break;
659 
660  case PortKind::Write:
661  portFields.push_back({getId("data"), dataType});
662  portFields.push_back({getId("mask"), dataType.getMaskType()});
663  break;
664 
665  case PortKind::ReadWrite:

773  << srcWidth;

776  // TODO(mikeurbach): verify source flow is source or duplex.

918 /// If LHS and RHS are both UInt or SInt types, the return true and fill in the
919 /// width of them if known. If unknown, return -1 for the widths.
920 ///
921 /// On failure, this reports and error and returns false. This function should
922 /// not be used if you don't want an error reported.
923 static bool isSameIntTypeKind(FIRRTLType lhs, FIRRTLType rhs, int32_t &lhsWidth,
924  int32_t &rhsWidth, Location loc) {
925  // Must have two integer types with the same signedness.
926  auto lhsi = lhs.dyn_cast<IntType>();
927  auto rhsi = rhs.dyn_cast<IntType>();
928  if (!lhsi || !rhsi || lhsi.isSigned() != rhsi.isSigned()) {
929  if (lhsi && !rhsi)
930  mlir::emitError(loc, "second operand must be an integer type, not ")
931  << rhs;
932  else if (!lhsi && rhsi)
933  mlir::emitError(loc, "first operand must be an integer type, not ")
934  << lhs;
935  else if (!lhsi && !rhsi)
936  mlir::emitError(loc, "operands must be integer types, not ")
937  << lhs << " and " << rhs;
938  else
939  mlir::emitError(loc, "operand signedness must match");
940  return false;
941  }
942 
943  lhsWidth = lhsi.getWidthOrSentinel();
944  rhsWidth = rhs.cast<IntType>().getWidthOrSentinel();
945  return true;
946 }
