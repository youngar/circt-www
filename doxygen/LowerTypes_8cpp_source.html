38 // Convert a nested bundle of fields into a flat list of fields. This is used
39 // when working with instances and mems to flatten them.
40 static void flattenBundleTypes(FIRRTLType type, StringRef suffixSoFar,
41  bool isFlipped,
42  SmallVectorImpl<FlatBundleFieldEntry> &results) {
43  if (auto flip = type.dyn_cast<FlipType>())
44  return flattenBundleTypes(flip.getElementType(), suffixSoFar, !isFlipped,
45  results);
46 
47  // In the base case we record this field.
48  auto bundle = type.dyn_cast<BundleType>();
49  if (!bundle) {
50  results.push_back({type, suffixSoFar.str(), isFlipped});
51  return;
52  }
53 
54  SmallString<16> tmpSuffix(suffixSoFar);
55 
56  // Otherwise, we have a bundle type. Break it down.
57  for (auto &elt : bundle.getElements()) {
58  // Construct the suffix to pass down.
59  tmpSuffix.resize(suffixSoFar.size());
60  tmpSuffix.push_back('_');
61  tmpSuffix.append(elt.name.strref());
62  // Recursively process subelements.
63  flattenBundleTypes(elt.type, tmpSuffix, isFlipped, results);
64  }
65 }

67 // Helper to peel off the outer most flip type from a bundle that has all flips
68 // canonicalized to the outer level, or just return the bundle directly. For any
69 // other type, returns null.
70 static BundleType getCanonicalBundleType(Type originalType) {
71  BundleType originalBundleType;
72 
73  if (auto flipType = originalType.dyn_cast<FlipType>())
74  originalBundleType = flipType.getElementType().dyn_cast<BundleType>();
75  else
76  originalBundleType = originalType.dyn_cast<BundleType>();
77 
78  return originalBundleType;
79 }

130 // This is the main entrypoint for the lowering pass.
131 void FIRRTLTypesLowering::runOnOperation() {
132  auto module = getOperation();
133  auto *body = module.getBodyBlock();
134 
135  ImplicitLocOpBuilder theBuilder(module.getLoc(), &getContext());
136  builder = &theBuilder;
137 
138  // Lower the module block arguments.
139  SmallVector<BlockArgument, 8> args(body->args_begin(), body->args_end());
140  for (auto arg : args)
141  if (auto type = arg.getType().dyn_cast<FIRRTLType>())
142  lowerArg(arg, type);
143 
144  // Lower the operations.
145  for (auto &op : body->getOperations()) {
146  builder->setInsertionPoint(&op);
147  builder->setLoc(op.getLoc());
148  dispatchVisitor(&op);
149  }
150 
151  // Remove ops that have been lowered.
152  while (!opsToRemove.empty())
153  opsToRemove.pop_back_val()->erase();
154 
155  // Remove args that have been lowered.
156  getOperation().eraseArguments(argsToRemove);
157  argsToRemove.clear();
158 
159  // Reset lowered state.
160  loweredBundleValues.clear();
161 }

167 // Lower arguments with bundle type by flattening them.
168 void FIRRTLTypesLowering::lowerArg(BlockArgument arg, FIRRTLType type) {
169  unsigned argNumber = arg.getArgNumber();
170 
171  // Flatten any bundle types.
172  SmallVector<FlatBundleFieldEntry, 8> fieldTypes;
173  flattenBundleTypes(type, "", false, fieldTypes);
174 
175  for (auto field : fieldTypes) {
176  // Create new block arguments.
177  auto type = field.getPortType();
178  auto newValue = addArg(type, argNumber, field.suffix);
179 
180  // If this field was flattened from a bundle.
181  if (!field.suffix.empty()) {
182  // Remove field separator prefix for consitency with the rest of the pass.
183  auto fieldName = StringRef(field.suffix).drop_front(1);
184 
185  // Map the flattened suffix for the original bundle to the new value.
186  setBundleLowering(arg, fieldName, newValue);
187  } else {
188  // Lower any other arguments by copying them to keep the relative order.
189  arg.replaceAllUsesWith(newValue);
190  }
191  }
192 
193  // Remember to remove the original block argument.
194  argsToRemove.push_back(argNumber);
195 }

201 // Lower instance operations in the same way as module block arguments. Bundles
202 // are flattened, and other arguments are copied to keep the relative order. By
203 // ensuring both lowerings are the same, we can process every module in the
204 // circuit in parallel, and every instance will have the correct ports.
205 void FIRRTLTypesLowering::visitDecl(InstanceOp op) {
206  // Create a new, flat bundle type for the new result
207  SmallVector<Type, 8> resultTypes;
208  SmallVector<Attribute, 8> resultNames;
209  SmallVector<size_t, 8> numFieldsPerResult;
210  for (size_t i = 0, e = op.getNumResults(); i != e; ++i) {
211  // Flatten any nested bundle types the usual way.
212  SmallVector<FlatBundleFieldEntry, 8> fieldTypes;
213  flattenBundleTypes(op.getType(i).cast<FIRRTLType>(), op.getPortNameStr(i),
214  /*isFlip*/ false, fieldTypes);
215 
216  for (auto field : fieldTypes) {
217  // Store the flat type for the new bundle type.
218  resultNames.push_back(builder->getStringAttr(field.suffix));
219  resultTypes.push_back(field.getPortType());
220  }
221  numFieldsPerResult.push_back(fieldTypes.size());
222  }
223 
224  auto newInstance = builder->create<InstanceOp>(
225  resultTypes, op.moduleNameAttr(), builder->getArrayAttr(resultNames),
226  op.nameAttr());
227 
228  // Record the mapping of each old result to each new result.
229  size_t nextResult = 0;
230  for (size_t i = 0, e = op.getNumResults(); i != e; ++i) {
231  // If this result was a non-bundle value, just RAUW it.
232  auto origPortName = op.getPortNameStr(i);
233  if (numFieldsPerResult[i] == 1 &&
234  newInstance.getPortNameStr(nextResult) == origPortName) {
235  op.getResult(i).replaceAllUsesWith(newInstance.getResult(nextResult));
236  ++nextResult;
237  continue;
238  }
239 
240  // Otherwise lower bundles.
241  for (size_t j = 0, e = numFieldsPerResult[i]; j != e; ++j) {
242  auto newPortName = newInstance.getPortNameStr(nextResult);
243  // Drop the original port name and the underscore.
244  newPortName = newPortName.drop_front(origPortName.size() + 1);
245 
246  // Map the flattened suffix for the original bundle to the new value.
247  setBundleLowering(op.getResult(i), newPortName,
248  newInstance.getResult(nextResult));
249  ++nextResult;
250  }
251  }
252 
253  // Remember to remove the original op.
254  opsToRemove.push_back(op);
255 }

257 /// Lower memory operations. A new memory is created for every leaf
258 /// element in a memory's data type.
259 void FIRRTLTypesLowering::visitDecl(MemOp op) {
260  auto type = op.getDataType();
261 
262  SmallVector<FlatBundleFieldEntry, 8> fieldTypes;
263  flattenBundleTypes(type, "", false, fieldTypes);
264 
265  // Mutable store of the types of the ports of a new memory. This is
266  // cleared and re-used.
267  SmallVector<Type, 4> resultPortTypes;
268 
269  // Store any new wires created during lowering. This ensures that
270  // wires are re-used if they already exist.
271  DenseMap<StringRef, Value> newWires;
272 
273  // Loop over the leaf aggregates.
274  for (auto field : fieldTypes) {
275 
276  // Determine the new port type for this memory. New ports are
277  // constructed by checking the kind of the memory.
278  resultPortTypes.clear();
279  for (size_t i = 0, e = op.getNumResults(); i != e; ++i) {
280  auto kind = op.getPortKind(op.getPortName(i).getValue()).getValue();
281  auto flattenedPortType =
282  FlipType::get(op.getTypeForPort(op.depth(), field.type, kind));
283 
284  resultPortTypes.push_back(flattenedPortType);
285  }
286 
287  // Construct the new memory for this flattened field.
288  auto newName = op.name().getValue().str() + field.suffix;
289  auto newMem = builder->create<MemOp>(
290  resultPortTypes, op.readLatency(), op.writeLatency(), op.depth(),
291  op.ruw(), op.portNames(), builder->getStringAttr(newName));
292 
293  // Setup the lowering to the new memory.
294  for (size_t i = 0, e = newMem.getNumResults(); i != e; ++i) {
295 
296  BundleType underlying = newMem.getResult(i)
297  .getType()
298  .cast<FIRRTLType>()
299  .getPassiveType()
300  .cast<BundleType>();
301 
302  auto kind =
303  newMem.getPortKind(newMem.getPortName(i).getValue()).getValue();
304 
305  for (auto elt : underlying.getElements()) {
306 
307  // These ports require special handling. When these are
308  // lowered, they result in multiple new connections. E.g., an
309  // assignment to a clock needs to be split into an assignment
310  // to all clocks. This is handled by creating a dummy wire,
311  // setting the dummy wire as the lowering target, and then
312  // connecting every new port subfield to that.
313  if (elt.name == "clk" | elt.name == "en" | elt.name == "addr" |
314  elt.name == "wmode") {
315  Type theType = FlipType::get(elt.type);
316 
317  // Construct a new wire if needed.
318  auto wire =
319  newWires[op.getPortName(i).getValue().str() + elt.name.str()];
320  if (!wire) {
321  wire = builder->create<WireOp>(
322  theType, op.name().getValue().str() + "_" +
323  op.getPortName(i).getValue().str() + "_" +
324  elt.name.str());
325  newWires[op.getPortName(i).getValue().str() + elt.name.str()] =
326  wire;
327  setBundleLowering(op.getResult(i), elt.name.str(), wire);
328  }
329 
330  builder->create<ConnectOp>(
331  builder->create<SubfieldOp>(theType, newMem.getResult(i),
332  elt.name),
333  wire);
334  continue;
335  }
336 
337  // Data ports ("data", "rdata", "wdata", "mask", "wmask") are
338  // trivially lowered because each data leaf winds up in a new,
339  // separate memory. No wire creation is needed.
340  FIRRTLType theType = elt.type;
341  if (kind == MemOp::PortKind::Write || elt.name == "wdata" ||
342  elt.name == "wmask")
343  theType = FlipType::get(theType);
344 
345  setBundleLowering(op.getResult(i), elt.name.str() + field.suffix,
346  builder->create<SubfieldOp>(
347  theType, newMem.getResult(i), elt.name));
348  }
349  }
350  }
351 
352  opsToRemove.push_back(op);
353 }

355 // Lowering subfield operations has to deal with three different cases:
356 // a) the input value is from a module block argument
357 // b) the input value is from another subfield operation's result
358 // c) the input value is from an instance
359 //
360 // This is accomplished by storing value and suffix mappings that point to the
361 // flattened value. If the subfield op is accessing the leaf field of a bundle,
362 // it replaces all uses with the flattened value. Otherwise, it flattens the
363 // rest of the bundle and adds the flattened values to the mapping for each
364 // partial suffix.
365 void FIRRTLTypesLowering::visitExpr(SubfieldOp op) {
366  Value input = op.input();
367  StringRef fieldname = op.fieldname();
368  FIRRTLType resultType = op.getType();
369 
370  // Flatten any nested bundle types the usual way.
371  SmallVector<FlatBundleFieldEntry, 8> fieldTypes;
372  flattenBundleTypes(resultType, fieldname, false, fieldTypes);
373 
374  for (auto field : fieldTypes) {
375  // Look up the mapping for this suffix.
376  auto newValue = getBundleLowering(input, field.suffix);
377 
378  // The prefix is the field name and possibly field separator.
379  auto prefixSize = fieldname.size();
380  if (field.suffix.size() > fieldname.size())
381  prefixSize += 1;
382 
383  // Get the remaining field suffix by removing the prefix.
384  auto partialSuffix = StringRef(field.suffix).drop_front(prefixSize);
385 
386  // If we are at the leaf of a bundle.
387  if (partialSuffix.empty())
388  // Replace the result with the flattened value.
389  op.replaceAllUsesWith(newValue);
390  else
391  // Map the partial suffix for the result value to the flattened value.
392  setBundleLowering(op, partialSuffix, newValue);
393  }
394 
395  // Remember to remove the original op.
396  opsToRemove.push_back(op);
397 }

399 // Lowering connects only has to deal with one special case: connecting two
400 // bundles. This situation should only arise when both of the arguments are a
401 // bundle that was:
402 // a) originally a block argument
403 // b) originally an instance's port
404 //
405 // When two such bundles are connected, none of the subfield visits have a
406 // chance to lower them, so we must ensure they have the same number of
407 // flattened values and flatten out this connect into multiple connects.
408 void FIRRTLTypesLowering::visitStmt(ConnectOp op) {
409  Value dest = op.dest();
410  Value src = op.src();
411 
412  // Attempt to get the bundle types, potentially unwrapping an outer flip type
413  // that wraps the whole bundle.
414  BundleType destType = getCanonicalBundleType(dest.getType());
415  BundleType srcType = getCanonicalBundleType(src.getType());
416 
417  // If we aren't connecting two bundles, there is nothing to do.
418  if (!destType || !srcType)
419  return;
420 
421  // Get the lowered values for each side.
422  SmallVector<Value, 8> destValues;
423  getAllBundleLowerings(dest, destValues);
424 
425  SmallVector<Value, 8> srcValues;
426  getAllBundleLowerings(src, srcValues);
427 
428  // Check that we got out the same number of values from each bundle.
429  assert(destValues.size() == srcValues.size() &&
430  "connected bundles don't match");
431 
432  for (auto tuple : llvm::zip_first(destValues, srcValues)) {
433  Value newDest = std::get<0>(tuple);
434  Value newSrc = std::get<1>(tuple);
435  if (newDest.getType().isa<FlipType>())
436  builder->create<ConnectOp>(newDest, newSrc);
437  else
438  builder->create<ConnectOp>(newSrc, newDest);
439  }
440 
441  // Remember to remove the original op.
442  opsToRemove.push_back(op);
443 }

505 // field.
506 void FIRRTLTypesLowering::getAllBundleLowerings(
507  Value value, SmallVectorImpl<Value> &results) {
508  // Get the original value's bundle type.
509  BundleType bundleType = getCanonicalBundleType(value.getType());
510  assert(bundleType && "attempted to get bundle lowerings for non-bundle type");
511
