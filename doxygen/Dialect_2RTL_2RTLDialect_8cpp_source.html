29 // We implement the OpAsmDialectInterface so that RTL dialect operations
30 // automatically interpret the name attribute on operations as their SSA name.
31 struct RTLOpAsmDialectInterface : public OpAsmDialectInterface {
32  using OpAsmDialectInterface::OpAsmDialectInterface;
33 
34  /// Get a special name to use when printing the given operation. See
35  /// OpAsmInterface.td#getAsmResultNames for usage details and documentation.
36  void getAsmResultNames(Operation *op,
37  OpAsmSetValueNameFn setNameFn) const override {}
38 
39  /// Get a special name to use when printing the entry block arguments of the
40  /// region contained by an operation in this dialect.
41  void getAsmBlockArgumentNames(Block *block,
42  OpAsmSetValueNameFn setNameFn) const override {
43  // Check to see if the operation containing the arguments has 'rtl.name'
44  // attributes for them. If so, use that as the name.
45  auto *parentOp = block->getParentOp();
46 
47  for (size_t i = 0, e = block->getNumArguments(); i != e; ++i) {
48  // Scan for a 'rtl.name' attribute.
49  if (auto str = getRTLNameAttr(impl::getArgAttrs(parentOp, i)))
50  setNameFn(block->getArgument(i), str.getValue());
51  }
52  }
53 };
54 } // end anonymous namespace

78 /// Registered hook to materialize a single constant operation from a given
79 /// attribute value with the desired resultant type. This method should use
80 /// the provided builder to create the operation without changing the
81 /// insertion position. The generated operation is expected to be constant
82 /// like, i.e. single result, zero operands, non side-effecting, etc. On
83 /// success, this hook should return the value generated to represent the
84 /// constant value. Otherwise, it should return null on failure.
85 Operation *RTLDialect::materializeConstant(OpBuilder &builder, Attribute value,
86  Type type, Location loc) {
87  // Integer constants.
88  if (auto intType = type.dyn_cast<IntegerType>())
89  if (auto attrValue = value.dyn_cast<IntegerAttr>())
90  return builder.create<ConstantOp>(loc, type, attrValue);
91 
92  return nullptr;
93 }
