78 /// Parse a nested type, enclosed in angle brackts (`<...>`).
79 static Type parseNestedType(DialectAsmParser &parser) {
80  Type underlyingType;
81  if (parser.parseLess())
82  return Type();
83 
84  llvm::SMLoc loc = parser.getCurrentLocation();
85  if (parser.parseType(underlyingType)) {
86  parser.emitError(loc, "No signal type found. Signal needs an underlying "
87  "type.");
88  return nullptr;
89  }
90 
91  if (parser.parseGreater())
92  return Type();
93 
94  return underlyingType;
95 }

109 /// Parse an array type.
110 /// Syntax: array ::= !llhd.array<{length}x{elementType}>
111 static Type parseArrayType(DialectAsmParser &parser) {
112  Type elementType;
113  SmallVector<int64_t, 1> length;
114  if (parser.parseLess() || parser.parseDimensionList(length, false) ||
115  parser.parseType(elementType) || parser.parseGreater()) {
116  parser.emitError(parser.getCurrentLocation(),
117  "No array type found. Array needs a length and an element "
118  "type.");
119  return nullptr;
120  }
121  if (length.size() != 1) {
122  parser.emitError(parser.getCurrentLocation(),
123  "Array must have exactly one dimension");
124  return nullptr;
125  }
126  return ArrayType::get(length[0], elementType);
127 }

129 Type LLHDDialect::parseType(DialectAsmParser &parser) const {
130  llvm::StringRef typeKeyword;
131  // parse the type keyword first
132  if (parser.parseKeyword(&typeKeyword))
133  return Type();
134  if (typeKeyword == SigType::getKeyword()) {
135  return parseSigType(parser);
136  }
137  if (typeKeyword == TimeType::getKeyword())
138  return TimeType::get(getContext());
139  if (typeKeyword == ArrayType::getKeyword())
140  return parseArrayType(parser);
141  if (typeKeyword == PtrType::getKeyword())
142  return parsePtrType(parser);
143 
144  emitError(parser.getEncodedSourceLoc(parser.getCurrentLocation()),
145  "Invalid LLHD type!");
146  return nullptr;
147 }

159 void LLHDDialect::printType(Type type, DialectAsmPrinter &printer) const {
160  if (SigType sig = type.dyn_cast<SigType>()) {
161  printSigType(sig, printer);
162  } else if (TimeType time = type.dyn_cast<TimeType>()) {
163  printer << time.getKeyword();
164  } else if (ArrayType array = type.dyn_cast<ArrayType>()) {
165  printer << array.getKeyword() << "<" << array.getLength() << "x"
166  << array.getElementType() << ">";
167  } else if (PtrType ptr = type.dyn_cast<PtrType>()) {
168  printer << ptr.getKeyword() << "<" << ptr.getUnderlyingType() << ">";
169  } else {
170  llvm_unreachable("Unknown LLHD type!");
171  }
172 }

178 /// Parse a time attribute with the custom syntax:
179 /// time ::= #llhd.time<time time_unit, delta d, epsilon e>
180 static Attribute parseTimeAttribute(DialectAsmParser &parser, Type type) {
181  if (parser.parseLess())
182  return Attribute();
183 
184  // values to parse
185  llvm::SmallVector<unsigned, 3> values;
186  llvm::StringRef timeUnit;
187  unsigned time = 0;
188  unsigned delta = 0;
189  unsigned eps = 0;
190 
191  // parse the time value
192  if (parser.parseInteger(time) || parser.parseKeyword(&timeUnit))
193  return {};
194  values.push_back(time);
195 
196  // parse the delta step value
197  if (parser.parseComma() || parser.parseInteger(delta) ||
198  parser.parseKeyword("d"))
199  return {};
200  values.push_back(delta);
201 
202  // parse the epsilon value
203  if (parser.parseComma() || parser.parseInteger(eps) ||
204  parser.parseKeyword("e") || parser.parseGreater())
205  return Attribute();
206  values.push_back(eps);
207 
208  // return a new instance of time attribute
209  return TimeAttr::get(type, values, timeUnit);
210 }

212 /// Parse an LLHD attribute
213 Attribute LLHDDialect::parseAttribute(DialectAsmParser &parser,
214  Type type) const {
215  llvm::StringRef attrKeyword;
216  // parse keyword first
217  if (parser.parseKeyword(&attrKeyword))
218  return Attribute();
219  if (attrKeyword == TimeAttr::getKeyword()) {
220  return parseTimeAttribute(parser, type);
221  }
222 
223  emitError(parser.getEncodedSourceLoc(parser.getCurrentLocation()),
224  "Invalid LLHD attribute!");
225  return nullptr;
226 }

340 struct TimeAttrStorage : public mlir::AttributeStorage {
341 public:
342  // use the ArrayRef containign the timing attributes for uniquing
343  using KeyTy = std::tuple<Type, llvm::ArrayRef<unsigned>, llvm::StringRef>;
344 
345  /// Construct a time attribute storage
346  TimeAttrStorage(Type type, llvm::ArrayRef<unsigned> timeValues,
347  llvm::StringRef timeUnit)
348  : AttributeStorage(type), timeValues(timeValues), timeUnit(timeUnit) {}
349 
350  /// Compare two istances of the time attribute. Hashing and equality are done
351  /// only on the time values and time unit. The time type is implicitly always
352  /// equal.
353  bool operator==(const KeyTy &key) const {
354  return (std::get<1>(key) == timeValues && std::get<2>(key) == timeUnit);
355  }
356 
357  /// Generate hash key for uniquing.
358  static unsigned hashKey(const KeyTy &key) {
359  auto vals = std::get<1>(key);
360  auto unit = std::get<2>(key);
361  return llvm::hash_combine(vals, unit);
362  }
363 
364  /// Construction method for llhd's time attribute
365  static TimeAttrStorage *construct(mlir::AttributeStorageAllocator &allocator,
366  const KeyTy &key) {
367  auto keyValues = std::get<1>(key);
368  auto values = allocator.copyInto(keyValues);
369  auto keyUnit = std::get<2>(key);
370  auto unit = allocator.copyInto(keyUnit);
371 
372  return new (allocator.allocate<TimeAttrStorage>())
373  TimeAttrStorage(std::get<0>(key), values, unit);
374  }
375 
376  llvm::ArrayRef<unsigned> getValue() const { return timeValues; }
377 
378  unsigned getTime() { return timeValues[0]; }
379 
380  unsigned getDelta() { return timeValues[1]; }
381 
382  unsigned getEps() { return timeValues[2]; }
383 
384  llvm::StringRef getTimeUnit() { return timeUnit; }
385 
386 private:
387  llvm::ArrayRef<unsigned> timeValues;
388  llvm::StringRef timeUnit;
389 };

456 LogicalResult
457 TimeAttr::verifyConstructionInvariants(Location loc, Type type,
458  llvm::ArrayRef<unsigned> timeValues,
459  llvm::StringRef timeUnit) {
460  // Check the attribute type is of TimeType.
461  if (!type.isa<TimeType>())
462  return emitError(loc) << "Time attribute type has to be TimeType, but got "
463  << type;
464 
465  // Check the time unit is a legal SI unit
466  std::vector<std::string> legalUnits{"ys", "zs", "as", "fs", "ps",
467  "ns", "us", "ms", "s"};
468  if (std::find(legalUnits.begin(), legalUnits.end(), timeUnit) ==
469  legalUnits.end())
470  return emitError(loc) << "Illegal time unit.";
471 
472  // Check there are exactly 3 time values
473  if (timeValues.size() != 3)
474  return emitError(loc) << "Got a wrong number of time values. Expected "
475  "exactly 3, but got "
476  << timeValues.size();
477 
478  // Check the time values are positive or zero integers.
479  if (timeValues[0] < 0 || timeValues[1] < 0 || timeValues[2] < 0)
480  return emitError(loc) << "Received a negative time value.";
481 
482  return success();
483 }
