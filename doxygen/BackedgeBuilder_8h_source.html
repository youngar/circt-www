31 /// Instantiate one of these and use it to build typed backedges. Backedges
32 /// which get used as operands must be assigned to with the actual value before
33 /// this class is destructed, usually at the end of a scope. It will check that
34 /// invariant then erase all the backedge ops during destruction.
35 ///
36 /// Example use:
37 /// ```
38 /// circt::BackedgeBuilder back(rewriter, loc);
39 /// circt::Backedge ready = back.get(rewriter.getI1Type());
40 /// // Use `ready` as a `Value`.
41 /// auto addOp = rewriter.create<addOp>(loc, ready);
42 /// // When the actual value is available,
43 /// ready.set(anotherOp.getResult(0));
44 /// ```
45 class BackedgeBuilder {
46  friend class Backedge;
47 
48 public:
49  /// To build a backedge op and manipulate it, we need a `PatternRewriter` and
50  /// a `Location`. Store them during construct of this instance and use them
51  /// when building.
52  BackedgeBuilder(mlir::PatternRewriter &rewriter, mlir::Location loc);
53  ~BackedgeBuilder();
54  /// Create a typed backedge.
55  Backedge get(mlir::Type resultType);
56 
57 private:
58  mlir::PatternRewriter &rewriter;
59  mlir::Location loc;
60  llvm::SmallVector<mlir::Operation *, 16> edges;
61 };
