118 
119  /// Insert a scheduled process wakeup.

124  llvm::SmallVector<std::pair<unsigned, unsigned>, 32> changes;

127  // The number of used change buffers in the slot.
128  size_t changesSize = 0;
129 
130  // Processes with scheduled wakeup.
