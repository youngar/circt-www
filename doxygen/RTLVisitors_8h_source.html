22 /// This helps visit Combinatorial nodes.
23 template <typename ConcreteType, typename ResultType = void,
24  typename... ExtraArgs>
25 class CombinatorialVisitor {
26 public:
27  ResultType dispatchCombinatorialVisitor(Operation *op, ExtraArgs... args) {
28  auto *thisCast = static_cast<ConcreteType *>(this);
29  return TypeSwitch<Operation *, ResultType>(op)
30  .template Case<ConstantOp,
31  // Arithmetic and Logical Binary Operations.
32  AddOp, SubOp, MulOp, DivUOp, DivSOp, ModUOp, ModSOp,
33  ShlOp, ShrUOp, ShrSOp,
34  // Bitwise operations
35  AndOp, OrOp, XorOp,
36  // Comparison operations
37  ICmpOp,
38  // Reduction Operators
39  AndROp, OrROp, XorROp,
40  // Other operations.
41  SExtOp, ConcatOp, ExtractOp, MuxOp,
42  // Cast operation
43  BitcastOp,
44  // Array operations
45  ArraySliceOp, ArrayCreateOp, ArrayGetOp>(
46  [&](auto expr) -> ResultType {
47  return thisCast->visitComb(expr, args...);
48  })
49  .Default([&](auto expr) -> ResultType {
50  return thisCast->visitInvalidComb(op, args...);
51  });
52  }
53 
54  /// This callback is invoked on any non-expression operations.
55  ResultType visitInvalidComb(Operation *op, ExtraArgs... args) {
56  op->emitOpError("unknown RTL combinatorial node");
57  abort();
58  }
59 
60  /// This callback is invoked on any combinatorial operations that are not
61  /// handled by the concrete visitor.
62  ResultType visitUnhandledComb(Operation *op, ExtraArgs... args) {
63  return ResultType();
64  }
65 
66  /// This fallback is invoked on any binary node that isn't explicitly handled.
67  /// The default implementation delegates to the 'unhandled' fallback.
68  ResultType visitBinaryComb(Operation *op, ExtraArgs... args) {
69  return static_cast<ConcreteType *>(this)->visitUnhandledComb(op, args...);
70  }
71 
72  ResultType visitUnaryComb(Operation *op, ExtraArgs... args) {
73  return static_cast<ConcreteType *>(this)->visitUnhandledComb(op, args...);
74  }
75 
76  ResultType visitVariadicComb(Operation *op, ExtraArgs... args) {
77  return static_cast<ConcreteType *>(this)->visitUnhandledComb(op, args...);
78  }
79 
80 #define HANDLE(OPTYPE, OPKIND) \
81  ResultType visitComb(OPTYPE op, ExtraArgs... args) { \
82  return static_cast<ConcreteType *>(this)->visit##OPKIND##Comb(op, \
83  args...); \
84  }
85 
86  // Basic nodes.
87  HANDLE(ConstantOp, Unhandled);
88 
89  // Arithmetic and Logical Binary Operations.
90  HANDLE(AddOp, Binary);
91  HANDLE(SubOp, Binary);
92  HANDLE(MulOp, Binary);
93  HANDLE(DivUOp, Binary);
94  HANDLE(DivSOp, Binary);
95  HANDLE(ModUOp, Binary);
96  HANDLE(ModSOp, Binary);
97  HANDLE(ShlOp, Binary);
98  HANDLE(ShrUOp, Binary);
99  HANDLE(ShrSOp, Binary);
100 
101  HANDLE(AndOp, Variadic);
102  HANDLE(OrOp, Variadic);
103  HANDLE(XorOp, Variadic);
104 
105  HANDLE(AndROp, Unary);
106  HANDLE(OrROp, Unary);
107  HANDLE(XorROp, Unary);
108 
109  HANDLE(ICmpOp, Binary);
110 
111  // Other operations.
112  HANDLE(SExtOp, Unhandled);
113  HANDLE(ConcatOp, Unhandled);
114  HANDLE(ExtractOp, Unhandled);
115  HANDLE(MuxOp, Unhandled);
116  HANDLE(BitcastOp, Unary);
117  HANDLE(ArraySliceOp, Unhandled);
118  HANDLE(ArrayGetOp, Unhandled);
119  HANDLE(ArrayCreateOp, Unhandled);
120 #undef HANDLE
121 };

123 /// This helps visit Combinatorial nodes.
124 template <typename ConcreteType, typename ResultType = void,
125  typename... ExtraArgs>
126 class StmtVisitor {
127 public:
128  ResultType dispatchStmtVisitor(Operation *op, ExtraArgs... args) {
129  auto *thisCast = static_cast<ConcreteType *>(this);
130  return TypeSwitch<Operation *, ResultType>(op)
131  .template Case<OutputOp, InstanceOp>([&](auto expr) -> ResultType {
132  return thisCast->visitStmt(expr, args...);
133  })
134  .Default([&](auto expr) -> ResultType {
135  return thisCast->visitInvalidStmt(op, args...);
136  });
137  }
138 
139  /// This callback is invoked on any non-expression operations.
140  ResultType visitInvalidStmt(Operation *op, ExtraArgs... args) {
141  op->emitOpError("unknown RTL combinatorial node");
142  abort();
143  }
144 
145  /// This callback is invoked on any combinatorial operations that are not
146  /// handled by the concrete visitor.
147  ResultType visitUnhandledComb(Operation *op, ExtraArgs... args) {
148  return ResultType();
149  }
150 
151  /// This fallback is invoked on any binary node that isn't explicitly handled.
152  /// The default implementation delegates to the 'unhandled' fallback.
153  ResultType visitBinaryComb(Operation *op, ExtraArgs... args) {
154  return static_cast<ConcreteType *>(this)->visitUnhandledComb(op, args...);
155  }
156 
157  ResultType visitUnaryComb(Operation *op, ExtraArgs... args) {
158  return static_cast<ConcreteType *>(this)->visitUnhandledComb(op, args...);
159  }
160 
161 #define HANDLE(OPTYPE, OPKIND) \
162  ResultType visitStmt(OPTYPE op, ExtraArgs... args) { \
163  return static_cast<ConcreteType *>(this)->visit##OPKIND##Stmt(op, \
164  args...); \
165  }
166 
167  // Basic nodes.
168  HANDLE(OutputOp, Unhandled);
169  HANDLE(InstanceOp, Unhandled);
170 #undef HANDLE
171 };
