42 /// Get the corresponding FIRRTL type given the built-in data type. Current
43 /// supported data types are integer (signed, unsigned, and signless), index,
44 /// and none.
45 static FIRRTLType getFIRRTLType(Type type) {
46  MLIRContext *context = type.getContext();
47  return TypeSwitch<Type, FIRRTLType>(type)
48  .Case<IntegerType>([&](IntegerType integerType) -> FIRRTLType {
49  unsigned width = integerType.getWidth();
50 
51  switch (integerType.getSignedness()) {
52  case IntegerType::Signed:
53  return SIntType::get(context, width);
54  case IntegerType::Unsigned:
55  return UIntType::get(context, width);
56  // ISSUE: How to handle signless integers? Should we use the
57  // AsSIntPrimOp or AsUIntPrimOp to convert?
58  case IntegerType::Signless:
59  return UIntType::get(context, width);
60  }
61  })
62  .Case<IndexType>([&](IndexType indexType) -> FIRRTLType {
63  // Currently we consider index type as 64-bits unsigned integer.
64  unsigned width = indexType.kInternalStorageBitWidth;
65  return UIntType::get(context, width);
66  })
67  .Default([&](Type) { return FIRRTLType(); });
68 }

70 /// Return a FIRRTL bundle type (with data, valid, and ready subfields) given a
71 /// standard data type.
72 static FIRRTLType getBundleType(Type type, bool isFlip) {
73  // If the input is already converted to a bundle type elsewhere, itself will
74  // be returned after cast.
75  if (auto bundleType = type.dyn_cast<BundleType>())
76  return bundleType;
77 
78  MLIRContext *context = type.getContext();
79  using BundleElement = BundleType::BundleElement;
80  llvm::SmallVector<BundleElement, 3> elements;
81 
82  // Add valid and ready subfield to the bundle.
83  auto validId = Identifier::get("valid", context);
84  auto readyId = Identifier::get("ready", context);
85  auto signalType = UIntType::get(context, 1);
86  if (isFlip) {
87  elements.push_back(BundleElement(validId, FlipType::get(signalType)));
88  elements.push_back(BundleElement(readyId, signalType));
89  } else {
90  elements.push_back(BundleElement(validId, signalType));
91  elements.push_back(BundleElement(readyId, FlipType::get(signalType)));
92  }
93 
94  // Add data subfield to the bundle if dataType is not a null.
95  auto dataType = getFIRRTLType(type);
96  if (dataType) {
97  auto dataId = Identifier::get("data", context);
98  if (isFlip)
99  elements.push_back(BundleElement(dataId, FlipType::get(dataType)));
100  else
101  elements.push_back(BundleElement(dataId, dataType));
102  }
103 
104  return BundleType::get(elements, context);
105 }

120 static Type getHandshakeDataType(Operation *op) {
121  if (auto memOp = dyn_cast<MemoryOp>(op))
122  return memOp.getMemRefType().getElementType();
123 
124  else if (auto sinkOp = dyn_cast<SinkOp>(op)) {
125  // As SinkOp only has one argument, which at this stage is already converted
126  // to a bundled FIRRTLType, here we convert it back to a normal data type.
127  // Is there a better way to do this?
128  auto type = sinkOp.getOperand().getType().cast<BundleType>();
129 
130  if (auto dataType = type.getElementType("data")) {
131  auto intType = dataType.cast<firrtl::IntType>();
132  return IntegerType::get(type.getContext(), intType.getWidthOrSentinel(),
133  intType.isSigned() ? IntegerType::Signed
134  : IntegerType::Unsigned);
135  } else
136  return NoneType::get(type.getContext());
137  } else
138  return op->getResult(0).getType();
139 }

141 /// Construct a name for creating FIRRTL sub-module.
142 static std::string getSubModuleName(Operation *oldOp) {
143  // The dialect name is separated from the operation name by '.', which is not
144  // valid in SystemVerilog module names. In case this name is used in
145  // SystemVerilog output, replace '.' with '_'.
146  std::string prefix = oldOp->getName().getStringRef().str();
147  std::replace(prefix.begin(), prefix.end(), '.', '_');
148 
149  std::string subModuleName = prefix + "_" +
150  std::to_string(oldOp->getNumOperands()) + "ins_" +
151  std::to_string(oldOp->getNumResults()) + "outs";
152 
153  // Add value of the constant operation.
154  if (auto constOp = dyn_cast<handshake::ConstantOp>(oldOp)) {
155  if (auto intAttr = constOp.getValue().dyn_cast<IntegerAttr>()) {
156  auto intType = intAttr.getType();
157 
158  if (intType.isSignedInteger())
159  subModuleName += "_c" + std::to_string(intAttr.getSInt());
160  else if (intType.isUnsignedInteger())
161  subModuleName += "_c" + std::to_string(intAttr.getUInt());
162  else
163  subModuleName += "_c" + std::to_string((uint64_t)intAttr.getInt());
164  } else
165  oldOp->emitError("unsupported constant type");
166  }
167 
168  // Add operation data type. Currently we only support integer or index types.
169  // The emitted type aligns with the getFIRRTLType() method. Thus all integers
170  // other than signed integers will be emitted as unsigned.
171  auto type = getHandshakeDataType(oldOp);
172  if (type.isIntOrIndex()) {
173  if (auto indexType = type.dyn_cast<IndexType>())
174  subModuleName +=
175  "_ui" + std::to_string(indexType.kInternalStorageBitWidth);
176  else if (type.isSignedInteger())
177  subModuleName += "_si" + std::to_string(type.getIntOrFloatBitWidth());
178  else
179  subModuleName += "_ui" + std::to_string(type.getIntOrFloatBitWidth());
180 
181  } else if (type.isa<NoneType>()) {
182  auto ctrlAttr = oldOp->getAttrOfType<BoolAttr>("control");
183  if (ctrlAttr.getValue())
184  subModuleName += "_ctrl";
185  else
186  oldOp->emitError("non-control component has invalid data type");
187  } else
188  oldOp->emitError("unsupported data type");
189 
190  // Add memory ID.
191  if (auto memOp = dyn_cast<handshake::MemoryOp>(oldOp))
192  subModuleName += "_id" + std::to_string(memOp.id());
193 
194  // Add compare kind.
195  if (auto comOp = dyn_cast<mlir::CmpIOp>(oldOp))
196  subModuleName += "_" + stringifyEnum(comOp.getPredicate()).str();
197 
198  // Add buffer information.
199  if (auto bufferOp = dyn_cast<handshake::BufferOp>(oldOp)) {
200  subModuleName += "_" + bufferOp.getNumSlots().toString(10, false) + "slots";
201  if (bufferOp.isSequential())
202  subModuleName += "_seq";
203  }
204 
205  return subModuleName;
206 }

213 /// Construct a tree of 1-bit muxes to multiplex arbitrary numbers of signals
214 /// using a binary-encoded select value.
215 static Value createMuxTree(ArrayRef<Value> inputs, Value select,
216  Location insertLoc,
217  ConversionPatternRewriter &rewriter) {
218  // Variables used to control iteration and select the appropriate bit.
219  size_t numInputs = inputs.size();
220  size_t numLayers = getNumIndexBits(numInputs);
221  size_t selectIdx = 0;
222 
223  // Keep a vector of ValueRanges to represent the mux tree. Each value in the
224  // range is the output of a mux.
225  SmallVector<ArrayRef<Value>, 2> muxes;
226 
227  // Helpers for repetetive calls.
228  auto createBits = [&](Value select, size_t idx) {
229  return rewriter.create<BitsPrimOp>(insertLoc, select, idx, idx);
230  };
231 
232  auto createMux = [&](Value select, ArrayRef<Value> operands, size_t idx) {
233  return rewriter.create<MuxPrimOp>(insertLoc, operands[0].getType(), select,
234  operands[idx + 1], operands[idx]);
235  };
236 
237  // Create an op to extract the least significant select bit.
238  auto selectBit = createBits(select, selectIdx);
239 
240  // Create the first layer of muxes for the inputs.
241  SmallVector<Value, 4> initialValues;
242  for (size_t i = 0; i < numInputs - 1; i += 2)
243  initialValues.push_back(createMux(selectBit, inputs, i));
244 
245  // If the number of inputs is odd, we need to add the last input as well.
246  if (numInputs % 2)
247  initialValues.push_back(inputs[numInputs - 1]);
248 
249  muxes.push_back(initialValues);
250 
251  // Create any inner layers of muxes.
252  for (size_t layer = 1; layer < numLayers; ++layer, ++selectIdx) {
253  // Get the previous layer of muxes.
254  ArrayRef<Value> prevLayer = muxes[layer - 1];
255  size_t prevSize = prevLayer.size();
256 
257  // Create an op to extract the select bit.
258  selectBit = createBits(select, selectIdx);
259 
260  // Create this layer of muxes.
261  SmallVector<Value, 4> values;
262  for (size_t i = 0; i < prevSize - 1; i += 2)
263  values.push_back(createMux(selectBit, prevLayer, i));
264 
265  // If the number of values in the previous layer is odd, we need to add the
266  // last value as well.
267  if (prevSize % 2)
268  values.push_back(prevLayer[prevSize - 1]);
269 
270  muxes.push_back(values);
271  }
272 
273  // Get the last layer of muxes, which has a single value, and return it.
274  ArrayRef<Value> lastLayer = muxes.back();
275  assert(lastLayer.size() == 1 && "mux tree didn't result in a single value");
276  return lastLayer[0];
277 }

279 /// Construct a tree of 1-bit muxes to multiplex arbitrary numbers of signals
280 /// using a one-hot select value. Assumes select has a UIntType.
281 static Value createOneHotMuxTree(ArrayRef<Value> inputs, Value select,
282  Location insertLoc,
283  ConversionPatternRewriter &rewriter) {
284  // Confirm the select input can be a one-hot encoding for the inputs.
285  int32_t numInputs = inputs.size();
286  auto selectType = select.getType().cast<UIntType>();
287  assert(numInputs == selectType.getWidthOrSentinel() &&
288  "one-hot select can't mux inputs");
289 
290  // Start the mux tree with zero value.
291  auto inputType = inputs[0].getType().cast<FIRRTLType>();
292  auto inputWidth = inputType.getBitWidthOrSentinel();
293  auto muxValue =
294  createConstantOp(inputType, APInt(inputWidth, 0), insertLoc, rewriter);
295 
296  // Iteratively chain together muxes from the high bit to the low bit.
297  for (size_t i = numInputs; i > 0; --i) {
298  size_t inputIndex = i - 1;
299 
300  Value input = inputs[inputIndex];
301 
302  Value selectBit =
303  rewriter.create<BitsPrimOp>(insertLoc, select, inputIndex, inputIndex);
304 
305  muxValue = rewriter.create<MuxPrimOp>(insertLoc, input.getType(), selectBit,
306  input, muxValue);
307  }
308 
309  return muxValue;
310 }

312 /// Construct a decoder by dynamically shifting 1 bit by the input amount.
313 /// See http://www.imm.dtu.dk/~masca/chisel-book.pdf Section 5.2.
314 static Value createDecoder(Value input, unsigned width, Location insertLoc,
315  ConversionPatternRewriter &rewriter) {
316  auto *context = rewriter.getContext();
317 
318  // Get a type for a single unsigned bit.
319  auto bitType = UIntType::get(context, 1);
320 
321  // Create a constant of for one bit.
322  auto bit =
323  rewriter.create<firrtl::ConstantOp>(insertLoc, bitType, APInt(1, 1));
324 
325  auto resultType =
326  DShlPrimOp::getResultType(bit.getType().cast<FIRRTLType>(),
327  input.getType().cast<FIRRTLType>(), insertLoc);
328  // Shift the bit dynamically by the input amount.
329  auto shift = rewriter.create<DShlPrimOp>(insertLoc, resultType, bit, input);
330 
331  // Get a type for the result based on the explicitly specified width.
332  Type padType = UIntType::get(context, width);
333  return rewriter.create<PadPrimOp>(insertLoc, padType, shift, width);
334 }

336 /// Construct an arbiter based on a simple priority-encoding scheme. In addition
337 /// to returning the arbiter result, the index for each input is added to a
338 /// mapping for other lowerings to make use of.
339 static Value createPriorityArbiter(ArrayRef<Value> inputs, Value defaultValue,
340  DenseMap<size_t, Value> &indexMapping,
341  Location insertLoc,
342  ConversionPatternRewriter &rewriter) {
343  auto numInputs = inputs.size();
344  auto indexType = UIntType::get(rewriter.getContext(), numInputs);
345  auto priorityArb = defaultValue;
346 
347  for (size_t i = numInputs; i > 0; --i) {
348  size_t inputIndex = i - 1;
349  size_t oneHotIndex = 1 << inputIndex;
350 
351  auto constIndex = createConstantOp(indexType, APInt(numInputs, oneHotIndex),
352  insertLoc, rewriter);
353 
354  priorityArb = rewriter.create<MuxPrimOp>(
355  insertLoc, indexType, inputs[inputIndex], constIndex, priorityArb);
356 
357  indexMapping[inputIndex] = constIndex;
358  }
359 
360  return priorityArb;
361 }

363 /// Construct the logic to assign the ready outputs for ControlMergeOp and
364 /// MergeOp. The logic is identical for each output. If the fired value is
365 /// asserted, and the win value holds the output's index, that output is ready.
366 static void createMergeArgReady(ArrayRef<Value> outputs, Value fired,
367  Value winner, Value defaultValue,
368  DenseMap<size_t, Value> &indexMappings,
369  Location insertLoc,
370  ConversionPatternRewriter &rewriter) {
371  auto bitType = fired.getType();
372  auto indexType = winner.getType();
373 
374  Value winnerOrDefault = rewriter.create<MuxPrimOp>(
375  insertLoc, indexType, fired, winner, defaultValue);
376 
377  for (size_t i = 0, e = outputs.size(); i != e; ++i) {
378  auto constIndex = indexMappings[i];
379  assert(constIndex && "index mapping not found");
380 
381  Value argReadyWire = rewriter.create<EQPrimOp>(insertLoc, bitType,
382  winnerOrDefault, constIndex);
383 
384  rewriter.create<ConnectOp>(insertLoc, outputs[i], argReadyWire);
385  }

398 /// Currently we are not considering clock and registers, thus the generated
399 /// circuit is pure combinational logic. If graph cycle exists, at least one
400 /// buffer is required to be inserted for breaking the cycle, which will be
401 /// supported in the next patch.
402 static FModuleOp createTopModuleOp(handshake::FuncOp funcOp, unsigned numClocks,
403  ConversionPatternRewriter &rewriter) {
404  llvm::SmallVector<ModulePortInfo, 8> ports;
405 
406  // Add all inputs of funcOp.
407  unsigned argIndex = 0;
408  for (auto &arg : funcOp.getArguments()) {
409  auto portName = rewriter.getStringAttr("arg" + std::to_string(argIndex));
410  auto bundlePortType = getBundleType(arg.getType(), /*isFlip=*/false);
411 
412  if (!bundlePortType)
413  funcOp.emitError("Unsupported data type. Supported data types: integer "
414  "(signed, unsigned, signless), index, none.");
415 
416  ports.push_back({portName, bundlePortType});
417  ++argIndex;
418  }
419 
420  // Add all outputs of funcOp.
421  for (auto portType : funcOp.getType().getResults()) {
422  auto portName = rewriter.getStringAttr("arg" + std::to_string(argIndex));
423  auto bundlePortType = getBundleType(portType, /*isFlip=*/true);
424 
425  if (!bundlePortType)
426  funcOp.emitError("Unsupported data type. Supported data types: integer "
427  "(signed, unsigned, signless), index, none.");
428 
429  ports.push_back({portName, bundlePortType});
430  ++argIndex;
431  }
432 
433  // Add clock and reset signals.
434  if (numClocks == 1) {
435  ports.push_back(
436  {rewriter.getStringAttr("clock"), rewriter.getType<ClockType>()});
437  ports.push_back(
438  {rewriter.getStringAttr("reset"), rewriter.getType<UIntType>(1)});
439  } else if (numClocks > 1) {
440  for (unsigned i = 0; i < numClocks; ++i) {
441  auto clockName = "clock" + std::to_string(i);
442  auto resetName = "reset" + std::to_string(i);
443  ports.push_back(
444  {rewriter.getStringAttr(clockName), rewriter.getType<ClockType>()});
445  ports.push_back(
446  {rewriter.getStringAttr(resetName), rewriter.getType<UIntType>(1)});
447  }
448  }
449 
450  // Create a FIRRTL module, and inline the funcOp into it.
451  auto topModuleOp = rewriter.create<FModuleOp>(
452  funcOp.getLoc(), rewriter.getStringAttr(funcOp.getName()), ports);
453  rewriter.inlineRegionBefore(funcOp.getBody(), topModuleOp.getBody(),
454  topModuleOp.end());
455 
456  // Merge the second block (inlined from funcOp) of the top-module into the
457  // entry block.
458  auto blockIterator = topModuleOp.getBody().begin();
459  Block *entryBlock = &*blockIterator;
460  Block *secondBlock = &*(++blockIterator);
461 
462  // Replace uses of each argument of the second block with the corresponding
463  // argument of the entry block.
464  argIndex = 0;
465  for (auto &oldArg : secondBlock->getArguments()) {
466  oldArg.replaceAllUsesWith(entryBlock->getArgument(argIndex));
467  ++argIndex;
468  }
469 
470  // Move all operations of the second block to the entry block.
471  while (!secondBlock->empty()) {
472  Operation &op = secondBlock->front();
473  op.moveBefore(entryBlock->getTerminator());
474  }
475  rewriter.eraseBlock(secondBlock);
476 
477  return topModuleOp;
478 }

497 /// All standard expressions and handshake elastic components will be converted
498 /// to a FIRRTL sub-module and be instantiated in the top-module.
499 static FModuleOp createSubModuleOp(FModuleOp topModuleOp, Operation *oldOp,
500  bool hasClock,
501  ConversionPatternRewriter &rewriter) {
502  rewriter.setInsertionPoint(topModuleOp);
503  llvm::SmallVector<ModulePortInfo, 8> ports;
504 
505  // Add all inputs of oldOp.
506  unsigned argIndex = 0;
507  for (auto portType : oldOp->getOperands().getTypes()) {
508  auto portName = rewriter.getStringAttr("arg" + std::to_string(argIndex));
509  auto bundlePortType = getBundleType(portType, /*isFlip=*/false);
510 
511  if (!bundlePortType)
512  oldOp->emitError("Unsupported data type. Supported data types: integer "
513  "(signed, unsigned, signless), index, none.");
514 
515  ports.push_back({portName, bundlePortType});
516  ++argIndex;
517  }
518 
519  // Add all outputs of oldOp.
520  for (auto portType : oldOp->getResults().getTypes()) {
521  auto portName = rewriter.getStringAttr("arg" + std::to_string(argIndex));
522  auto bundlePortType = getBundleType(portType, /*isFlip=*/true);
523 
524  if (!bundlePortType)
525  oldOp->emitError("Unsupported data type. Supported data types: integer "
526  "(signed, unsigned, signless), index, none.");
527 
528  ports.push_back({portName, bundlePortType});
529  ++argIndex;
530  }
531 
532  // Add clock and reset signals.
533  if (hasClock) {
534  ports.push_back(
535  {rewriter.getStringAttr("clock"), rewriter.getType<ClockType>()});
536  ports.push_back(
537  {rewriter.getStringAttr("reset"), rewriter.getType<UIntType>(1)});
538  }
539 
540  return rewriter.create<FModuleOp>(
541  topModuleOp.getLoc(), rewriter.getStringAttr(getSubModuleName(oldOp)),
542  ports);
543 }

545 /// Extract all subfields of all ports of the sub-module.
546 static ValueVectorList extractSubfields(FModuleOp subModuleOp,
547  Location insertLoc,
548  ConversionPatternRewriter &rewriter) {
549  ValueVectorList portList;
550  for (auto &arg : subModuleOp.getArguments()) {
551  ValueVector subfields;
552  if (auto argType = arg.getType().dyn_cast<BundleType>()) {
553  // Extract all subfields of all bundle ports.
554  for (auto &element : argType.getElements()) {
555  StringRef elementName = element.name.strref();
556  FIRRTLType elementType = element.type;
557  subfields.push_back(rewriter.create<SubfieldOp>(
558  insertLoc, elementType, arg, rewriter.getStringAttr(elementName)));
559  }
560  } else if (arg.getType().isa<ClockType>() ||
561  arg.getType().dyn_cast<UIntType>().getWidthOrSentinel() == 1) {
562  // Extract clock and reset signals.
563  subfields.push_back(arg);
564  }
565  portList.push_back(subfields);
566  }
567 
568  return portList;
569 }

637 template <typename OpType>
638 void StdExprBuilder::buildBinaryLogic() {
639  ValueVector arg0Subfield = portList[0];
640  ValueVector arg1Subfield = portList[1];
641  ValueVector resultSubfields = portList[2];
642 
643  Value arg0Valid = arg0Subfield[0];
644  Value arg0Ready = arg0Subfield[1];
645  Value arg0Data = arg0Subfield[2];
646  Value arg1Valid = arg1Subfield[0];
647  Value arg1Ready = arg1Subfield[1];
648  Value arg1Data = arg1Subfield[2];
649  Value resultValid = resultSubfields[0];
650  Value resultReady = resultSubfields[1];
651  Value resultData = resultSubfields[2];
652 
653  // Carry out the binary operation.
654  auto resultTy =
655  OpType::getResultType(arg0Data.getType().cast<FIRRTLType>(),
656  arg1Data.getType().cast<FIRRTLType>(), insertLoc);
657  assert(resultTy && "invalid binary operands");
658  Value resultDataOp =
659  rewriter.create<OpType>(insertLoc, resultTy, arg0Data, arg1Data);
660 
661  // Truncate the result type down if needed.
662  auto resultWidth = resultData.getType()
663  .cast<FIRRTLType>()
664  .getPassiveType()
665  .getBitWidthOrSentinel();
666  if (resultWidth < resultTy.getBitWidthOrSentinel()) {
667  resultDataOp = rewriter.create<BitsPrimOp>(insertLoc, resultDataOp,
668  resultWidth - 1, 0);
669  }
670 
671  rewriter.create<ConnectOp>(insertLoc, resultData, resultDataOp);
672 
673  // Generate valid signal.
674  auto resultValidOp = rewriter.create<AndPrimOp>(
675  insertLoc, arg0Valid.getType(), arg0Valid, arg1Valid);
676  rewriter.create<ConnectOp>(insertLoc, resultValid, resultValidOp);
677 
678  // Generate ready signals.
679  auto argReadyOp = rewriter.create<AndPrimOp>(insertLoc, resultReady.getType(),
680  resultReady, resultValidOp);
681  rewriter.create<ConnectOp>(insertLoc, arg0Ready, argReadyOp);
682  rewriter.create<ConnectOp>(insertLoc, arg1Ready, argReadyOp);
683 }

762 bool HandshakeBuilder::buildJoinLogic(SmallVector<ValueVector *, 4> inputs,
763  ValueVector *output) {
764  if (output == nullptr)
765  return false;

767  for (auto *input : inputs)
768  if (input == nullptr)
769  return false;
770 
771  // Unpack the output subfields.

781  buildAllReadyLogic(inputs, output, tmpValid);

833 /// Lowers the MuxOp into primitive FIRRTL ops.
834 /// See http://www.cs.columbia.edu/~sedwards/papers/edwards2019compositional.pdf

836 bool HandshakeBuilder::visitHandshake(MuxOp op) {
837  ValueVector selectSubfields = portList.front();
838  Value selectValid = selectSubfields[0];
839  Value selectReady = selectSubfields[1];
840  Value selectData = selectSubfields[2];
841 
842  ValueVector resultSubfields = portList.back();
843  Value resultValid = resultSubfields[0];
844  Value resultReady = resultSubfields[1];
845  Value resultData = resultSubfields[2];
846 
847  // Walk through each arg data to collect the subfields.
848  SmallVector<Value, 4> argValid;
849  SmallVector<Value, 4> argReady;
850  SmallVector<Value, 4> argData;
851  for (unsigned i = 1, e = portList.size() - 1; i < e; ++i) {
852  ValueVector argSubfields = portList[i];
853  argValid.push_back(argSubfields[0]);
854  argReady.push_back(argSubfields[1]);
855  argData.push_back(argSubfields[2]);
856  }
857 
858  // Mux the arg data.
859  auto muxedData = createMuxTree(argData, selectData, insertLoc, rewriter);
860 
861  // Connect the selected data signal to the result data.
862  rewriter.create<ConnectOp>(insertLoc, resultData, muxedData);
863 
864  // Mux the arg valids.
865  auto muxedValid = createMuxTree(argValid, selectData, insertLoc, rewriter);
866 
867  // And that with the select valid.
868  auto muxedAndSelectValid = rewriter.create<AndPrimOp>(
869  insertLoc, muxedValid.getType(), muxedValid, selectValid);
870 
871  // Connect that to the result valid.
872  rewriter.create<ConnectOp>(insertLoc, resultValid, muxedAndSelectValid);
873 
874  // And the result valid with the result ready.
875  auto resultValidAndReady =
876  rewriter.create<AndPrimOp>(insertLoc, muxedAndSelectValid.getType(),
877  muxedAndSelectValid, resultReady);
878 
879  // Connect that to the select ready.
880  rewriter.create<ConnectOp>(insertLoc, selectReady, resultValidAndReady);
881 
882  // Create a decoder for the select data.
883  auto decodedSelect =
884  createDecoder(selectData, argData.size(), insertLoc, rewriter);
885 
886  // Walk through each arg data.
887  for (unsigned i = 0, e = argData.size(); i != e; ++i) {
888  // Select the bit for this arg.
889  auto oneHot = rewriter.create<BitsPrimOp>(insertLoc, decodedSelect, i, i);
890 
891  // And that with the result valid and ready.
892  auto oneHotAndResultValidAndReady = rewriter.create<AndPrimOp>(
893  insertLoc, oneHot.getType(), oneHot, resultValidAndReady);
894 
895  // Connect that to this arg ready.
896  rewriter.create<ConnectOp>(insertLoc, argReady[i],
897  oneHotAndResultValidAndReady);
898  }
899 
900  return true;
901 }

903 /// Please refer to test_merge.mlir test case.
904 /// Lowers the MergeOp into primitive FIRRTL ops. This is a simplification of
905 /// the ControlMergeOp lowering, since it doesn't need to wait for more than one
906 /// output to become ready.
907 bool HandshakeBuilder::visitHandshake(MergeOp op) {
908  auto *context = rewriter.getContext();
909 
910  size_t numPorts = portList.size();
911  size_t numInputs = numPorts - 1;
912 
913  // The last output has the result's ready, valid, and data signal.
914  ValueVector resultSubfields = portList[numInputs];
915  Value resultValid = resultSubfields[0];
916  Value resultReady = resultSubfields[1];
917  Value resultData = resultSubfields[2];
918 
919  // Walk through each arg data to collect the subfields.
920  SmallVector<Value, 4> argValid;
921  SmallVector<Value, 4> argReady;
922  SmallVector<Value, 4> argData;
923  for (size_t i = 0; i < numInputs; ++i) {
924  ValueVector argSubfields = portList[i];
925  argValid.push_back(argSubfields[0]);
926  argReady.push_back(argSubfields[1]);
927  argData.push_back(argSubfields[2]);
928  }
929 
930  // Define some common types and values that will be used.
931  auto bitType = UIntType::get(context, 1);
932  auto indexType = UIntType::get(context, numInputs);
933  auto noWinner =
934  createConstantOp(indexType, APInt(numInputs, 0), insertLoc, rewriter);
935 
936  // Declare wire for arbitration winner.
937  auto winName = rewriter.getStringAttr("win");
938  auto win = rewriter.create<WireOp>(insertLoc, indexType, winName);
939 
940  // Declare wires for if each output is done.
941  auto resultDoneName = rewriter.getStringAttr("resultDone");
942  auto resultDone = rewriter.create<WireOp>(insertLoc, bitType, resultDoneName);
943 
944  // Create predicates to assert if the win wire holds a valid index.
945  auto hasWinnerCondition = rewriter.create<OrRPrimOp>(insertLoc, bitType, win);
946 
947  // Create an arbiter based on a simple priority-encoding scheme to assign an
948  // index to the win wire. In the case that no input is valid, set a sentinel
949  // value to indicate no winner was chosen. The constant values are remembered
950  // in a map so they can be re-used later to assign the arg ready outputs.
951  DenseMap<size_t, Value> argIndexValues;
952  auto priorityArb = createPriorityArbiter(argValid, noWinner, argIndexValues,
953  insertLoc, rewriter);
954 
955  rewriter.create<ConnectOp>(insertLoc, win, priorityArb);
956 
957  // Create the logic to assign the result outputs. The result valid and data
958  // outputs will always be assigned. The win wire from the arbiter is used to
959  // index into a tree of muxes to select the chosen input's signal(s). The
960  // result outputs are gated on the win wire being non-zero.
961  rewriter.create<ConnectOp>(insertLoc, resultValid, hasWinnerCondition);
962 
963  auto resultDataMux = createOneHotMuxTree(argData, win, insertLoc, rewriter);
964  rewriter.create<ConnectOp>(insertLoc, resultData, resultDataMux);
965 
966  // Create the logic to set the done wires for the result. The done wire is
967  // asserted when the output is valid and ready, or the emitted register is
968  // set.
969  auto resultValidAndReady = rewriter.create<AndPrimOp>(
970  insertLoc, bitType, hasWinnerCondition, resultReady);
971 
972  rewriter.create<ConnectOp>(insertLoc, resultDone, resultValidAndReady);
973 
974  // Create the logic to assign the arg ready outputs. The logic is identical
975  // for each arg. If the fired wire is asserted, and the win wire holds an
976  // arg's index, that arg is ready.
977  createMergeArgReady(argReady, resultDone, win, noWinner, argIndexValues,
978  insertLoc, rewriter);
979 
980  return true;
981 }

983 /// Please refer to test_cmerge.mlir test case.
984 /// Lowers the ControlMergeOp into primitive FIRRTL ops.
985 /// See http://www.cs.columbia.edu/~sedwards/papers/edwards2019compositional.pdf
986 /// Section 3.4.
987 bool HandshakeBuilder::visitHandshake(ControlMergeOp op) {
988  auto *context = rewriter.getContext();
989 
990  bool isControl = op.isControl();
991  unsigned numPorts = portList.size();
992  unsigned numInputs = numPorts - 4;
993 
994  // The clock and reset signals will be used for registers.
995  Value clock = portList[numPorts - 2][0];
996  Value reset = portList[numPorts - 1][0];
997 
998  // The second to last output has the result's ready and valid signals, and
999  // possibly data signal if isControl is not set.
1000  ValueVector resultSubfields = portList[numInputs];
1001  Value resultValid = resultSubfields[0];
1002  Value resultReady = resultSubfields[1];
1003 
1004  // The last output indicates which input is active now.
1005  ValueVector controlSubfields = portList[numInputs + 1];
1006  Value controlValid = controlSubfields[0];
1007  Value controlReady = controlSubfields[1];
1008  Value controlData = controlSubfields[2];
1009 
1010  // Walk through each arg data to collect the subfields.
1011  SmallVector<Value, 4> argValid;
1012  SmallVector<Value, 4> argReady;
1013  SmallVector<Value, 4> argData;
1014  for (unsigned i = 0; i < numInputs; ++i) {
1015  ValueVector argSubfields = portList[i];
1016  argValid.push_back(argSubfields[0]);
1017  argReady.push_back(argSubfields[1]);
1018  if (!isControl)
1019  argData.push_back(argSubfields[2]);
1020  }
1021 
1022  // Define some common types and values that will be used.
1023  auto bitType = UIntType::get(context, 1);
1024  auto indexType = UIntType::get(context, numInputs);
1025  auto noWinner =
1026  createConstantOp(indexType, APInt(numInputs, 0), insertLoc, rewriter);
1027  auto falseConst = createConstantOp(bitType, APInt(1, 0), insertLoc, rewriter);
1028 
1029  // Declare register for storing arbitration winner.
1030  auto wonName = rewriter.getStringAttr("won");
1031  auto won = rewriter.create<RegResetOp>(insertLoc, indexType, clock, reset,
1032  noWinner, wonName);
1033 
1034  // Declare wire for arbitration winner.
1035  auto winName = rewriter.getStringAttr("win");
1036  auto win = rewriter.create<WireOp>(insertLoc, indexType, winName);
1037 
1038  // Declare wire for whether the circuit just fired and emitted both outputs.
1039  auto firedName = rewriter.getStringAttr("fired");
1040  auto fired = rewriter.create<WireOp>(insertLoc, bitType, firedName);
1041 
1042  // Declare registers for storing if each output has been emitted.
1043  auto resultEmittedName = rewriter.getStringAttr("resultEmitted");
1044  auto resultEmitted = rewriter.create<RegResetOp>(
1045  insertLoc, bitType, clock, reset, falseConst, resultEmittedName);
1046 
1047  auto controlEmittedName = rewriter.getStringAttr("controlEmitted");
1048  auto controlEmitted = rewriter.create<RegResetOp>(
1049  insertLoc, bitType, clock, reset, falseConst, controlEmittedName);
1050 
1051  // Declare wires for if each output is done.
1052  auto resultDoneName = rewriter.getStringAttr("resultDone");
1053  auto resultDone = rewriter.create<WireOp>(insertLoc, bitType, resultDoneName);
1054 
1055  auto controlDoneName = rewriter.getStringAttr("controlDone");
1056  auto controlDone =
1057  rewriter.create<WireOp>(insertLoc, bitType, controlDoneName);
1058 
1059  // Create predicates to assert if the win wire or won register hold a valid
1060  // index.
1061  auto hasWinnerCondition = rewriter.create<OrRPrimOp>(insertLoc, bitType, win);
1062 
1063  auto hadWinnerCondition = rewriter.create<OrRPrimOp>(insertLoc, bitType, won);
1064 
1065  // Create an arbiter based on a simple priority-encoding scheme to assign an
1066  // index to the win wire. If the won register is set, just use that. In
1067  // the case that won is not set and no input is valid, set a sentinel value to
1068  // indicate no winner was chosen. The constant values are remembered in a map
1069  // so they can be re-used later to assign the arg ready outputs.
1070  DenseMap<size_t, Value> argIndexValues;
1071  auto priorityArb = createPriorityArbiter(argValid, noWinner, argIndexValues,
1072  insertLoc, rewriter);
1073 
1074  priorityArb = rewriter.create<MuxPrimOp>(
1075  insertLoc, indexType, hadWinnerCondition, won, priorityArb);
1076 
1077  rewriter.create<ConnectOp>(insertLoc, win, priorityArb);
1078 
1079  // Create the logic to assign the result and control outputs. The result valid
1080  // output will always be assigned, and if isControl is not set, the result
1081  // data output will also be assigned. The control valid and data outputs will
1082  // always be assigned. The win wire from the arbiter is used to index into a
1083  // tree of muxes to select the chosen input's signal(s), and is fed directly
1084  // to the control output. Both the result and control valid outputs are gated
1085  // on the win wire being set to something other than the sentinel value.
1086  auto resultNotEmitted =
1087  rewriter.create<NotPrimOp>(insertLoc, bitType, resultEmitted);
1088 
1089  auto resultValidWire = rewriter.create<AndPrimOp>(
1090  insertLoc, bitType, hasWinnerCondition, resultNotEmitted);
1091  rewriter.create<ConnectOp>(insertLoc, resultValid, resultValidWire);
1092 
1093  if (!isControl) {
1094  Value resultData = resultSubfields[2];
1095  auto resultDataMux = createOneHotMuxTree(argData, win, insertLoc, rewriter);
1096  rewriter.create<ConnectOp>(insertLoc, resultData, resultDataMux);
1097  }
1098 
1099  auto controlNotEmitted =
1100  rewriter.create<NotPrimOp>(insertLoc, bitType, controlEmitted);
1101 
1102  auto controlValidWire = rewriter.create<AndPrimOp>(
1103  insertLoc, bitType, hasWinnerCondition, controlNotEmitted);
1104  rewriter.create<ConnectOp>(insertLoc, controlValid, controlValidWire);
1105 
1106  // Use the one-hot win wire to select the index to output in the control data.
1107  size_t controlOutputBits = getNumIndexBits(numInputs);
1108  auto controlOutputType = UIntType::get(context, controlOutputBits);
1109  SmallVector<Value, 8> controlOutputs;
1110  for (size_t i = 0; i < numInputs; ++i)
1111  controlOutputs.push_back(createConstantOp(
1112  controlOutputType, APInt(controlOutputBits, i), insertLoc, rewriter));
1113 
1114  auto controlOutput =
1115  createOneHotMuxTree(controlOutputs, win, insertLoc, rewriter);
1116  rewriter.create<ConnectOp>(insertLoc, controlData, controlOutput);
1117 
1118  // Create the logic to set the won register. If the fired wire is asserted, we
1119  // have finished this round and can and reset the register to the sentinel
1120  // value that indicates there is no winner. Otherwise, we need to hold the
1121  // value of the win register until we can fire.
1122  auto wonMux =
1123  rewriter.create<MuxPrimOp>(insertLoc, indexType, fired, noWinner, win);
1124  rewriter.create<ConnectOp>(insertLoc, won, wonMux);
1125 
1126  // Create the logic to set the done wires for the result and control. For both
1127  // outputs, the done wire is asserted when the output is valid and ready, or
1128  // the emitted register for that output is set.
1129  auto resultValidAndReady = rewriter.create<AndPrimOp>(
1130  insertLoc, bitType, resultValidWire, resultReady);
1131 
1132  auto resultDoneWire = rewriter.create<OrPrimOp>(
1133  insertLoc, bitType, resultEmitted, resultValidAndReady);
1134  rewriter.create<ConnectOp>(insertLoc, resultDone, resultDoneWire);
1135 
1136  auto controlValidAndReady = rewriter.create<AndPrimOp>(
1137  insertLoc, bitType, controlValidWire, controlReady);
1138 
1139  auto controlDoneWire = rewriter.create<OrPrimOp>(
1140  insertLoc, bitType, controlEmitted, controlValidAndReady);
1141  rewriter.create<ConnectOp>(insertLoc, controlDone, controlDoneWire);
1142 
1143  // Create the logic to set the fired wire. It is asserted when both result and
1144  // control are done.
1145  auto firedWire =
1146  rewriter.create<AndPrimOp>(insertLoc, bitType, resultDone, controlDone);
1147  rewriter.create<ConnectOp>(insertLoc, fired, firedWire);
1148 
1149  // Create the logic to assign the emitted registers. If the fired wire is
1150  // asserted, we have finished this round and can reset the registers to 0.
1151  // Otherwise, we need to hold the values of the done registers until we can
1152  // fire.
1153  auto resultEmittedWire = rewriter.create<MuxPrimOp>(insertLoc, bitType, fired,
1154  falseConst, resultDone);
1155  rewriter.create<ConnectOp>(insertLoc, resultEmitted, resultEmittedWire);
1156 
1157  auto controlEmittedWire = rewriter.create<MuxPrimOp>(
1158  insertLoc, bitType, fired, falseConst, controlDone);
1159  rewriter.create<ConnectOp>(insertLoc, controlEmitted, controlEmittedWire);
1160 
1161  // Create the logic to assign the arg ready outputs. The logic is identical
1162  // for each arg. If the fired wire is asserted, and the win wire holds an
1163  // arg's index, that arg is ready.
1164  createMergeArgReady(argReady, fired, win, noWinner, argIndexValues, insertLoc,
1165  rewriter);
1166 
1167  return true;
1168 }

1281 bool HandshakeBuilder::buildForkLogic(ValueVector *input,
1282  SmallVector<ValueVector *, 4> outputs,
1283  Value clock, Value reset,
1284  bool isControl) {
1285  if (input == nullptr)
1286  return false;
1287 
1288  auto argSubfields = *input;
1289  Value argValid = argSubfields[0];
1290  Value argReady = argSubfields[1];
1291 
1292  unsigned resultNum = outputs.size();
1293 
1294  // Values that are useful.
1295  auto bitType = UIntType::get(rewriter.getContext(), 1);
1296  auto falseConst = createConstantOp(bitType, APInt(1, 0), insertLoc, rewriter);
1297 
1298  // Create done wire for all results.
1299  SmallVector<Value, 4> doneWires;
1300  for (unsigned i = 0; i < resultNum; ++i) {
1301  auto doneName = rewriter.getStringAttr("done" + std::to_string(i));
1302  auto doneWire = rewriter.create<WireOp>(insertLoc, bitType, doneName);
1303  doneWires.push_back(doneWire);
1304  }
1305 
1306  // Create an AndPrimOp chain for generating the ready signal. Only if all
1307  // result ports are handshaked (done), the argument port is ready to accept
1308  // the next token.
1309  Value allDoneWire = rewriter.create<WireOp>(
1310  insertLoc, bitType, rewriter.getStringAttr("allDone"));
1311  buildReductionTree<AndPrimOp>(doneWires, allDoneWire);
1312 
1313  // Connect the allDoneWire to the input ready.
1314  rewriter.create<ConnectOp>(insertLoc, argReady, allDoneWire);
1315 
1316  // Create a notAllDoneWire for later use.
1317  auto notAllDoneWire = rewriter.create<WireOp>(
1318  insertLoc, bitType, rewriter.getStringAttr("notAllDone"));
1319  rewriter.create<ConnectOp>(
1320  insertLoc, notAllDoneWire,
1321  rewriter.create<NotPrimOp>(insertLoc, bitType, allDoneWire));
1322 
1323  // Create logic for each result port.
1324  unsigned idx = 0;
1325  for (auto doneWire : doneWires) {
1326  if (outputs[idx] == nullptr)
1327  return false;
1328 
1329  // Extract valid and ready from the current result port.
1330  auto resultSubfields = *outputs[idx];
1331  Value resultValid = resultSubfields[0];
1332  Value resultReady = resultSubfields[1];
1333 
1334  // If this is not a control component, extract data from the current result
1335  // port and connect it with the argument data.
1336  if (!isControl) {
1337  Value argData = argSubfields[2];
1338  Value resultData = resultSubfields[2];
1339  rewriter.create<ConnectOp>(insertLoc, resultData, argData);
1340  }
1341 
1342  // Create a emitted register.
1343  auto emtdName = rewriter.getStringAttr("emtd" + std::to_string(idx));
1344  auto emtdReg = rewriter.create<RegResetOp>(insertLoc, bitType, clock, reset,
1345  falseConst, emtdName);
1346 
1347  // Connect the emitted register with {doneWire && notallDoneWire}. Only if
1348  // notallDone, the emtdReg will be set to the value of doneWire. Otherwise,
1349  // all emtdRegs will be cleared to zero.
1350  auto emtd = rewriter.create<AndPrimOp>(insertLoc, bitType, doneWire,
1351  notAllDoneWire);
1352  rewriter.create<ConnectOp>(insertLoc, emtdReg, emtd);
1353 
1354  // Create a notEmtdWire for later use.
1355  auto notEmtdName = rewriter.getStringAttr("notEmtd" + std::to_string(idx));
1356  auto notEmtdWire = rewriter.create<WireOp>(insertLoc, bitType, notEmtdName);
1357  rewriter.create<ConnectOp>(
1358  insertLoc, notEmtdWire,
1359  rewriter.create<NotPrimOp>(insertLoc, bitType, emtdReg));
1360 
1361  // Create valid signal and connect to the result valid. The reason of this
1362  // AndPrimOp is each result can only be emitted once.
1363  auto valid =
1364  rewriter.create<AndPrimOp>(insertLoc, bitType, notEmtdWire, argValid);
1365  rewriter.create<ConnectOp>(insertLoc, resultValid, valid);
1366 
1367  // Create validReady wire signal, which indicates a successful handshake in
1368  // the current clock cycle.
1369  auto validReadyName =
1370  rewriter.getStringAttr("validReady" + std::to_string(idx));
1371  auto validReadyWire =
1372  rewriter.create<WireOp>(insertLoc, bitType, validReadyName);
1373  rewriter.create<ConnectOp>(
1374  insertLoc, validReadyWire,
1375  rewriter.create<AndPrimOp>(insertLoc, bitType, resultReady, valid));
1376 
1377  // Finally, we can drive the doneWire we created in the beginning with
1378  // {validReadyWire || emtdReg}, where emtdReg indicates a successful
1379  // handshake in a previous clock cycle.
1380  rewriter.create<ConnectOp>(
1381  insertLoc, doneWire,
1382  rewriter.create<OrPrimOp>(insertLoc, bitType, validReadyWire, emtdReg));
1383 
1384  // All done, move to the next result port.
1385  ++idx;
1386  }
1387  return true;
1388 }

1929 /// Create InstanceOp in the top-module. This will be called after the
1930 /// corresponding sub-module and combinational logic are created.
1931 static void createInstOp(Operation *oldOp, FModuleOp subModuleOp,
1932  FModuleOp topModuleOp, unsigned clockDomain,
1933  ConversionPatternRewriter &rewriter) {
1934  rewriter.setInsertionPointAfter(oldOp);
1935 
1936  llvm::SmallVector<Type> resultTypes;
1937  llvm::SmallVector<Attribute> resultNames;
1938 
1939  // Bundle all ports of the instance into a new flattened bundle type.
1940  SmallVector<ModulePortInfo, 8> portInfo;
1941  getModulePortInfo(subModuleOp, portInfo);
1942  for (auto &port : portInfo) {
1943  // All ports of the instance operation are flipped.
1944  resultTypes.push_back(FlipType::get(port.type));
1945  resultNames.push_back(rewriter.getStringAttr(port.getName()));
1946  }
1947 
1948  // Create a instance operation.
1949  auto instanceOp = rewriter.create<firrtl::InstanceOp>(
1950  oldOp->getLoc(), resultTypes, subModuleOp.getName(),
1951  rewriter.getArrayAttr(resultNames), rewriter.getStringAttr(""));
1952 
1953  // Connect the new created instance with its predecessors and successors in
1954  // the top-module.
1955  unsigned portIndex = 0;
1956  for (auto result : instanceOp.getResults()) {
1957  unsigned numIns = oldOp->getNumOperands();
1958  unsigned numArgs = numIns + oldOp->getNumResults();
1959 
1960  auto topArgs = topModuleOp.getBody().front().getArguments();
1961  auto firstClock = std::find_if(topArgs.begin(), topArgs.end(),
1962  [](BlockArgument &arg) -> bool {
1963  return arg.getType().isa<ClockType>();
1964  });
1965  if (portIndex < numIns) {
1966  // Connect input ports.
1967  rewriter.create<ConnectOp>(oldOp->getLoc(), result,
1968  oldOp->getOperand(portIndex));
1969  } else if (portIndex < numArgs) {
1970  // Connect output ports.
1971  Value newResult = oldOp->getResult(portIndex - numIns);
1972  newResult.replaceAllUsesWith(result);
1973  } else {
1974  // Connect clock or reset signal.
1975  auto signal = *(firstClock + 2 * clockDomain + portIndex - numArgs);
1976  rewriter.create<ConnectOp>(oldOp->getLoc(), result, signal);
1977  }
1978  ++portIndex;
1979  }
1980  rewriter.eraseOp(oldOp);
1981 }

2005 /// Process of lowering:
2006 ///
2007 /// 0) Create and go into a new FIRRTL circuit;
2008 /// 1) Create and go into a new FIRRTL top-module;
2009 /// 2) Inline Handshake FuncOp region into the FIRRTL top-module;
2010 /// 3) Traverse and convert each Standard or Handshake operation:
2011 /// i) Check if an identical sub-module exists. If so, skip to vi);
2012 /// ii) Create and go into a new FIRRTL sub-module;
2013 /// iii) Extract data (if applied), valid, and ready subfield from each port
2014 /// of the sub-module;
2015 /// iv) Build combinational logic;
2016 /// v) Exit the sub-module and go back to the top-module;
2017 /// vi) Create an new instance for the sub-module;
2018 /// vii) Connect the instance with its predecessors and successors;
2019 /// 4) Erase the Handshake FuncOp.
2020 ///
2021 /// createTopModuleOp(): 1) and 2)
2022 /// checkSubModuleOp(): 3.i)
2023 /// createSubModuleOp(): 3.ii)
2024 /// extractSubfields(): 3.iii)
2025 /// build*Logic(): 3.iv)
2026 /// createInstOp(): 3.v), 3.vi), and 3.vii)
2027 ///
2028 /// Please refer to test_addi.mlir test case.
