47 static bool isVerilogExpression(Operation *op) {
48  // Merge is an expression according to the RTL dialect, but we need it emitted
49  // as a statement with its own wire declaration.
50  if (isa<MergeOp>(op))
51  return false;
52 
53  // These are SV dialect expressions.
54  if (isa<ReadInOutOp>(op) || isa<ArrayIndexInOutOp>(op))
55  return true;
56 
57  // All RTL combinatorial logic ops and SV expression ops are Verilog
58  // expressions.
59  return isCombinatorial(op) || isExpression(op);
60 }

62 /// Return the width of the specified type in bits or -1 if it isn't
63 /// supported.
64 static int getBitWidthOrSentinel(Type type) {
65  return TypeSwitch<Type, int>(type)
66  .Case<IntegerType>([](IntegerType integerType) {
67  // Verilog doesn't support zero bit integers. We only support them in
68  // limited cases.
69  return integerType.getWidth();
70  })
71  .Case<InOutType>([](InOutType inoutType) {
72  return getBitWidthOrSentinel(inoutType.getElementType());
73  })
74  .Default([](Type) { return -1; });
75 }

77 /// Push this type's dimension into a vector.
78 static void getTypeDims(SmallVectorImpl<int64_t> &dims, Type type,
79  Location loc) {
80  if (auto inout = type.dyn_cast<rtl::InOutType>())
81  return getTypeDims(dims, inout.getElementType(), loc);
82  if (auto uarray = type.dyn_cast<rtl::UnpackedArrayType>())
83  return getTypeDims(dims, uarray.getElementType(), loc);
84  if (type.isa<InterfaceType>())
85  return;
86 
87  int width;
88  if (auto arrayType = type.dyn_cast<rtl::ArrayType>()) {
89  width = arrayType.getSize();
90  } else {
91  width = getBitWidthOrSentinel(type);
92  }
93  if (width == -1)
94  mlir::emitError(loc, "value has an unsupported verilog type ") << type;
95 
96  if (width != 1) // Width 1 is implicit.
97  dims.push_back(width);
98 
99  if (auto arrayType = type.dyn_cast<rtl::ArrayType>()) {
100  getTypeDims(dims, arrayType.getElementType(), loc);
101  }
102 }

104 /// Emit a type's packed dimensions, returning whether or not text was emitted.
105 static bool emitTypeDims(Type type, Location loc, raw_ostream &os) {
106  SmallVector<int64_t, 4> dims;
107  getTypeDims(dims, type, loc);
108 
109  bool emitted = false;
110  for (int64_t width : dims)
111  switch (width) {
112  case -1: // -1 is an invalid type.
113  os << "<<invalid type>>";
114  emitted = true;
115  return true;
116  case 1: // Width 1 is implicit.
117  assert(false && "Width 1 shouldn't be in the dim vector");
118  break;
119  case 0:
120  os << "/*Zero Width*/";
121  emitted = true;
122  break;
123  default:
124  os << '[' << (width - 1) << ":0]";
125  emitted = true;
126  break;
127  }
128  return emitted;
129 }

377 /// Add the specified name to the name table, auto-uniquing the name if
378 /// required. If the name is empty, then this creates a unique temp name.
379 ///
380 /// "valueOrOp" is typically the Value for an intermediate wire etc, but it can
381 /// also be an op for an instance, since we want the instances op uniqued and
382 /// tracked. It can also be null for things like outputs which are not tracked
383 /// in the nameTable.
384 StringRef ModuleEmitter::addName(ValueOrOp valueOrOp, StringRef name) {
385  if (name.empty())
386  name = "_T";
387 
388  // Check to see if this name is valid. The first character cannot be a
389  // number of other weird thing. If it is, start with an underscore.
390  if (!isalpha(name.front()) && name.front() != '_') {
391  SmallString<16> tmpName("_");
392  tmpName += name;
393  return addName(valueOrOp, tmpName);
394  }
395 
396  auto isValidVerilogCharacter = [](char ch) -> bool {
397  return isalpha(ch) || isdigit(ch) || ch == '_';
398  };
399 
400  // Check to see if the name consists of all-valid identifiers. If not, we
401  // need to escape them.
402  for (char ch : name) {
403  if (isValidVerilogCharacter(ch))
404  continue;
405 
406  // Otherwise, we need to escape it.
407  SmallString<16> tmpName;
408  for (char ch : name) {
409  if (isValidVerilogCharacter(ch))
410  tmpName += ch;
411  else if (ch == ' ')
412  tmpName += '_';
413  else {
414  tmpName += llvm::utohexstr((unsigned char)ch);
415  }
416  }
417  return addName(valueOrOp, tmpName);
418  }
419 
420  // Get the list of reserved words we need to avoid. We could prepopulate this
421  // into the used words cache, but it is large and immutable, so we just query
422  // it when needed.
423  auto &reservedWords = getReservedWords();
424 
425  // Check to see if this name is available - if so, use it.
426  if (!reservedWords.count(name)) {
427  auto insertResult = usedNames.insert(name);
428  if (insertResult.second) {
429  if (valueOrOp)
430  nameTable[valueOrOp] = &*insertResult.first;
431  return insertResult.first->getKey();
432  }
433  }
434 
435  // If not, we need to auto-unique it.
436  SmallVector<char, 16> nameBuffer(name.begin(), name.end());
437  nameBuffer.push_back('_');
438  auto baseSize = nameBuffer.size();
439 
440  // Try until we find something that works.
441  while (1) {
442  auto suffix = llvm::utostr(nextGeneratedNameID++);
443  nameBuffer.append(suffix.begin(), suffix.end());
444  name = StringRef(nameBuffer.data(), nameBuffer.size());
445 
446  if (!reservedWords.count(name)) {
447  auto insertResult = usedNames.insert(name);
448  if (insertResult.second) {
449  if (valueOrOp)
450  nameTable[valueOrOp] = &*insertResult.first;
451  return insertResult.first->getKey();
452  }
453  }
454 
455  // Chop off the suffix and try again.
456  nameBuffer.resize(baseSize);
457  }
458 }

460 /// Return the location information as a (potentially empty) string.
461 std::string
462 ModuleEmitter::getLocationInfoAsString(const SmallPtrSet<Operation *, 8> &ops) {
463  std::string resultStr;
464  llvm::raw_string_ostream sstr(resultStr);
465 
466  // Multiple operations may come from the same location or may not have useful
467  // location info. Unique it now.
468  SmallPtrSet<Attribute, 8> locations;
469  for (auto *op : ops) {
470  if (auto loc = op->getLoc().dyn_cast<FileLineColLoc>())
471  locations.insert(loc);
472  }
473 
474  auto printLoc = [&](FileLineColLoc loc) {
475  sstr << loc.getFilename();
476  if (auto line = loc.getLine()) {
477  sstr << ':' << line;
478  if (auto col = loc.getColumn())
479  sstr << ':' << col;
480  }
481  };
482 
483  switch (locations.size()) {
484  case 1:
485  printLoc((*locations.begin()).cast<FileLineColLoc>());
486  LLVM_FALLTHROUGH;
487  case 0:
488  return sstr.str();
489  default:
490  break;
491  }
492 
493  // Sort the entries.
494  SmallVector<FileLineColLoc, 8> locVector;
495  locVector.reserve(locations.size());
496  for (auto loc : locations)
497  locVector.push_back(loc.cast<FileLineColLoc>());
498 
499  llvm::array_pod_sort(
500  locVector.begin(), locVector.end(),
501  [](const FileLineColLoc *lhs, const FileLineColLoc *rhs) -> int {
502  if (auto fn = lhs->getFilename().compare(rhs->getFilename()))
503  return fn;
504  if (lhs->getLine() != rhs->getLine())
505  return lhs->getLine() < rhs->getLine() ? -1 : 1;
506  return lhs->getColumn() < rhs->getColumn() ? -1 : 1;
507  });
508 
509  // The entries are sorted by filename, line, col. Try to merge together
510  // entries to reduce verbosity on the column info.
511  StringRef lastFileName;
512  for (size_t i = 0, e = locVector.size(); i != e;) {
513  if (i != 0)
514  sstr << ", ";
515 
516  // Print the filename if it changed.
517  auto first = locVector[i];
518  if (first.getFilename() != lastFileName) {
519  lastFileName = first.getFilename();
520  sstr << lastFileName;
521  }
522 
523  // Scan for entires with the same file/line.
524  size_t end = i + 1;
525  while (end != e && first.getFilename() == locVector[end].getFilename() &&
526  first.getLine() == locVector[end].getLine())
527  ++end;
528 
529  // If we have one entry, print it normally.
530  if (end == i + 1) {
531  if (auto line = first.getLine()) {
532  sstr << ':' << line;
533  if (auto col = first.getColumn())
534  sstr << ':' << col;
535  }
536  ++i;
537  continue;
538  }
539 
540  // Otherwise print a brace enclosed list.
541  sstr << ':' << first.getLine() << ":{";
542  while (i != end) {
543  sstr << locVector[i++].getColumn();
544 
545  if (i != end)
546  sstr << ',';
547  }
548  sstr << '}';
549  }
550 
551  return sstr.str();
552 }

754 SubExprInfo ExprEmitter::emitBinary(Operation *op, VerilogPrecedence prec,
755  const char *syntax,
756  SubExprSignRequirement operandSignReq) {
757  auto lhsInfo = emitSubExpr(op->getOperand(0), prec, operandSignReq);
758  os << ' ' << syntax << ' ';
759 
760  // The precedence of the RHS operand must be tighter than this operator if
761  // they have a different opcode in order to handle things like "x-(a+b)".
762  // This isn't needed on the LHS, because the relevant Verilog operators are
763  // left-associative.
764  //
765  auto *rhsOperandOp = lookThroughNoopCasts(op->getOperand(1)).getDefiningOp();
766  auto rhsPrec = VerilogPrecedence(prec - 1);
767  if (rhsOperandOp && op->getName() == rhsOperandOp->getName())
768  rhsPrec = prec;
769 
770  auto rhsInfo = emitSubExpr(op->getOperand(1), rhsPrec, operandSignReq);
771 
772  // SystemVerilog 11.8.1 says that the result of a binary expression is signed
773  // only if both operands are signed.
774  SubExprSignResult signedness = IsUnsigned;
775  if (lhsInfo.signedness == IsSigned && rhsInfo.signedness == IsSigned)
776  signedness = IsSigned;
777 
778  return {prec, signedness};
779 }

803 /// Emit the specified value as a subexpression to the stream.
804 SubExprInfo ExprEmitter::emitSubExpr(Value exp,
805  VerilogPrecedence parenthesizeIfLooserThan,
806  SubExprSignRequirement signRequirement) {
807  auto *op = exp.getDefiningOp();
808  bool shouldEmitInlineExpr = op && isVerilogExpression(op);
809 
810  // Don't emit this expression inline if it has multiple uses.
811  if (shouldEmitInlineExpr && parenthesizeIfLooserThan != ForceEmitMultiUse &&
812  emitter.outOfLineExpressions.count(op))
813  shouldEmitInlineExpr = false;
814 
815  // If this is a non-expr or shouldn't be done inline, just refer to its name.
816  if (!shouldEmitInlineExpr) {
817  // All wires are declared as unsigned, so if the client needed it signed,
818  // emit a conversion.
819  if (signRequirement == RequireSigned) {
820  os << "$signed(" << emitter.getName(exp) << ')';
821  return {Symbol, IsSigned};
822  }
823 
824  os << emitter.getName(exp);
825  return {Symbol, IsUnsigned};
826  }
827 
828  unsigned subExprStartIndex = resultBuffer.size();
829 
830  // Inform the visit method about the preferred sign we want from the result.
831  // It may choose to ignore this, but some emitters can change behavior based
832  // on contextual desired sign.
833  signPreference = signRequirement;
834 
835  // Okay, this is an expression we should emit inline. Do this through our
836  // visitor.
837  auto expInfo = dispatchCombinatorialVisitor(exp.getDefiningOp());
838 
839  // Check cases where we have to insert things before the expression now that
840  // we know things about it.
841  auto addPrefix = [&](StringRef prefix) {
842  resultBuffer.insert(resultBuffer.begin() + subExprStartIndex,
843  prefix.begin(), prefix.end());
844  };
845  if (signRequirement == RequireSigned && expInfo.signedness == IsUnsigned) {
846  addPrefix("$signed(");
847  os << ')';
848  expInfo.signedness = IsSigned;
849  } else if (signRequirement == RequireUnsigned &&
850  expInfo.signedness == IsSigned) {
851  addPrefix("$unsigned(");
852  os << ')';
853  expInfo.signedness = IsUnsigned;
854  } else if (expInfo.precedence > parenthesizeIfLooserThan) {
855  // If this subexpression would bind looser than the expression it is bound
856  // into, then we need to parenthesize it. Insert the parentheses
857  // retroactively.
858  addPrefix("(");
859  os << ')';
860  // Reset the precedence to the () level.
861  expInfo.precedence = Selection;
862  }
863 
864  // Remember that we emitted this.
865  emittedExprs.insert(exp.getDefiningOp());
866  return expInfo;
867 }

1235 LogicalResult ModuleEmitter::visitSV(VerbatimOp op) {
1236  SmallPtrSet<Operation *, 8> ops;
1237  ops.insert(op);
1238 
1239  // Drop an extraneous \n off the end of the string if present.
1240  StringRef string = op.string();
1241  if (string.endswith("\n"))
1242  string = string.drop_back();
1243 
1244  // Emit each \n separated piece of the string with each piece properly
1245  // indented. The convention is to not emit the \n so
1246  // emitLocationInfoAndNewLine can do that for the last line.
1247  bool isFirst = true;
1248  indent();
1249 
1250  // Emit each line of the string at a time.
1251  while (!string.empty()) {
1252  auto lhsRhs = string.split('\n');
1253  if (isFirst)
1254  isFirst = false;
1255  else {
1256  os << '\n';
1257  indent();
1258  }
1259 
1260  StringRef line = lhsRhs.first;
1261 
1262  // Perform operand substitions as we emit the line string. We turn {{42}}
1263  // into the value of operand 42.
1264 
1265  // Scan 'line' for a substitution, emitting any non-substitution prefix,
1266  // then the mentioned operand, chopping the relevant text off 'line' and
1267  // returning true. This returns false if no substitution is found.
1268  auto emitUntilSubstitution = [&](size_t next = 0) -> bool {
1269  size_t start = 0;
1270  while (1) {
1271  next = line.find("{{", next);
1272  if (next == StringRef::npos)
1273  return false;
1274 
1275  // Check to make sure we have a number followed by }}. If not, we
1276  // ignore the {{ sequence as something that could happen in Verilog.
1277  next += 2;
1278  start = next;
1279  while (next < line.size() && isdigit(line[next]))
1280  ++next;
1281  // We need at least one digit.
1282  if (start == next)
1283  continue;
1284 
1285  // We must have a }} right after the digits.
1286  if (!line.substr(next).startswith("}}"))
1287  continue;
1288 
1289  // We must be able to decode the integer into an unsigned.
1290  unsigned operandNo = 0;
1291  if (line.drop_front(start)
1292  .take_front(next - start)
1293  .getAsInteger(10, operandNo)) {
1294  op.emitError("operand substitution too large");
1295  continue;
1296  }
1297  next += 2;
1298 
1299  if (operandNo >= op.operands().size()) {
1300  op.emitError("operand " + llvm::utostr(operandNo) + " isn't valid");
1301  continue;
1302  }
1303 
1304  // Emit any text before the substitution.
1305  os << line.take_front(start - 2);
1306 
1307  // Emit the operand.
1308  os << emitExpressionToString(op.operands()[operandNo], ops);
1309 
1310  // Forget about the part we emitted.
1311  line = line.drop_front(next);
1312  return true;
1313  }
1314  };
1315 
1316  // Emit all the substitutions.
1317  while (emitUntilSubstitution())
1318  ;
1319 
1320  // Emit any text after the last substitution.
1321  os << line;
1322  string = lhsRhs.second;
1323  }
1324 
1325  emitLocationInfoAndNewLine(ops);
1326  return success();
1327 }

1733 /// Return true if we are unable to ever inline the specified operation. This
1734 /// happens because not all Verilog expressions are composable, notably you can
1735 /// only use bit selects like x[4:6] on simple expressions, you cannot use
1736 /// expressions in the sensitivity list of always blocks, etc.
1737 static bool isExpressionUnableToInline(Operation *op) {
1738  if (auto cast = dyn_cast<BitcastOp>(op))
1739  if (!haveMatchingDims(cast.input().getType(), cast.result().getType(),
1740  op->getLoc()))

1742  return true;
1743

1745  // emitted out-of-line.
1746  for (auto user : op->getUsers()) {
1747  // Verilog bit selection is required by the standard to be:
1748  // "a vector, packed array, packed structure, parameter or concatenation".
1749  // It cannot be an arbitrary expression.
1750  if (isa<ExtractOp>(user))
1751  if (!isOkToBitSelectFrom(op->getResult(0)))
1752  return true;
1753 
1754  // Sign extend (when the operand isn't a single bit) requires a bitselect
1755  // syntactically.
1756  if (auto sext = dyn_cast<SExtOp>(user)) {
1757  auto sextOperandType = sext.getOperand().getType().cast<IntegerType>();
1758  if (sextOperandType.getWidth() != 1 &&
1759  !isOkToBitSelectFrom(op->getResult(0)))
1760  return true;
1761  }
1762  // ArraySliceOp uses its operand twice, so we want to assign it first then
1763  // use that variable in the ArraySliceOp expression.
1764  if (isa<ArraySliceOp>(user) && !isa<ConstantOp>(op))
1765  return true;
1766 
1767  // Always blocks must have a name in their sensitivity list, not an expr.
1768  if (isa<AlwaysOp>(user) || isa<AlwaysFFOp>(user))
1769  return true;

1819 void ModuleEmitter::collectNamesEmitDecls(Block &block) {
1820  // In the first pass, we fill in the symbol table, calculate the max width
1821  // of the declaration words and the max type width.
1822  size_t maxDeclNameWidth = 0, maxTypeWidth = 0;
1823 
1824  // Return the word (e.g. "wire") in Verilog to declare the specified thing.
1825  auto getVerilogDeclWord = [](Operation *op) -> StringRef {
1826  if (isa<RegOp>(op))
1827  return "reg";
1828 
1829  // Interfaces instances use the name of the declared interface.
1830  if (auto interface = dyn_cast<InterfaceInstanceOp>(op))
1831  return interface.getInterfaceType().getInterface().getValue();
1832 
1833  return "wire";
1834  };
1835 
1836  // This is information we keep track of for each wire/reg/interface
1837  // declaration we're going to emit.
1838  struct ValuesToEmitRecord {
1839  Value value;
1840  SmallString<8> typeString;
1841  };
1842  SmallVector<ValuesToEmitRecord, 16> valuesToEmit;
1843 
1844  SmallString<32> nameTmp;
1845 
1846  // Loop over all of the results of all of the ops. Anything that defines a
1847  // value needs to be noticed.
1848  for (auto &op : block) {
1849  bool isExpr = isVerilogExpression(&op);
1850 
1851  // If the op is an instance, add its name to the name table as an op.
1852  auto instance = dyn_cast<InstanceOp>(&op);
1853  if (instance)
1854  addName(ValueOrOp(instance), instance.instanceName());
1855 
1856  for (auto result : op.getResults()) {
1857  // If this is an expression emitted inline or unused, it doesn't need a
1858  // name.
1859  if (isExpr) {
1860  // If this expression is dead, or can be emitted inline, ignore it.
1861  if (result.use_empty() || isExpressionEmittedInline(&op))
1862  continue;
1863 
1864  // Remember that this expression should be emitted out of line.
1865  outOfLineExpressions.insert(&op);
1866  }
1867 
1868  // Otherwise, it must be an expression or a declaration like a
1869  // RegOp/WireOp. Remember and unique the name for this result.
1870  if (instance) {
1871  // The name for an instance result is custom.
1872  nameTmp = getName(ValueOrOp(instance)).str() + "_";
1873  unsigned resultNumber = result.getResultNumber();
1874  auto resultName = instance.getResultName(resultNumber);
1875  if (resultName)
1876  nameTmp += resultName.getValue().str();
1877  else
1878  nameTmp += std::to_string(resultNumber);
1879  addName(result, nameTmp);
1880  } else {
1881  addName(result, op.getAttrOfType<StringAttr>("name"));
1882  }

1884  // If we are emitting out-of-line expressions using inline wire decls,
1885  // don't measure or emit this wire, it will be emitted inline.
1886  if (isExpr && emitInlineWireDecls)
1887  continue;

1890  valuesToEmit.push_back(ValuesToEmitRecord{result, {}});
1891  auto &typeString = valuesToEmit.back().typeString;
1892 
1893  maxDeclNameWidth =
1894  std::max(getVerilogDeclWord(&op).size(), maxDeclNameWidth);
1895 
1896  // Convert the port's type to a string and measure it.
1897  {
1898  llvm::raw_svector_ostream stringStream(typeString);
1899  emitTypeDimWithSpaceIfNeeded(result.getType(), op.getLoc(),

1990  else

1995  if (!portInfo.empty())
1996  os << '\n';
1997

2006  hasZeroWidth |= isZeroBitType(port.type);

2009  portTypeStrings.push_back({});

2011  llvm::raw_svector_ostream stringStream(portTypeStrings.back());
2012  emitTypeDimWithSpaceIfNeeded(port.type, module.getLoc(), stringStream);
2013  }
2014 
2015  maxTypeWidth = std::max(portTypeStrings.back().size(), maxTypeWidth);
2016  }
2017 
2018  addIndent();
2019 
2020  for (size_t portIdx = 0, e = portInfo.size(); portIdx != e;) {
2021  size_t startOfLinePos = os.tell();
2022 
2023  indent();

2025  auto portType = portInfo[portIdx].type;
2026  bool isZeroWidth = false;
2027  if (hasZeroWidth) {
2028  isZeroWidth = isZeroBitType(portType);
2029  os << (isZeroWidth ? "// " : " ");
2030  }
2031

2033  switch (thisPortDirection) {
2034  case PortDirection::OUTPUT:
2035  os << "output ";
2036  break;
2037  case PortDirection::INPUT:
2038  os << (hasOutputs ? "input " : "input ");
2039  break;
2040  case PortDirection::INOUT:
2041  os << (hasOutputs ? "inout " : "inout ");
2042  break;
2043  }
2044 
2045  // Emit the type.
2046  os << portTypeStrings[portIdx];
2047  if (portTypeStrings[portIdx].size() < maxTypeWidth)
2048  os.indent(maxTypeWidth - portTypeStrings[portIdx].size());
2049

2051  if (portInfo[portIdx].isOutput())

2053  else
2054  return getName(module.getArgument(portInfo[portIdx].argNum));
2055  };

2058  os << getPortName(portIdx);
2059  printArraySubscripts(portType, os);
2060  ++portIdx;
2061

2063  // emit them in a list on the same line.
2064  while (portIdx != e && portInfo[portIdx].direction == thisPortDirection &&

2067  StringRef name = getPortName(portIdx);
2068  if (os.tell() + 2 + name.size() - startOfLinePos >

2071  preferredSourceWidth - 2)

2075  os << ", " << name;
2076  printArraySubscripts(portType, os);
2077  ++portIdx;
2078  }
2079 
2080  if (portIdx != e)

2082  else if (isZeroWidth)
2083  os << "\n );\n";
2084  else
2085  os << ");\n";
2086  os << '\n';
2087  }
2088

2093  // module.
2094  collectNamesEmitDecls(*module.getBodyBlock());
2095 
2096  // Emit the body.
2097  for (auto &op : *module.getBodyBlock()) {
2098  emitOperation(&op);
2099  }
2100 
2101  reduceIndent();
2102  os << "endmodule\n\n";
